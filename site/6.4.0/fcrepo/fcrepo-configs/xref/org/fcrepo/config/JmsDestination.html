7   package org.fcrepo.config;
8   
9   /**
10   * Indicates if a JMS topic or queue is the destination
11   *
12   * @author pwinckles
13   */
14  public enum JmsDestination {
15  
16      TOPIC("topic"),
17      QUEUE("queue");
18  
19      private final String value;
20  
21      JmsDestination(final String value) {
22          this.value = value;
23      }
24  
25      public String getValue() {
26          return value;
27      }
28  
29      public static JmsDestination fromString(final String value) {
30          for (final var destination : values()) {
31              if (destination.value.equalsIgnoreCase(value)) {
32                  return destination;
33              }
34          }
35          throw new IllegalArgumentException("Unknown JMS destination: " + value);
36      }
37  
38      @Override
39      public String toString() {
40          return value;
41      }
42  
43  }
