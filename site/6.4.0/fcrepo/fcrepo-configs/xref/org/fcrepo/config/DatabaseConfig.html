7   package org.fcrepo.config;
8   
9   import java.time.Instant;
10  import java.time.LocalDateTime;
11  import java.time.ZoneOffset;
12  import java.util.Map;
13  
14  import javax.annotation.PostConstruct;
15  import javax.sql.DataSource;
16  
17  import org.flywaydb.core.Flyway;
18  import org.slf4j.Logger;
19  import org.slf4j.LoggerFactory;
20  import org.springframework.beans.factory.annotation.Value;
21  import org.springframework.context.annotation.Bean;
22  import org.springframework.context.annotation.Configuration;
23  import org.springframework.core.convert.converter.Converter;
24  import org.springframework.core.convert.converter.ConverterRegistry;
25  import org.springframework.core.convert.support.DefaultConversionService;
26  import org.springframework.jdbc.datasource.DataSourceTransactionManager;
27  import org.springframework.transaction.PlatformTransactionManager;
28  import org.springframework.transaction.TransactionDefinition;
29  import org.springframework.transaction.annotation.EnableTransactionManagement;
30  import org.springframework.transaction.support.DefaultTransactionDefinition;
31  import org.springframework.transaction.support.TransactionTemplate;
32  
33  import com.zaxxer.hikari.HikariDataSource;
34  
35  /**
36   * @author pwinckles
37   */
38  @EnableTransactionManagement
39  @Configuration
40  public class DatabaseConfig extends BasePropsConfig {
41  
42      private static final Logger LOGGER = LoggerFactory.getLogger(DatabaseConfig.class);
43  
44      private static final String H2_FILE = "fcrepo-h2";
45  
46      @Value("${fcrepo.db.url:#{'jdbc:h2:'" +
47              " + fedoraPropsConfig.fedoraData.resolve('" + H2_FILE + "').toAbsolutePath().toString()" +
48              " + ';FILE_LOCK=SOCKET'}}")
49      private String dbUrl;
50  
51      @Value("${fcrepo.db.user:}")
52      private String dbUser;
53  
54      @Value("${fcrepo.db.password:}")
55      private String dbPassword;
56  
57      @Value("${fcrepo.db.max.pool.size:10}")
58      private Integer maxPoolSize;
59  
60      @Value("${fcrepo.db.connection.checkout.timeout:30000}")
61      private Integer checkoutTimeout;
62  
63      private static final Map<String, String> DB_DRIVER_MAP = Map.of(
64              "h2", "org.h2.Driver",
65              "postgresql", "org.postgresql.Driver",
66              "mariadb", "org.mariadb.jdbc.Driver",
67              "mysql", "com.mysql.cj.jdbc.Driver"
68      );
69  
70      @PostConstruct
71      public void setup() {
72          ((ConverterRegistry) DefaultConversionService.getSharedInstance())
73                  // Adds a converter for mapping local datetimes to instants. This is dubious and not supported
74                  // by default because you must make an assumption about the timezone
75                  .addConverter(new Converter<LocalDateTime, Instant>() {
76                      @Override
77                      public Instant convert(final LocalDateTime source) {
78                          return source.toInstant(ZoneOffset.UTC);
79                      }
80                  });
81      }
82  
83      @Bean
84      public DataSource dataSource() throws Exception {
85          final var driver = identifyDbDriver();
86  
87          LOGGER.info("JDBC URL: {}", dbUrl);
88          LOGGER.info("JDBC User: {}", dbUser);
89          LOGGER.info("JDBC Password length: {}", dbPassword == null ? 0 : dbPassword.length());
90          LOGGER.info("Using database driver: {}", driver);
91  
92          final var dataSource = new HikariDataSource();
93          dataSource.setDriverClassName(driver);
94          dataSource.setJdbcUrl(dbUrl);
95          dataSource.setUsername(dbUser);
96          dataSource.setPassword(dbPassword);
97          dataSource.setConnectionTimeout(checkoutTimeout);
98          dataSource.setMaximumPoolSize(maxPoolSize);
99  
100         flyway(dataSource);
101 
102         return dataSource;
103     }
104 
105     /**
106      * Get the database type in use
107      * @return database type from the connect url.
108      */
109     private String getDbType() {
110         final var parts = dbUrl.split(":");
111 
112         if (parts.length < 2) {
113             throw new IllegalArgumentException("Invalid DB url: " + dbUrl);
114         }
115         return parts[1].toLowerCase();
116     }
117 
118     private String identifyDbDriver() {
119         final var driver = DB_DRIVER_MAP.get(getDbType());
120 
121         if (driver == null) {
122             throw new IllegalStateException("No database driver found for: " + dbUrl);
123         }
124 
125         return driver;
126     }
127 
128     @Bean
129     public DataSourceTransactionManager txManager(final DataSource dataSource) {
130         final var txManager = new DataSourceTransactionManager();
131         txManager.setDataSource(dataSource);
132         return txManager;
133     }
134 
135     @Bean
136     public TransactionTemplate txTemplate(final PlatformTransactionManager txManager) {
137         final var txDefinition = new DefaultTransactionDefinition();
138         txDefinition.setIsolationLevel(TransactionDefinition.ISOLATION_READ_COMMITTED);
139         return new TransactionTemplate(txManager, txDefinition);
140     }
141 
142     @Bean
143     public Flyway flyway(final DataSource source) throws Exception {
144         LOGGER.debug("Instantiating a new flyway bean");
145         return FlywayFactory.create().setDataSource(source).setDatabaseType(getDbType()).getObject();
146     }
147 
148 }
