007package org.fcrepo.config;
008
009import io.micrometer.core.instrument.Meter;
010import io.micrometer.core.instrument.MeterRegistry;
011import io.micrometer.core.instrument.Metrics;
012import io.micrometer.core.instrument.binder.jvm.JvmGcMetrics;
013import io.micrometer.core.instrument.binder.jvm.JvmMemoryMetrics;
014import io.micrometer.core.instrument.binder.jvm.JvmThreadMetrics;
015import io.micrometer.core.instrument.binder.system.ProcessorMetrics;
016import io.micrometer.core.instrument.binder.system.UptimeMetrics;
017import io.micrometer.core.instrument.config.MeterFilter;
018import io.micrometer.core.instrument.distribution.DistributionStatisticConfig;
019import io.micrometer.core.instrument.simple.SimpleMeterRegistry;
020import io.micrometer.prometheus.PrometheusConfig;
021import io.micrometer.prometheus.PrometheusMeterRegistry;
022import io.prometheus.client.CollectorRegistry;
023import org.springframework.beans.factory.annotation.Value;
024import org.springframework.context.annotation.Bean;
025import org.springframework.context.annotation.Configuration;
026
027import java.time.Duration;
028
029/**
030 * @author pwinckles
031 */
032@Configuration
033public class MetricsConfig extends BasePropsConfig {
034
035    @Value("${fcrepo.metrics.enable:false}")
036    private boolean metricsEnabled;
037
038    @Bean
039    public MeterRegistry meterRegistry() {
040        final MeterRegistry registry;
041
042        if (metricsEnabled) {
043            registry = new PrometheusMeterRegistry(new PrometheusConfig() {
044                @Override
045                public Duration step() {
046                    return Duration.ofSeconds(30);
047                }
048                @Override
049                public String get(final String key) {
050                    return null;
051                }
052            });
053            // Enables distribution stats for all timer metrics
054            registry.config().meterFilter(new MeterFilter() {
055                @Override
056                public DistributionStatisticConfig configure(final Meter.Id id,
057                                                             final DistributionStatisticConfig config) {
058                    if (id.getType() == Meter.Type.TIMER) {
059                        return DistributionStatisticConfig.builder()
060                                .percentilesHistogram(true)
061                                .percentiles(0.5, 0.90, 0.99)
062                                .build().merge(config);
063                    }
064                    return config;
065                }
066            });
067            new JvmThreadMetrics().bindTo(registry);
068            new JvmGcMetrics().bindTo(registry);
069            new JvmMemoryMetrics().bindTo(registry);
070            new ProcessorMetrics().bindTo(registry);
071            new UptimeMetrics().bindTo(registry);
072        } else {
073            registry = new SimpleMeterRegistry();
074        }
075
076        Metrics.addRegistry(registry);
077
078        return registry;
079    }
080
081    @Bean
082    public CollectorRegistry collectorRegistry(final MeterRegistry meterRegistry) {
083        if (meterRegistry instanceof PrometheusMeterRegistry) {
084            return ((PrometheusMeterRegistry) meterRegistry).getPrometheusRegistry();
085        }
086        return CollectorRegistry.defaultRegistry;
087    }
088
089    /**
090     * @return whether metrics are enabled
091     */
092    public boolean isMetricsEnabled() {
093        return metricsEnabled;
094    }
095
096}
