135    @PostConstruct
136    private void postConstruct() throws IOException {
137        if (reindexThreads < 0L) {
138            reindexThreads = computeDefaultReindexThreads();
139        } else {
140            reindexThreads = checkReindexThreadLimit(reindexThreads);
141        }
142        storage = Storage.fromString(storageStr);
143        LOGGER.info("Fedora storage type: {}", storage);
144        LOGGER.info("Fedora staging: {}", fedoraOcflStaging);
145        LOGGER.info("Fedora OCFL temp: {}", ocflTemp);
146        LOGGER.info("Fedora OCFL reindexing threads: {}", reindexThreads);
147        LOGGER.info("Fedora OCFL reindexing batch size: {}", reindexBatchSize);
148        LOGGER.info("Fedora OCFL reindexing fail on error: {}", reindexFailOnError);
149        createDirectories(fedoraOcflStaging);
150        createDirectories(ocflTemp);
151
152        if (storage == Storage.OCFL_FILESYSTEM) {
153            LOGGER.info("Fedora OCFL root: {}", ocflRepoRoot);
154            createDirectories(ocflRepoRoot);
155        } else if (storage == Storage.OCFL_S3) {
156            Objects.requireNonNull(ocflS3Bucket,
157                    String.format("The property %s must be set when OCFL S3 storage is used", FCREPO_OCFL_S3_BUCKET));
158
159            LOGGER.info("Fedora AWS access key: {}", awsAccessKey);
160            LOGGER.info("Fedora AWS secret key set: {}", Objects.isNull(awsSecretKey));
161            LOGGER.info("Fedora AWS region: {}", awsRegion);
162            LOGGER.info("Fedora OCFL S3 bucket: {}", ocflS3Bucket);
163            LOGGER.info("Fedora OCFL S3 prefix: {}", ocflS3Prefix);
164        }
165        FCREPO_DIGEST_ALGORITHM = DigestAlgorithm.fromAlgorithm(FCREPO_DIGEST_ALGORITHM_VALUE);
166        // Throw error if the configured default digest is not known to fedora or is not a valid option
167        if (DigestAlgorithm.MISSING.equals(FCREPO_DIGEST_ALGORITHM) ||
168                !FCREPO_VALID_DIGEST_ALGORITHMS.contains(FCREPO_DIGEST_ALGORITHM)) {
169            throw new IllegalArgumentException(String.format("Invalid %s property configured: %s, must be one of %s",
170                    FCREPO_PERSISTENCE_ALGORITHM, FCREPO_DIGEST_ALGORITHM_VALUE,
171                    FCREPO_VALID_DIGEST_ALGORITHMS.stream().map(DigestAlgorithm::getAlgorithm)
172                            .collect(Collectors.joining(", "))));
173        }
174        LOGGER.info("Fedora OCFL digest algorithm: {}", FCREPO_DIGEST_ALGORITHM.getAlgorithm());
175    }

413    /**
414     * Check we don't create too few reindexing threads.
415     * @param threads the number of threads requested.
416     * @return higher of the requested amount or 1
417     */
418    private long checkReindexThreadLimit(final long threads) {
419       if (threads <= 0) {
420            LOGGER.warn("Can't have fewer than 1 reindexing thread, setting to 1.");
421            return 1;
422        } else {
423            return threads;
424        }
425    }
426
427    /**
428     * @return number of available processors minus 1.
429     */
430    private static long computeDefaultReindexThreads() {
431        return Math.max(availableThreads - 1, 1);
432    }
