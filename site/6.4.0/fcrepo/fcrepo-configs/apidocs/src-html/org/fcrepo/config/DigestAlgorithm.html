006package org.fcrepo.config;
007
008import static java.util.stream.Collectors.toSet;
009
010import java.util.Arrays;
011import java.util.Set;
012import java.util.stream.Stream;
013
014/**
015 * Digest Algorith enum
016 * @author cbeer
017 *
018 * Moved to its own class for Fedora 6.0.0
019 */
020public enum DigestAlgorithm {
021    SHA1("SHA", "urn:sha1", "sha-1", "sha1"),
022    SHA256("SHA-256", "urn:sha-256", "sha256"),
023    SHA512("SHA-512", "urn:sha-512", "sha512"),
024    SHA512256("SHA-512/256", "urn:sha-512/256", "sha512/256"),
025    MD5("MD5", "urn:md5"),
026    MISSING("NONE", "missing");
027
028    final private String algorithm;
029    final private String scheme;
030    final private Set<String> aliases;
031
032    DigestAlgorithm(final String alg, final String scheme, final String... aliases) {
033        this.algorithm = alg;
034        this.scheme = scheme;
035        this.aliases = Stream.concat(Arrays.stream(aliases), Stream.of(algorithm)).map(String::valueOf)
036                        .map(String::toLowerCase).collect(toSet());
037    }
038
039    /**
040     * Return the scheme associated with the provided algorithm (e.g. SHA-1 returns urn:sha1)
041     *
042     * @param alg for which scheme is requested
043     * @return scheme
044     */
045    public static String getScheme(final String alg) {
046        return Arrays.stream(values()).filter(value ->
047                value.algorithm.equalsIgnoreCase(alg) || value.algorithm.replace("-", "").equalsIgnoreCase(alg)
048        ).findFirst().orElse(MISSING).scheme;
049    }
050
051    /**
052     * Return enum value for the provided scheme (e.g. urn:sha1 returns SHA-1)
053     *
054     * @param argScheme for which enum is requested
055     * @return enum value associated with the arg scheme
056     */
057    public static DigestAlgorithm fromScheme(final String argScheme) {
058        return Arrays.stream(values()).filter(value -> value.scheme.equalsIgnoreCase(argScheme)
059        ).findFirst().orElse(MISSING);
060    }
061
062    /**
063     * Return enum value for the provided algorithm
064     *
065     * @param alg algorithm name to seek
066     * @return enum value associated with the algorithm name, or missing if not found
067     */
068    public static DigestAlgorithm fromAlgorithm(final String alg) {
069        final String seek = alg.toLowerCase();
070        return Arrays.stream(values())
071                .filter(value -> value.aliases.contains(seek))
072                .findFirst()
073                .orElse(MISSING);
074    }
075
076    /**
077     * Return true if the provided algorithm is included in this enum
078     *
079     * @param alg to test
080     * @return true if arg algorithm is supported
081     */
082    public static boolean isSupportedAlgorithm(final String alg) {
083        return !getScheme(alg).equals(MISSING.scheme);
084    }
085
086    /**
087     * @return the aliases
088     */
089    public Set<String> getAliases() {
090        return aliases;
091    }
092
093    /**
094     * @return the algorithm
095     */
096    public String getAlgorithm() {
097        return algorithm;
098    }
099}
