006package org.fcrepo.stats.api;
007
008import com.fasterxml.jackson.annotation.JsonProperty;
009import org.apache.commons.lang3.builder.ToStringBuilder;
010
011/**
012 * A data object representing a statistics result.
013 * @author dbernstein
014 */
015public class RepositoryStatsResult {
016
017    private Long resourceCount = 0l;
018
019    public void setResourceCount(final Long resourceCount) {
020        this.resourceCount = resourceCount;
021    }
022
023    @JsonProperty("resource_count")
024    public Long getResourceCount() {
025        return resourceCount;
026    }
027
028
029    @Override
030    public String toString() {
031        return ToStringBuilder.reflectionToString(this).toString();
032    }
033}
