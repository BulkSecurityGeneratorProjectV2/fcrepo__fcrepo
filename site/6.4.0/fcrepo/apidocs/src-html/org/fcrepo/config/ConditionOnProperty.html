007package org.fcrepo.config;
008
009import java.util.Objects;
010
011import org.slf4j.Logger;
012import org.slf4j.LoggerFactory;
013import org.springframework.context.annotation.ConditionContext;
014import org.springframework.context.annotation.ConfigurationCondition;
015import org.springframework.core.type.AnnotatedTypeMetadata;
016
017/**
018 * This condition enables a bean/configuration when the specified property matches the expected value
019 *
020 * Implementations must provide a no-arg constructor.
021 *
022 * @author pwinckles
023 */
024public abstract class ConditionOnProperty<T> implements ConfigurationCondition {
025
026    private static final Logger LOGGER = LoggerFactory.getLogger(ConditionOnProperty.class);
027
028    private final String name;
029    private final T expected;
030    private final T defaultValue;
031    private final Class<T> clazz;
032
033    public ConditionOnProperty(final String name, final T expected, final T defaultValue, final Class<T> clazz) {
034        this.name = name;
035        this.expected = expected;
036        this.defaultValue = defaultValue;
037        this.clazz = clazz;
038    }
039
040    @Override
041    public boolean matches(final ConditionContext context, final AnnotatedTypeMetadata metadata) {
042        LOGGER.debug("Prop {}: {}", name, context.getEnvironment().getProperty(name));
043        return Objects.equals(expected, context.getEnvironment().getProperty(name, clazz, defaultValue));
044    }
045
046    @Override
047    public ConfigurationPhase getConfigurationPhase() {
048        // This forces spring to not evaluate these conditions until after it has loaded other @Configuration classes,
049        // ensuring that the properties have been loaded.
050        return ConfigurationPhase.REGISTER_BEAN;
051    }
052}
