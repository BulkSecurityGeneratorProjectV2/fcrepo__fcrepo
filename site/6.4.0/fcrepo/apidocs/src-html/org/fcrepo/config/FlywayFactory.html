007package org.fcrepo.config;
008
009import javax.sql.DataSource;
010
011import org.flywaydb.core.Flyway;
012import org.springframework.beans.factory.FactoryBean;
013
014/**
015 * Factory to generate a Flyway instance for Fedora.
016 * @author whikloj
017 */
018public class FlywayFactory implements FactoryBean<Flyway> {
019
020    private DataSource dataSource;
021
022    private String databaseType;
023
024    /**
025     * Static constructor
026     * @return a new FlywayFactory instance.
027     */
028    public static FlywayFactory create() {
029        return new FlywayFactory();
030    }
031
032    @Override
033    public Flyway getObject() throws Exception {
034        if (dataSource == null) {
035            throw new IllegalStateException("Cannot get flyway instance without a configured datasource.");
036        }
037        final var fly = Flyway.configure().dataSource(dataSource)
038                .locations("classpath:sql/" + (databaseType == null ? "h2" : databaseType)).load();
039        fly.migrate();
040        return fly;
041    }
042
043    @Override
044    public Class<?> getObjectType() {
045        return Flyway.class;
046    }
047
048    /**
049     * Set the datasource for use with Flyway.
050     * @param source the data source.
051     * @return this factory
052     */
053    public FlywayFactory setDataSource(final DataSource source) {
054        dataSource = source;
055        return this;
056    }
057
058    /**
059     * Set the type of database to pick the correct schema files.
060     * @param type database type
061     * @return this factory
062     */
063    public FlywayFactory setDatabaseType(final String type) {
064        databaseType = type;
065        return this;
066    }
067}
