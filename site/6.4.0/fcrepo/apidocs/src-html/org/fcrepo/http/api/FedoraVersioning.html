101    /**
102     * Create a new version of a resource. If a memento-datetime header is provided, then the new version will be
103     * based off the provided body using that datetime. If one was not provided, then a version is created based off
104     * the current version of the resource.
105     *
106     * @return response
107     * @throws InvalidChecksumException thrown if one of the provided digests does not match the content
108     * @throws MementoDatetimeFormatException if the header value of memento-datetime is not RFC-1123 format
109     */
110    @POST
111    public Response addVersion() {
112
113        if (headers.getHeaderString("Slug") != null) {
114            throw new BadRequestException("Slug header is no longer supported for versioning label.");
115        }
116
117        if (headers.getHeaderString(MEMENTO_DATETIME_HEADER) != null) {
118            throw new CannotCreateMementoException(MEMENTO_DATETIME_HEADER +
119                    " header is no longer supported on versioning.");
120        }
121
122        final var transaction = transaction();
123
124        if (!transaction.isShortLived()) {
125            throw new BadRequestException("Version creation is not allowed within transactions.");
126        }
127
128        final var resource = resource();
129
130        try {
131            LOGGER.debug("Request to create version for <{}>", externalPath);
132
133            doInDbTxWithRetry(() -> {
134                versionService.createVersion(transaction, resource.getFedoraId(), getUserPrincipal());
135
136                // need to commit the transaction before loading the memento otherwise it won't exist
137                transaction.commitIfShortLived();
138            });
139
140            final var versions = reloadResource().getTimeMap().getChildren().collect(Collectors.toList());
141
142            if (versions.isEmpty()) {
143                throw new RepositoryRuntimeException(String.format("Failed to create a version for %s", externalPath));
144            }
145
146            final var memento = versions.get(versions.size() - 1);
147
148            return createUpdateResponse(memento, true);
149        } catch (final Exception e) {
150            checkForInsufficientStorageException(e, e);
151            return null; // not reachable
152        } finally {
153            transaction.releaseResourceLocksIfShortLived();
154        }
155    }

157    /**
158     * Get the list of versions for the object
159     *
160     * @param acceptValue the rdf media-type
161     * @return List of versions for the object as RDF
162     * @throws IOException in case of error extracting content
163     */
164    @GET
165    @HtmlTemplate(value = "fcr:versions")
166    @Produces({ TURTLE_WITH_CHARSET + ";qs=1.0", JSON_LD + ";qs=0.8",
167        N3_WITH_CHARSET, N3_ALT2_WITH_CHARSET, RDF_XML, NTRIPLES, TEXT_PLAIN_WITH_CHARSET,
168        TEXT_HTML_WITH_CHARSET, APPLICATION_LINK_FORMAT })
169    public Response getVersionList(@HeaderParam("Accept") final String acceptValue) throws IOException {
170
171        final FedoraResource theTimeMap = resource().getTimeMap();
172        checkCacheControlHeaders(request, servletResponse, theTimeMap, transaction());
173
174        LOGGER.debug("GET resource '{}'", externalPath());
175
176        addResourceHttpHeaders(theTimeMap);
177
178        if (acceptValue != null && acceptValue.equalsIgnoreCase(APPLICATION_LINK_FORMAT)) {
179            final String extUrl = identifierConverter().toDomain(externalPath());
180
181            final URI parentUri = URI.create(extUrl);
182            final Set<Link> versionLinks = new HashSet<>();
183            versionLinks.add(Link.fromUri(parentUri).rel("original").build());
184            versionLinks.add(Link.fromUri(parentUri).rel("timegate").build());
185            // So we don't collect the children twice, store them in an array.
186            final FedoraResource[] children = theTimeMap.getChildren().toArray(FedoraResource[]::new);
187
188            Arrays.stream(children).forEach(t -> {
189                final URI childUri = getUri(t);
190                versionLinks.add(Link.fromUri(childUri).rel("memento")
191                                     .param("datetime",
192                        MEMENTO_RFC_1123_FORMATTER.format(t.getMementoDatetime()))
193                                     .build());
194            });
195            // Based on the dates of the above mementos, add the range to the below link.
196            final Instant[] mementos = Arrays.stream(children).map(FedoraResource::getMementoDatetime)
197                .sorted(Comparator.naturalOrder())
198                .toArray(Instant[]::new);
199            final Builder linkBuilder =
200                Link.fromUri(parentUri + "/" + FCR_VERSIONS).rel("self").type(APPLICATION_LINK_FORMAT);
201            if (mementos.length >= 2) {
202                // There are 2 or more Mementos so make a range.
203                linkBuilder.param("from", MEMENTO_RFC_1123_FORMATTER.format(mementos[0].atZone(ZoneId.of("UTC"))));
204                linkBuilder.param("until",
205                    MEMENTO_RFC_1123_FORMATTER.format(mementos[mementos.length - 1].atZone(ZoneId.of("UTC"))));
206            }
207            versionLinks.add(linkBuilder.build());
208            return ok(new LinkFormatStream(versionLinks.stream())).build();
209        } else {
210            return getContent(getChildrenLimit(), theTimeMap);
211        }
212    }

232    @DELETE

234    public Response delete() {
235        final FedoraResource theTimeMap = resource().getTimeMap();
236        addResourceHttpHeaders(theTimeMap);
237        final String message = "Timemaps are deleted with their associated resource.";
238        return status(METHOD_NOT_ALLOWED).entity(message).type(TEXT_PLAIN_WITH_CHARSET).build();
239    }
