006package org.fcrepo.http.api;
007
008import static org.apache.commons.lang3.StringUtils.isEmpty;
009import static org.slf4j.LoggerFactory.getLogger;
010
011import java.io.File;
012import java.io.IOException;
013import java.net.URI;
014import java.nio.file.Files;
015import java.nio.file.Paths;
016import java.util.Arrays;
017import java.util.HashSet;
018import java.util.List;
019import java.util.Set;
020import java.util.regex.Matcher;
021import java.util.regex.Pattern;
022import java.util.stream.Collectors;
023import java.util.stream.Stream;
024
025import org.fcrepo.kernel.api.exception.ExternalMessageBodyException;
026import org.fcrepo.kernel.api.utils.AutoReloadingConfiguration;
027import org.slf4j.Logger;
028
029/**
030 * Validates external content paths to ensure that they are within a configured allowed list of paths.
031 *
032 * @author bbpennel
033 */
034public class ExternalContentPathValidator extends AutoReloadingConfiguration {
035
036    private static final Logger LOGGER = getLogger(ExternalContentPathValidator.class);
037
038    private static final Set<String> ALLOWED_SCHEMES = new HashSet<>(Arrays.asList("file", "http", "https"));
039
040    private static final Pattern SCHEME_PATTERN = Pattern.compile("^(http|https|file):/.*", Pattern.CASE_INSENSITIVE);
041
042    // Pattern to check that an http uri contains a / after the domain if a domain is present
043    private static final Pattern HTTP_DOMAIN_PATTERN = Pattern.compile("^(http|https)://([^/]+/.*|$)");
044
045    private static final Pattern RELATIVE_MOD_PATTERN = Pattern.compile(".*(^|/)\\.\\.($|/).*");
046
047    private static final Pattern NORMALIZE_FILE_URI = Pattern.compile("^file:/{2,3}");
048
049    private List<String> allowedList;
050
051    /**
052     * Validates that an external path is valid. The path must be an HTTP or file URI within the allow list of paths,
053     * be absolute, and contain no relative modifier.
054     *
055     * @param extPath external binary path to validate
056     * @throws ExternalMessageBodyException thrown if the path is invalid.
057     */
058    public void validate(final String extPath) throws ExternalMessageBodyException {
059        if (allowedList == null || allowedList.size() == 0) {
060            throw new ExternalMessageBodyException("External content is disallowed by the server");
061        }
062
063        if (isEmpty(extPath)) {
064            throw new ExternalMessageBodyException("External content path was empty");
065        }
066
067        final String path = normalizeUri(extPath);
068
069        final URI uri;
070        try {
071            // Ensure that the path is a valid URL
072            uri = new URI(path);
073            uri.toURL();
074        } catch (final Exception e) {
075            throw new ExternalMessageBodyException("Path was not a valid URI: " + extPath);
076        }
077
078        // Decode the uri and ensure that it does not contain modifiers
079        final String decodedPath = uri.getPath();
080        if (RELATIVE_MOD_PATTERN.matcher(decodedPath).matches()) {
081            throw new ExternalMessageBodyException("Path was not absolute: " + extPath);
082        }
083
084        // Require that the path is absolute
085        if (!uri.isAbsolute()) {
086            throw new ExternalMessageBodyException("Path was not absolute: " + extPath);
087        }
088
089        // Ensure that an accept scheme was provided
090        final String scheme = uri.getScheme();
091        if (!ALLOWED_SCHEMES.contains(scheme)) {
092            throw new ExternalMessageBodyException("Path did not provide an allowed scheme: " + extPath);
093        }
094
095        // If a file, verify that it exists
096        if (scheme.equals("file") && !Paths.get(uri).toFile().exists()) {
097            throw new ExternalMessageBodyException("Path did not match any allowed external content paths: " +
098                    extPath);
099        }
100
101        // Check that the uri is within an allowed path
102        if (allowedList.stream().anyMatch(allowed -> path.startsWith(allowed))) {
103            return;
104        }
105        throw new ExternalMessageBodyException("Path did not match any allowed external content paths: " + extPath);
106    }
107
108    private String normalizeUri(final String path) {
109        // lowercase the scheme since it is case insensitive
110        final String[] parts = path.split(":", 2);
111        final String normalized;
112        if (parts.length == 2) {
113            normalized = parts[0].toLowerCase() + ":" + parts[1];
114        } else {
115            return path;
116        }
117        // file uris can have between 1 and 3 slashes depending on if the authority is present
118        if (normalized.startsWith("file://")) {
119            return NORMALIZE_FILE_URI.matcher(normalized).replaceFirst("file:/");
120        }
121        return normalized;
122    }
123
124    /**
125     * Loads the allowed list.
126     *
127     * @throws IOException thrown if the allowed list configuration file cannot be read.
128     */
129    @Override
130    protected synchronized void loadConfiguration() throws IOException {
131        LOGGER.info("Loading list of allowed external content locations from {}", configPath);
132        try (final Stream<String> stream = Files.lines(Paths.get(configPath))) {
133            allowedList = stream.map(line -> normalizeUri(line.trim()))
134                    .filter(line -> isAllowanceValid(line))
135                    .collect(Collectors.toList());
136        }
137    }
138
139    private boolean isAllowanceValid(final String allowance) {
140        final Matcher schemeMatcher = SCHEME_PATTERN.matcher(allowance);
141        final boolean schemeMatches = schemeMatcher.matches();
142        if (!schemeMatches || RELATIVE_MOD_PATTERN.matcher(allowance).matches()) {
143            LOGGER.error("Invalid path {} specified in external path configuration {}",
144                    allowance, configPath);
145            return false;
146        }
147
148        final String protocol = schemeMatcher.group(1).toLowerCase();
149        if ("file".equals(protocol)) {
150            // If a file uri ends with / it must be a directory, otherwise it must be a file.
151            final File allowing = new File(URI.create(allowance).getPath());
152            if ((allowance.endsWith("/") && !allowing.isDirectory()) || (!allowance.endsWith("/") && !allowing
153                    .isFile())) {
154                LOGGER.error("Invalid path {} in configuration {}, directories must end with a '/'",
155                        allowance, configPath);
156                return false;
157            }
158        } else if ("http".equals(protocol) || "https".equals(protocol)) {
159            if (!HTTP_DOMAIN_PATTERN.matcher(allowance).matches()) {
160                LOGGER.error("Invalid path {} in configuration {}, domain must end with a '/'",
161                        allowance, configPath);
162                return false;
163            }
164        }
165        return true;
166    }
167}
