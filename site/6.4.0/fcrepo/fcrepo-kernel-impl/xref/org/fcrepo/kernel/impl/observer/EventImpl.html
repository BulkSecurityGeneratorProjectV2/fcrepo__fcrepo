7   package org.fcrepo.kernel.impl.observer;
8   
9   import org.fcrepo.kernel.api.identifiers.FedoraId;
10  import org.fcrepo.kernel.api.observer.Event;
11  import org.fcrepo.kernel.api.observer.EventType;
12  
13  import java.net.URI;
14  import java.time.Instant;
15  import java.util.Set;
16  
17  import static com.google.common.base.Preconditions.checkNotNull;
18  import static java.util.UUID.randomUUID;
19  
20  /**
21   * An event that describes one or more actions that a user preformed on a resource.
22   *
23   * @author pwinckles
24   */
25  public class EventImpl implements Event {
26  
27      private final String eventId;
28      private final FedoraId fedoraId;
29      private final Set<EventType> types;
30      private final Set<String> resourceTypes;
31      private final String userID;
32      private final URI userURI;
33      private final String userAgent;
34      private final String baseUrl;
35      private final Instant date;
36  
37      /**
38       * Create a new FedoraEvent
39       *
40       * @param fedoraId the FedoraId of the resource the event is on
41       * @param types a collection of Fedora EventTypes
42       * @param resourceTypes the rdf types of the corresponding resource
43       * @param userID the acting user for this event
44       * @param userURI the uri of the acting user for this event
45       * @param userAgent the user-agent associated with the request
46       * @param baseUrl the originating request's baseUrl
47       * @param date the timestamp for this event
48       */
49      public EventImpl(final FedoraId fedoraId, final Set<EventType> types,
50                       final Set<String> resourceTypes, final String userID,
51                       final URI userURI, final String userAgent, final String baseUrl,
52                       final Instant date) {
53          this.eventId = "urn:uuid:" + randomUUID().toString();
54          this.fedoraId = checkNotNull(fedoraId, "fedoraId cannot be null");
55          this.types = Set.copyOf(checkNotNull(types, "types cannot be null"));
56          this.resourceTypes = Set.copyOf(checkNotNull(resourceTypes, "resourceTypes cannot be null"));
57          this.userID = userID;
58          this.userURI = userURI;
59          this.userAgent = userAgent;
60          checkNotNull(baseUrl, "baseUrl cannot be null");
61          // baseUrl is expected to not have a trailing slash
62          this.baseUrl = baseUrl.replaceAll("/+$", "");
63          this.date = checkNotNull(date, "date cannot be null");
64      }
65  
66      @Override
67      public FedoraId getFedoraId() {
68          return fedoraId;
69      }
70  
71      @Override
72      public Set<EventType> getTypes() {
73          return types;
74      }
75  
76      @Override
77      public Set<String> getResourceTypes() {
78          return resourceTypes;
79      }
80  
81      @Override
82      public String getPath() {
83          return fedoraId.getFullIdPath();
84      }
85  
86      @Override
87      public String getUserID() {
88          return userID;
89      }
90  
91      @Override
92      public Instant getDate() {
93          return date;
94      }
95  
96      @Override
97      public String getEventID() {
98          return eventId;
99      }
100 
101     @Override
102     public URI getUserURI() {
103         return userURI;
104     }
105 
106     @Override
107     public String getUserAgent() {
108         return userAgent;
109     }
110 
111     @Override
112     public String getBaseUrl() {
113         return baseUrl;
114     }
115 
116     @Override
117     public String toString() {
118         return "EventImpl{" +
119                 "eventId='" + eventId + '\'' +
120                 ", fedoraId=" + fedoraId +
121                 ", types=" + types +
122                 ", resourceTypes=" + resourceTypes +
123                 ", userID='" + userID + '\'' +
124                 ", userURI=" + userURI +
125                 ", userAgent=" + userAgent +
126                 ", baseUrl=" + baseUrl +
127                 ", date=" + date +
128                 '}';
129     }
130 
131 }
