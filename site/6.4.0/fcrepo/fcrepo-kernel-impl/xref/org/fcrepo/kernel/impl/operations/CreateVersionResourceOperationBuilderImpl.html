7   package org.fcrepo.kernel.impl.operations;
8   
9   import org.fcrepo.kernel.api.Transaction;
10  import org.fcrepo.kernel.api.identifiers.FedoraId;
11  import org.fcrepo.kernel.api.operations.CreateVersionResourceOperationBuilder;
12  import org.fcrepo.kernel.api.operations.ResourceOperation;
13  
14  /**
15   * Default impl of {@link CreateVersionResourceOperationBuilder}
16   *
17   * @author pwinckles
18   */
19  public class CreateVersionResourceOperationBuilderImpl extends AbstractResourceOperationBuilder
20          implements CreateVersionResourceOperationBuilder {
21  
22      /**
23       * Create a new builder
24       *
25       * @param transaction the transaction.
26       * @param resourceId the resource id
27       */
28      public CreateVersionResourceOperationBuilderImpl(final Transaction transaction, final FedoraId resourceId) {
29          super(transaction, resourceId);
30      }
31  
32      @Override
33      public ResourceOperation build() {
34          final var operation = new CreateVersionResourceOperationImpl(transaction, rescId);
35          operation.setUserPrincipal(userPrincipal);
36          return operation;
37      }
38  
39  }
