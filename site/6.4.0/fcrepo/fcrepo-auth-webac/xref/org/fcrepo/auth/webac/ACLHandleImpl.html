6   package org.fcrepo.auth.webac;
7   
8   import java.util.List;
9   
10  import org.fcrepo.kernel.api.auth.ACLHandle;
11  import org.fcrepo.kernel.api.auth.WebACAuthorization;
12  import org.fcrepo.kernel.api.models.FedoraResource;
13  
14  /**
15   * A simple class connecting an URI pointing to an ACL to a {@link FedoraResource} that points to that ACL.
16   *
17   * @author ajs6f
18   */
19  public class ACLHandleImpl implements ACLHandle {
20  
21      private final FedoraResource resource;
22  
23      private final List<WebACAuthorization> authorizations;
24      /**
25       * Default constructor.
26       *
27       * @param resource the requested FedoraResource
28       * @param authorizations any authorizations associated with the uri
29       */
30      public ACLHandleImpl(final FedoraResource resource, final List<WebACAuthorization> authorizations) {
31          this.resource = resource;
32          this.authorizations = authorizations;
33      }
34  
35      @Override
36      public FedoraResource getResource() {
37          return resource;
38      }
39  
40      @Override
41      public List<WebACAuthorization> getAuthorizations() {
42          return authorizations;
43      }
44  }
