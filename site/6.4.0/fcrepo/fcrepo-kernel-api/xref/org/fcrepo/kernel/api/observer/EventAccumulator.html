7   package org.fcrepo.kernel.api.observer;
8   
9   import org.fcrepo.kernel.api.Transaction;
10  import org.fcrepo.kernel.api.identifiers.FedoraId;
11  import org.fcrepo.kernel.api.operations.ResourceOperation;
12  
13  /**
14   * Accumulates events for changes made to resources, grouped by transaction. The events are not emitted until after the
15   * transaction has been committed. If a transaction is rolled back, {@link #clearEvents} MUST be called to release the
16   * stored events.
17   *
18   * @author pwinckles
19   */
20  public interface EventAccumulator {
21  
22      /**
23       * Registers a new event to a transaction.
24       *
25       * @param transaction the transaction
26       * @param fedoraId the id of the affected resource
27       * @param operation the operation affecting the resource
28       */
29      void recordEventForOperation(Transaction transaction, FedoraId fedoraId, ResourceOperation operation);
30  
31      /**
32       * Emits all of the events that were accumulated within the transaction. Multiple events affecting the same resource
33       * are combined into a single event.
34       *
35       * <p>This method SHOULD NOT throw an exception if an event fails to be emitted. It should always attempt to emit
36       * all events accumulated within a transaction.
37       *
38       * @param transaction the transaction
39       * @param baseUrl the baseUrl of the requests
40       * @param userAgent the user-agent of the user making the requests
41       */
42      void emitEvents(Transaction transaction, String baseUrl, String userAgent);
43  
44      /**
45       * Removes all of a transaction's accumulated events without emitting them. This must be called when a transaction
46       * is rolled back.
47       *
48       * @param transaction the id of the transaction
49       */
50      void clearEvents(Transaction transaction);
51  
52  }
