6   package org.fcrepo.kernel.api.services;
7   
8   import org.apache.jena.rdf.model.Model;
9   import org.fcrepo.kernel.api.Transaction;
10  import org.fcrepo.kernel.api.identifiers.FedoraId;
11  import org.fcrepo.kernel.api.models.ExternalContent;
12  
13  import java.io.InputStream;
14  import java.net.URI;
15  import java.util.Collection;
16  import java.util.List;
17  
18  /**
19   * Interface for a service to create a new resource via a POST request.
20   * @author whikloj
21   * @since 2019-11-05
22   */
23  public interface CreateResourceService {
24  
25      /**
26       * Create a new NonRdfSource resource.
27       *
28       * @param tx The transaction for the request.
29       * @param userPrincipal the principal of the user performing the service
30       * @param fedoraId The internal identifier of the resource.
31       * @param contentType The content-type header or null if none.
32       * @param filename The original filename of the binary
33       * @param contentSize The size of the content stream
34       * @param linkHeaders The original LINK headers or null if none.
35       * @param digest The binary digest or null if none.
36       * @param requestBody The request body or null if none.
37       * @param externalContent The external content handler or null if none.
38       */
39      void perform(Transaction tx, String userPrincipal, FedoraId fedoraId,
40                   String contentType, String filename, long contentSize, List<String> linkHeaders,
41                   Collection<URI> digest, InputStream requestBody, ExternalContent externalContent);
42  
43      /**
44       * Create a new RdfSource resource.
45       *
46       * @param tx The transaction for the request.
47       * @param userPrincipal the principal of the user performing the service
48       * @param fedoraId The internal identifier of the resource
49       * @param linkHeaders The original LINK headers or null if none.
50       * @param model The request body RDF as a Model
51       */
52      void perform(Transaction tx, String userPrincipal, FedoraId fedoraId,
53              List<String> linkHeaders, Model model);
54  
55  }
