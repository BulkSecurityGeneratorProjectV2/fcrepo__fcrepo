6   package org.fcrepo.kernel.api.operations;
7   
8   import org.fcrepo.config.ServerManagedPropsMode;
9   import org.fcrepo.kernel.api.Transaction;
10  import org.fcrepo.kernel.api.identifiers.FedoraId;
11  
12  import java.io.InputStream;
13  import java.net.URI;
14  
15  /**
16   * Factory for constructing operations on non-rdf sources
17   *
18   * @author bbpennel
19   */
20  public interface NonRdfSourceOperationFactory extends ResourceOperationFactory {
21  
22      /**
23       * Get a builder for a external binary update operation
24       *
25       * @param transaction the transaction
26       * @param rescId id of the resource targeted by the operation
27       * @param handling the type of handling to be used for the external binary content
28       * @param contentUri the URI of the external binary content
29       * @return a new builder
30       */
31      NonRdfSourceOperationBuilder updateExternalBinaryBuilder(Transaction transaction, FedoraId rescId, String handling,
32                                                               URI contentUri);
33  
34      /**
35       * Get a builder for an internal binary update operation
36       *
37       * @param transaction the transaction
38       * @param rescId id of the resource targeted by the operation
39       * @param contentStream inputstream for the content of this binary
40       * @return a new builder
41       */
42      NonRdfSourceOperationBuilder updateInternalBinaryBuilder(Transaction transaction, FedoraId rescId,
43                                                               InputStream contentStream);
44  
45      /**
46       * Get a builder for a external binary create operation
47       *
48       * @param transaction the transaction
49       * @param rescId id of the resource targeted by the operation
50       * @param handling the type of handling to be used for the external binary content
51       * @param contentUri the URI of the external binary content
52       * @return a new builder
53       */
54      CreateNonRdfSourceOperationBuilder createExternalBinaryBuilder(Transaction transaction, FedoraId rescId,
55                                                                     String handling,
56                                                                     URI contentUri);
57  
58      /**
59       * Get a builder for an internal binary create operation
60       *
61       * @param transaction the transaction
62       * @param rescId id of the resource targeted by the operation
63       * @param contentStream inputstream for the content of this binary
64       * @return a new builder
65       */
66      CreateNonRdfSourceOperationBuilder createInternalBinaryBuilder(Transaction transaction, FedoraId rescId,
67                                                                     InputStream contentStream);
68  
69      /**
70       * Get a builder for an operation to update headers of a Non-RDF resource
71       *
72       * @param transaction the transaction
73       * @param resourceId id of the resource targeted by the operation
74       * @param serverManagedPropsMode server managed props mode
75       * @return new builder
76       */
77      UpdateNonRdfSourceHeadersOperationBuilder updateHeadersBuilder(Transaction transaction,
78                                                            FedoraId resourceId,
79                                                            ServerManagedPropsMode serverManagedPropsMode);
80  }
