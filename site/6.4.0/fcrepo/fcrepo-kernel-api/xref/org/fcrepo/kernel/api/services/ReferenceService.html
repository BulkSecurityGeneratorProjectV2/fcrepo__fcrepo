6   package org.fcrepo.kernel.api.services;
7   
8   import org.fcrepo.kernel.api.RdfStream;
9   import org.fcrepo.kernel.api.Transaction;
10  import org.fcrepo.kernel.api.identifiers.FedoraId;
11  import org.fcrepo.kernel.api.models.FedoraResource;
12  
13  /**
14   * Service to retrieve references to repository resources.
15   * @author whikloj
16   * @since 6.0.0
17   */
18  public interface ReferenceService {
19  
20      /**
21       * Return a RDF stream of statements referring to the provided resource.
22       *
23       * @param tx the transaction or null if no transaction.
24       * @param resource the resource to get inbound references for.
25       * @return RDF stream of inbound reference triples.
26       */
27      RdfStream getInboundReferences(final Transaction tx, final FedoraResource resource);
28  
29      /**
30       * Delete all references from a resource to any other resource.
31       *
32       * @param tx the transaction
33       * @param resourceId the ID of the resource referencing others.
34       */
35      void deleteAllReferences(final Transaction tx, final FedoraId resourceId);
36  
37      /**
38       * Parse the stream of triples for references, add any new ones and remove any missing ones.
39       * @param tx the transaction
40       * @param resourceId the subject ID of the triples.
41       * @param userPrincipal the user who's action is updating references.
42       * @param rdfStream the RDF stream.
43       */
44      void updateReferences(final Transaction tx, final FedoraId resourceId, final String userPrincipal,
45                            final RdfStream rdfStream);
46  
47      /**
48       * Commit any pending references.
49       * @param tx the transaction.
50       */
51      void commitTransaction(final Transaction tx);
52  
53      /**
54       * Rollback any pending references.
55       * @param tx the transaction.
56       */
57      void rollbackTransaction(final Transaction tx);
58  
59      /**
60       * Truncates the reference index. This should only be called when rebuilding the index.
61       */
62      void reset();
63  }
