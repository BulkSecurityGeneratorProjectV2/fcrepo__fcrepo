6   package org.fcrepo.kernel.api.operations;
7   
8   import java.io.InputStream;
9   import java.net.URI;
10  import java.util.Collection;
11  
12  /**
13   * An operation for interacting with a non-RDF source resource.
14   *
15   * @author bbpennel
16   */
17  public interface NonRdfSourceOperation extends ResourceOperation {
18  
19      /**
20       * @return the content stream for a local binary
21       */
22      InputStream getContentStream();
23  
24      /**
25       * @return the handling method for external content in this resource
26       */
27      String getExternalHandling();
28  
29      /**
30       * @return the URI for external content in this resource
31       */
32      URI getContentUri();
33  
34      /**
35       * @return The MimeType of content associated with this resource.
36       */
37      String getMimeType();
38  
39      /**
40       * Return the file name for the binary content
41       *
42       * @return original file name for the binary content, or the object's id.
43       */
44      String getFilename();
45  
46      /**
47       * @return the URIs of digests for the content in this resource
48       */
49      Collection<URI> getContentDigests();
50  
51      /**
52       * @return The size in bytes of content associated with this resource.
53       */
54      long getContentSize();
55  }
