6   package org.fcrepo.persistence.ocfl.impl;
7   
8   import org.fcrepo.kernel.api.ReadOnlyTransaction;
9   import org.fcrepo.kernel.api.Transaction;
10  import org.fcrepo.persistence.api.PersistentStorageSession;
11  import org.fcrepo.persistence.api.PersistentStorageSessionManager;
12  import org.fcrepo.persistence.ocfl.api.FedoraToOcflObjectIndex;
13  import org.fcrepo.storage.ocfl.OcflObjectSessionFactory;
14  import org.slf4j.Logger;
15  import org.slf4j.LoggerFactory;
16  import org.springframework.beans.factory.annotation.Autowired;
17  import org.springframework.stereotype.Component;
18  
19  import javax.inject.Inject;
20  import java.util.Map;
21  import java.util.concurrent.ConcurrentHashMap;
22  
23  /**
24   * OCFL implementation of PersistentStorageSessionManager
25   *
26   * @author whikloj
27   * @author dbernstein
28   * @since 2019-09-20
29   */
30  @Component
31  public class OcflPersistentSessionManager implements PersistentStorageSessionManager {
32  
33      private static final Logger LOGGER = LoggerFactory.getLogger(OcflPersistentSessionManager.class);
34  
35      private volatile PersistentStorageSession readOnlySession;
36  
37      private Map<String, PersistentStorageSession> sessionMap;
38  
39      @Inject
40      private OcflObjectSessionFactory objectSessionFactory;
41  
42      @Inject
43      private FedoraToOcflObjectIndex ocflIndex;
44  
45      @Inject
46      private ReindexService reindexService;
47  
48      /**
49       * Default constructor
50       */
51      @Autowired
52      public OcflPersistentSessionManager() {
53          this.sessionMap = new ConcurrentHashMap<>();
54      }
55  
56      @Override
57      public PersistentStorageSession getSession(final Transaction transaction) {
58          if (transaction == null) {
59              throw new IllegalArgumentException("session id must be non-null");
60          }
61  
62          return sessionMap.computeIfAbsent(transaction.getId(), key -> {
63              LOGGER.debug("Creating storage session {}", transaction);
64              return new OcflPersistentStorageSessionMetrics(
65                      new OcflPersistentStorageSession(
66                              transaction,
67                              ocflIndex,
68                              objectSessionFactory,
69                              reindexService));
70          });
71      }
72  
73      @Override
74      public PersistentStorageSession getReadOnlySession() {
75          var localSession = this.readOnlySession;
76  
77          if (localSession == null) {
78              synchronized (this) {
79                  localSession = this.readOnlySession;
80                  if (localSession == null) {
81                      this.readOnlySession = new OcflPersistentStorageSessionMetrics(
82                              new OcflPersistentStorageSession(ReadOnlyTransaction.INSTANCE,
83                                      ocflIndex, objectSessionFactory, reindexService));
84                      localSession = this.readOnlySession;
85                  }
86              }
87          }
88  
89          return localSession;
90      }
91  
92      @Override
93      public PersistentStorageSession removeSession(final String sessionId) {
94          LOGGER.debug("Removing storage session {}", sessionId);
95          return sessionMap.remove(sessionId);
96      }
97  }
