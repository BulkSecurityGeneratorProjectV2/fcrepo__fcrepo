30  /**
31   * An implementation of {@link IndexBuilder}.  This implementation rebuilds the following indexable state derived
32   * from the underlying OCFL directory:
33   * 1) the link between a {@link org.fcrepo.kernel.api.identifiers.FedoraId} and an OCFL object identifier
34   * 2) the containment relationships between {@link org.fcrepo.kernel.api.identifiers.FedoraId}s
35   * 3) the reference relationships between {@link org.fcrepo.kernel.api.identifiers.FedoraId}s
36   * 4) the search index
37   * 5) the membership relationships for Direct and Indirect containers.
38   *
39   * @author dbernstein
40   * @author whikloj
41   * @since 6.0.0
42   */
43  @Component
44  public class IndexBuilderImpl implements IndexBuilder {
45  
46      private static final Logger LOGGER = LoggerFactory.getLogger(IndexBuilderImpl.class);
47  
48      @Autowired
49      @Qualifier("ocflIndex")
50      private FedoraToOcflObjectIndex ocflIndex;
51  
52      @Autowired
53      @Qualifier("containmentIndex")
54      private ContainmentIndex containmentIndex;
55  
56      @Inject
57      private OcflRepository ocflRepository;
58  
59      @Inject
60      private ReindexService reindexService;
61  
62      @Inject
63      private OcflPropsConfig ocflPropsConfig;
64  
65      @Inject
66      private FedoraPropsConfig fedoraPropsConfig;
67  
68      @Inject
69      private TransactionManager txManager;
70  
71      @Inject
72      private DbTransactionExecutor dbTransactionExecutor;
73  
74      @Override
75      public void rebuildIfNecessary() {
76          if (shouldRebuild()) {
77              rebuild();
78          } else {
79              LOGGER.debug("No index rebuild necessary");
80          }
81      }
82  
83      private void rebuild() {
84          LOGGER.info("Initiating index rebuild. This may take a while. Progress will be logged periodically.");
85  
86          reindexService.reset();
87  
88          try (var objectIds = ocflRepository.listObjectIds()) {
89              final ReindexManager reindexManager = new ReindexManager(objectIds,
90                      reindexService, ocflPropsConfig, txManager, dbTransactionExecutor);
91  
92              LOGGER.debug("Reading object ids...");
93              final var startTime = Instant.now();
94              try {
95                  reindexManager.start();
96              } catch (final InterruptedException e) {
97                  throw new RuntimeException(e);
98              } finally {
99                  reindexManager.shutdown();
100             }
101             final var endTime = Instant.now();
102             final var count = reindexManager.getCompletedCount();
103             final var errors = reindexManager.getErrorCount();
104             LOGGER.info("Index rebuild completed {} objects successfully and {} objects had errors in {} ",
105                     count, errors, getDurationMessage(Duration.between(startTime, endTime)));
106         }
107     }
108 
109     private boolean shouldRebuild() {
110         final var repoRoot = getRepoRootMapping();
111         if (fedoraPropsConfig.isRebuildOnStart()) {
112             return true;
113         } else if (repoRoot == null) {
114             return true;
115         } else {
116             return !repoContainsRootObject(repoRoot);
117         }
118     }
119 
120     private String getRepoRootMapping() {
121         try {
122             return ocflIndex.getMapping(ReadOnlyTransaction.INSTANCE, FedoraId.getRepositoryRootId()).getOcflObjectId();
123         } catch (final FedoraOcflMappingNotFoundException e) {
124             return null;
125         }
126     }
127 
128     private boolean repoContainsRootObject(final String id) {
129         return ocflRepository.containsObject(id);
130     }
131 
132     private String getDurationMessage(final Duration duration) {
133         String message = String.format("%d seconds", duration.toSecondsPart());
134         if (duration.getSeconds() > 60) {
135             message = String.format("%d mins, ", duration.toMinutesPart()) + message;
136         }
137         if (duration.getSeconds() > 3600) {
138             message = String.format("%d hours, ", duration.getSeconds() / 3600) + message;
139         }
140         return message;
141     }
142 }
