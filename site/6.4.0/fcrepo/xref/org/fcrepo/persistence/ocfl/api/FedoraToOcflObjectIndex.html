6   package org.fcrepo.persistence.ocfl.api;
7   
8   import javax.annotation.Nonnull;
9   
10  import org.fcrepo.kernel.api.Transaction;
11  import org.fcrepo.kernel.api.identifiers.FedoraId;
12  import org.fcrepo.persistence.ocfl.impl.FedoraOcflMapping;
13  
14  /**
15   * @author dbernstein
16   * @since 6.0.0
17   */
18  public interface FedoraToOcflObjectIndex {
19  
20      /**
21       * Retrieve identification information for the OCFL object which either contains, or is identified by,
22       * the provided fedora resource id. In other words the method will find the closest resource that is persisted
23       * as an OCFL object and returns its identifiers.
24       *
25       * If you pass fedora identifier that is not part of an archival group such as
26       * "my/fedora/binary/fcr:metadata"  the  fedora resource returned in the mapping will be "my/fedora/binary".
27       *
28       * Contrast this  with an Archival Group example:  if you pass in "my/archival-group/binary/fcr:metadata" the
29       * resource returned in the mapping would be "my/archival-group".
30       *
31       * @param session the current session, or null for read-only.
32       * @param fedoraResourceIdentifier the fedora resource identifier
33       *
34       * @return the mapping
35       * @throws FedoraOcflMappingNotFoundException when no mapping exists for the specified identifier.
36       */
37      FedoraOcflMapping getMapping(final Transaction session, final FedoraId fedoraResourceIdentifier)
38              throws FedoraOcflMappingNotFoundException;
39  
40      /**
41       * Adds a mapping to the index
42       *
43       * @param session the current session.
44       * @param fedoraResourceIdentifier The fedora resource
45       * @param fedoraRootObjectIdentifier   The identifier of the root fedora object resource
46       * @param ocflObjectId             The ocfl object id
47       * @return  The newly created mapping
48       */
49      FedoraOcflMapping addMapping(@Nonnull Transaction session, final FedoraId fedoraResourceIdentifier,
50                                   final FedoraId fedoraRootObjectIdentifier, final String ocflObjectId);
51  
52      /**
53       * Removes a mapping
54       *
55       * @param session the current session.
56       * @param fedoraResourceIdentifier The fedora resource to remove the mapping for
57       */
58      void removeMapping(@Nonnull final Transaction session, final FedoraId fedoraResourceIdentifier);
59  
60      /**
61       * Remove all persistent state associated with the index.
62       */
63      void reset();
64  
65      /**
66       * Commit mapping changes for the session.
67       *
68       * @param session the session to commit.
69       */
70      void commit(@Nonnull final Transaction session);
71  
72      /**
73       * Rollback mapping changes for the session.
74       *
75       * @param session the session to rollback.
76       */
77      void rollback(@Nonnull final Transaction session);
78  
79  }
