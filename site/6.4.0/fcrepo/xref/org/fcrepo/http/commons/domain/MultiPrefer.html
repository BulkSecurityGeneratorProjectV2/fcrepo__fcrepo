6   package org.fcrepo.http.commons.domain;
7   
8   import java.util.Set;
9   
10  import javax.ws.rs.HeaderParam;
11  
12  /**
13   * Aggregates information from multiple Prefer HTTP headers.
14   *
15   * @author ajs6f
16   * @since 23 October 2014
17   */
18  public class MultiPrefer extends SinglePrefer {
19  
20      /**
21       * @param header the header
22       */
23      public MultiPrefer(final String header) {
24          super(header);
25      }
26  
27      /**
28       * @param prefers the prefers
29       */
30      public MultiPrefer(final @HeaderParam("Prefer") Set<SinglePrefer> prefers) {
31          super("");
32          prefers.forEach(p -> preferTags().addAll(p.preferTags()));
33      }
34  }
