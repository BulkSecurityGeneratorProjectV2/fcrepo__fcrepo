6   package org.fcrepo.http.commons.exceptionhandlers;
7   
8   import org.slf4j.Logger;
9   
10  /**
11   * @author barmintor
12   * @since 6/28/16
13   */
14  public interface ExceptionDebugLogging {
15      /**
16       * Log a Throwable at the DEBUG level, log the stacktrace at the TRACE level.
17       *
18       * @param context ExceptionDebugLogging the exception intercepting context.
19       * @param error Throwable the intercepted error.
20       * @param logger Logger the logger to use
21       */
22      default void debugException(final ExceptionDebugLogging context, final Throwable error, final Logger logger) {
23          /*
24           * Because the majority case is not debug logging, trade one additional
25           * accessor call in a guard clause for multiple in message construction
26           * and internal log level checks
27           */
28          if (!logger.isDebugEnabled()) {
29              return;
30          }
31          logger.debug("{} intercepted exception:{} \n", context.getClass()
32                  .getSimpleName(), error);
33          logger.trace(error.getMessage(), error);
34      }
35  }
