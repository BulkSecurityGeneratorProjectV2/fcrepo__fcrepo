7   package org.fcrepo.http.commons.exceptionhandlers;
8   
9   import org.fcrepo.kernel.api.exception.InsufficientStorageException;
10  import org.slf4j.Logger;
11  
12  import javax.ws.rs.core.Response;
13  import javax.ws.rs.ext.ExceptionMapper;
14  import javax.ws.rs.ext.Provider;
15  
16  import static javax.ws.rs.core.Response.status;
17  import static org.fcrepo.http.commons.domain.RDFMediaType.TEXT_PLAIN_WITH_CHARSET;
18  import static org.slf4j.LoggerFactory.getLogger;
19  
20  /**
21   * Translate InsufficientStorageException errors into HTTP error codes
22   *
23   * @author Daniel Bernstein
24   * @since Oct 7, 2016
25   */
26  @Provider
27  public class InsufficientStorageExceptionMapper implements
28          ExceptionMapper<InsufficientStorageException>, ExceptionDebugLogging {
29  
30      private static final Logger LOGGER =
31              getLogger(InsufficientStorageException.class);
32  
33      public static final int INSUFFICIENT_STORAGE_HTTP_CODE = 507;
34  
35      @Override
36      public Response toResponse(final InsufficientStorageException e) {
37          LOGGER.error("Insufficient storage", e);
38          return status(INSUFFICIENT_STORAGE_HTTP_CODE).entity(e.getMessage()).type(TEXT_PLAIN_WITH_CHARSET).build();
39      }
40  }
