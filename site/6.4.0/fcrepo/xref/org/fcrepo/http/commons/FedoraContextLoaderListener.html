6   package org.fcrepo.http.commons;
7   
8   import org.springframework.beans.factory.BeanDefinitionStoreException;
9   import org.springframework.web.context.ContextLoaderListener;
10  
11  import javax.servlet.ServletContextEvent;
12  
13  import static org.slf4j.LoggerFactory.getLogger;
14  
15  /**
16   * This class wraps the standard Spring ContextLoaderListener in order to catch initialization errors.
17   * 
18   * @author awoods
19   * @since 2016-06-09
20   */
21  public class FedoraContextLoaderListener extends ContextLoaderListener {
22  
23      private static final org.slf4j.Logger LOGGER = getLogger(FedoraContextLoaderListener.class);
24  
25      @Override
26      public void contextInitialized(final ServletContextEvent event) {
27          try {
28              super.contextInitialized(event);
29          } catch (final BeanDefinitionStoreException e) {
30              final String msg = "\n" +
31                      "=====================================================================\n" +
32                      "=====================================================================\n" +
33                      "---------- FEDORA CONFIGURATION ERROR ----------\n" +
34                      "\n" +
35                      "See documentation specific to your version of Fedora\n" +
36                      "https://wiki.lyrasis.org/display/FEDORA6x/Application+Configuration\n" +
37                      "\n" +
38                      "=====================================================================\n" +
39                      "=====================================================================\n";
40              LOGGER.error(msg);
41          }
42  
43      }
44  
45  }
