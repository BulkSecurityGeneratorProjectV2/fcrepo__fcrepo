6   package org.fcrepo.http.commons.responses;
7   
8   import java.io.IOException;
9   import java.io.InputStream;
10  
11  import org.apache.commons.io.input.BoundedInputStream;
12  
13  /**
14   * An {@link InputStream} that skips bytes and only returns the data up to a certain limit
15   *
16   * @author awoods
17   * @author ajs6f
18   */
19  public class RangeRequestInputStream extends BoundedInputStream {
20  
21      /**
22       * @param in the underlying input stream, or <code>null</code> if
23       *           this instance is to be created without an underlying stream.
24       * @param skip the number of bytes to skip at the beginning of the stream
25       * @param length the number of bytes from the inputstream to read
26       * @throws IOException if IO exception occurred
27       */
28      public RangeRequestInputStream(final InputStream in, final long skip, final long length) throws IOException {
29          super(in, length);
30          in.skip(skip);
31      }
32  }
