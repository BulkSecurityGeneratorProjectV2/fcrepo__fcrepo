6   package org.fcrepo.jms;
7   
8   import static java.lang.String.join;
9   import static java.util.stream.Collectors.joining;
10  import static org.slf4j.LoggerFactory.getLogger;
11  
12  import java.util.Set;
13  import javax.jms.JMSException;
14  import javax.jms.Message;
15  import javax.jms.Session;
16  
17  import org.fcrepo.kernel.api.observer.Event;
18  import org.fcrepo.kernel.api.observer.EventType;
19  import org.fcrepo.event.serialization.EventSerializer;
20  import org.fcrepo.event.serialization.JsonLDSerializer;
21  
22  import org.slf4j.Logger;
23  
24  /**
25   * Generates JMS {@link Message}s composed entirely of headers, based entirely
26   * on information found in the {@link Event} that triggers publication.
27   *
28   * @author ajs6f
29   * @author escowles
30   * @since Dec 2, 2013
31   */
32  public class DefaultMessageFactory implements JMSEventMessageFactory {
33  
34      private static final String JMS_NAMESPACE = "org.fcrepo.jms.";
35  
36      public static final String TIMESTAMP_HEADER_NAME = JMS_NAMESPACE
37              + "timestamp";
38  
39      public static final String IDENTIFIER_HEADER_NAME = JMS_NAMESPACE
40              + "identifier";
41  
42      public static final String EVENT_TYPE_HEADER_NAME = JMS_NAMESPACE
43              + "eventType";
44  
45      public static final String BASE_URL_HEADER_NAME = JMS_NAMESPACE
46              + "baseURL";
47  
48      public static final String RESOURCE_TYPE_HEADER_NAME = JMS_NAMESPACE + "resourceType";
49  
50      public static final String USER_HEADER_NAME = JMS_NAMESPACE + "user";
51      public static final String USER_AGENT_HEADER_NAME = JMS_NAMESPACE + "userAgent";
52      public static final String EVENT_ID_HEADER_NAME = JMS_NAMESPACE + "eventID";
53  
54      @Override
55      public Message getMessage(final Event event, final Session jmsSession)
56              throws JMSException {
57  
58          final EventSerializer serializer = new JsonLDSerializer();
59          final String body = serializer.serialize(event);
60          final Message message = jmsSession.createTextMessage(body);
61  
62          message.setLongProperty(TIMESTAMP_HEADER_NAME, event.getDate().toEpochMilli());
63          message.setStringProperty(BASE_URL_HEADER_NAME, event.getBaseUrl());
64  
65          if (event.getUserAgent() != null) {
66              message.setStringProperty(USER_AGENT_HEADER_NAME, event.getUserAgent());
67          }
68  
69          message.setStringProperty(IDENTIFIER_HEADER_NAME, event.getPath());
70          message.setStringProperty(EVENT_TYPE_HEADER_NAME, getEventURIs(event.getTypes()));
71          message.setStringProperty(USER_HEADER_NAME, event.getUserID());
72          message.setStringProperty(RESOURCE_TYPE_HEADER_NAME, join(",", event.getResourceTypes()));
73          message.setStringProperty(EVENT_ID_HEADER_NAME, event.getEventID());
74  
75          LOGGER.trace("getMessage() returning: {}", message);
76          return message;
77      }
78  
79      private static String getEventURIs(final Set<EventType> types) {
80          final String uris = types.stream()
81                                   .map(EventType::getType)
82                                   .collect(joining(","));
83  
84          LOGGER.debug("Constructed event type URIs: {}", uris);
85          return uris;
86      }
87  
88      private static final Logger LOGGER = getLogger(DefaultMessageFactory.class);
89  
90  }
