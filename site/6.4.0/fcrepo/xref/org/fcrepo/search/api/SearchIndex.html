6   package org.fcrepo.search.api;
7   
8   import org.fcrepo.kernel.api.Transaction;
9   import org.fcrepo.kernel.api.identifiers.FedoraId;
10  import org.fcrepo.kernel.api.models.ResourceHeaders;
11  
12  /**
13   * An interface defining search index management operations
14   *
15   * @author dbernstein
16   */
17  public interface SearchIndex {
18  
19      /**
20       * Adds or updates the index with the resource header information.
21       * @param transaction The externally generated transaction.
22       * @param resourceHeaders The resource headers associated with the resource
23       */
24      void addUpdateIndex(Transaction transaction, ResourceHeaders resourceHeaders);
25  
26      /**
27       * Removes indexed fields associated with the specified Fedora ID
28       *
29       * @param transaction The transaction
30       * @param fedoraId    The Fedora ID
31       */
32      void removeFromIndex(Transaction transaction, FedoraId fedoraId);
33  
34      /**
35       * Performs a search based on the parameters and returns the result.
36       *
37       * @param parameters The parameters defining the search
38       * @return The result of the search
39       */
40      SearchResult doSearch(SearchParameters parameters) throws InvalidQueryException;
41  
42      /**
43       * Remove all persistent state associated with the index.
44       */
45      void reset();
46  
47      /**
48       * Commit the changes made in the transaction.
49       *
50       * @param tx The transaction .
51       */
52      void commitTransaction(final Transaction tx);
53  
54      /**
55       * Rollback the changes in the transaction.
56       *
57       * @param tx The transaction.
58       */
59      void rollbackTransaction(final Transaction tx);
60  
61  }
