7   package org.fcrepo.kernel.impl;
8   
9   import io.micrometer.core.instrument.Metrics;
10  import io.micrometer.core.instrument.Timer;
11  import org.fcrepo.common.metrics.MetricsHelper;
12  import org.fcrepo.kernel.api.ContainmentIndex;
13  import org.fcrepo.kernel.api.Transaction;
14  import org.fcrepo.kernel.api.identifiers.FedoraId;
15  import org.springframework.beans.factory.annotation.Autowired;
16  import org.springframework.beans.factory.annotation.Qualifier;
17  import org.springframework.stereotype.Component;
18  
19  import java.time.Instant;
20  import java.util.stream.Stream;
21  
22  /**
23   * ContainmentIndex wrapper for adding metrics
24   *
25   * @author pwinckles
26   */
27  @Component("containmentIndex")
28  public class ContainmentIndexMetrics implements ContainmentIndex {
29  
30      private static final String METRIC_NAME = "fcrepo.db";
31      private static final String DB = "db";
32      private static final String CONTAINMENT = "containment";
33      private static final String OPERATION = "operation";
34  
35      private static final Timer getContainsTimer = Metrics.timer(METRIC_NAME,
36              DB, CONTAINMENT, OPERATION, "getContains");
37      private static final Timer getContainsDeletedTimer = Metrics.timer(METRIC_NAME,
38              DB, CONTAINMENT, OPERATION, "getContainsDeleted");
39      private static final Timer getContainsByTimer = Metrics.timer(METRIC_NAME,
40              DB, CONTAINMENT, OPERATION, "getContainsBy");
41      private static final Timer removeContainedByTimer = Metrics.timer(METRIC_NAME,
42              DB, CONTAINMENT, OPERATION, "removeContainedBy");
43      private static final Timer removeResourceTimer = Metrics.timer(METRIC_NAME,
44              DB, CONTAINMENT, OPERATION, "removeResource");
45      private static final Timer purgeResourceTimer = Metrics.timer(METRIC_NAME,
46              DB, CONTAINMENT, OPERATION, "purgeResource");
47      private static final Timer addContainedByTimer = Metrics.timer(METRIC_NAME,
48              DB, CONTAINMENT, OPERATION, "addContainedBy");
49      private static final Timer commitTransactionTimer = Metrics.timer(METRIC_NAME,
50              DB, CONTAINMENT, OPERATION, "commitTransaction");
51      private static final Timer rollbackTransactionTimer = Metrics.timer(METRIC_NAME,
52              DB, CONTAINMENT, OPERATION, "rollbackTransaction");
53      private static final Timer resourceExistsTimer = Metrics.timer(METRIC_NAME,
54              DB, CONTAINMENT, OPERATION, "resourceExists");
55      private static final Timer getContainerIdByPathTimer = Metrics.timer(METRIC_NAME,
56              DB, CONTAINMENT, OPERATION, "getContainerIdByPath");
57      private static final Timer resetTimer = Metrics.timer(METRIC_NAME,
58              DB, CONTAINMENT, OPERATION, "reset");
59      private static final Timer hasResourcesStartingWithTimer = Metrics.timer(METRIC_NAME,
60              DB, CONTAINMENT, OPERATION, "hasResourcesStartingWith");
61      private static final Timer containmentLastUpdateTimer = Metrics.timer(METRIC_NAME, DB, CONTAINMENT, OPERATION,
62              "containmentLastUpdated");
63  
64      @Autowired
65      @Qualifier("containmentIndexImpl")
66      private ContainmentIndex containmentIndexImpl;
67  
68      @Override
69      public Stream<String> getContains(final Transaction tx, final FedoraId fedoraId) {
70          return MetricsHelper.time(getContainsTimer, () -> {
71              return containmentIndexImpl.getContains(tx, fedoraId);
72          });
73      }
74  
75      @Override
76      public Stream<String> getContainsDeleted(final Transaction tx, final FedoraId fedoraId) {
77          return MetricsHelper.time(getContainsDeletedTimer, () -> {
78              return containmentIndexImpl.getContainsDeleted(tx, fedoraId);
79          });
80      }
81  
82      @Override
83      public String getContainedBy(final Transaction tx, final FedoraId resource) {
84          return MetricsHelper.time(getContainsByTimer, () -> {
85              return containmentIndexImpl.getContainedBy(tx, resource);
86          });
87      }
88  
89      @Override
90      public void removeContainedBy(final Transaction tx, final FedoraId parent, final FedoraId child) {
91          removeContainedByTimer.record(() -> {
92              containmentIndexImpl.removeContainedBy(tx, parent, child);
93          });
94      }
95  
96      @Override
97      public void removeResource(final Transaction tx, final FedoraId resource) {
98          removeResourceTimer.record(() -> {
99              containmentIndexImpl.removeResource(tx, resource);
100         });
101     }
102 
103     @Override
104     public void purgeResource(final Transaction tx, final FedoraId resource) {
105         purgeResourceTimer.record(() -> {
106             containmentIndexImpl.purgeResource(tx, resource);
107         });
108     }
109 
110     @Override
111     public void addContainedBy(final Transaction tx, final FedoraId parent, final FedoraId child) {
112         addContainedByTimer.record(() -> {
113             containmentIndexImpl.addContainedBy(tx, parent, child);
114         });
115     }
116 
117     @Override
118     public void addContainedBy(final Transaction tx, final FedoraId parent, final FedoraId child,
119                                final Instant startTime, final Instant endTime) {
120         addContainedByTimer.record(() -> containmentIndexImpl.addContainedBy(tx, parent, child, startTime, endTime));
121     }
122 
123     @Override
124     public void commitTransaction(final Transaction tx) {
125         commitTransactionTimer.record(() -> {
126             containmentIndexImpl.commitTransaction(tx);
127         });
128     }
129 
130     @Override
131     public void rollbackTransaction(final Transaction tx) {
132         rollbackTransactionTimer.record(() -> {
133             containmentIndexImpl.rollbackTransaction(tx);
134         });
135     }
136 
137     @Override
138     public boolean resourceExists(final Transaction tx, final FedoraId fedoraId, final boolean includeDeleted) {
139         return MetricsHelper.time(resourceExistsTimer, () -> {
140             return containmentIndexImpl.resourceExists(tx, fedoraId, includeDeleted);
141         });
142     }
143 
144     @Override
145     public FedoraId getContainerIdByPath(final Transaction tx, final FedoraId fedoraId, final boolean checkDeleted) {
146         return MetricsHelper.time(getContainerIdByPathTimer, () -> {
147             return containmentIndexImpl.getContainerIdByPath(tx, fedoraId, checkDeleted);
148         });
149     }
150 
151     @Override
152     public void reset() {
153         resetTimer.record(() -> {
154             containmentIndexImpl.reset();
155         });
156     }
157 
158     @Override
159     public boolean hasResourcesStartingWith(final Transaction tx, final FedoraId fedoraId) {
160         return MetricsHelper.time(hasResourcesStartingWithTimer, () ->
161                 containmentIndexImpl.hasResourcesStartingWith(tx, fedoraId));
162     }
163 
164     @Override
165     public Instant containmentLastUpdated(final Transaction tx, final FedoraId fedoraId) {
166         return MetricsHelper.time(containmentLastUpdateTimer, () ->
167                 containmentIndexImpl.containmentLastUpdated(tx, fedoraId));
168     }
169 }
