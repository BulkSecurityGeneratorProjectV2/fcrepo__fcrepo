6   package org.fcrepo.kernel.impl.operations;
7   
8   import static org.fcrepo.kernel.api.utils.RelaxedPropertiesHelper.checkTripleForDisallowed;
9   
10  import org.fcrepo.config.ServerManagedPropsMode;
11  import org.fcrepo.kernel.api.RdfStream;
12  import org.fcrepo.kernel.api.Transaction;
13  import org.fcrepo.kernel.api.identifiers.FedoraId;
14  import org.fcrepo.kernel.api.operations.RdfSourceOperationBuilder;
15  import org.fcrepo.kernel.api.rdf.DefaultRdfStream;
16  
17  import org.apache.jena.rdf.model.Model;
18  
19  /**
20   * Abstract builder for interacting with an Rdf Source Operation Builder
21   * @author bseeger
22   */
23  public abstract class AbstractRdfSourceOperationBuilder extends AbstractRelaxableResourceOperationBuilder
24                                                          implements RdfSourceOperationBuilder {
25  
26      /**
27       * Holds the stream of user's triples.
28       */
29      protected RdfStream tripleStream;
30  
31      /**
32       * Principal of the user performing the operation
33       */
34      protected String userPrincipal;
35  
36      /**
37       * The interaction model of this resource, null in case of update.
38       */
39      protected final String interactionModel;
40  
41      protected AbstractRdfSourceOperationBuilder(final Transaction transaction, final FedoraId rescId,
42                                                  final String model,
43                                                  final ServerManagedPropsMode serverManagedPropsMode) {
44          super(transaction, rescId, serverManagedPropsMode);
45          interactionModel = model;
46      }
47  
48      @Override
49      public RdfSourceOperationBuilder userPrincipal(final String userPrincipal) {
50          this.userPrincipal = userPrincipal;
51          return this;
52      }
53  
54      @Override
55      public RdfSourceOperationBuilder triples(final RdfStream triples) {
56          if (this.serverManagedPropsMode.equals(ServerManagedPropsMode.RELAXED)) {
57              // Filter out server managed properties, they should only matter to the relaxedProperties method.
58              this.tripleStream = new DefaultRdfStream(triples.topic(), triples.filter(t -> {
59                  try {
60                      checkTripleForDisallowed(t);
61                  } catch (final Exception e) {
62                      return false;
63                  }
64                  return true;
65              }));
66          } else {
67              this.tripleStream = triples;
68          }
69          return this;
70      }
71  
72      @Override
73      public RdfSourceOperationBuilder relaxedProperties(final Model model) {
74          return (RdfSourceOperationBuilder) super.relaxedProperties(model);
75      }
76  }
