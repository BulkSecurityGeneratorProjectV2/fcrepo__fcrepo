7   package org.fcrepo.kernel.impl.services;
8   
9   import io.micrometer.core.instrument.Metrics;
10  import io.micrometer.core.instrument.Timer;
11  import org.fcrepo.common.metrics.MetricsHelper;
12  import org.fcrepo.kernel.api.RdfStream;
13  import org.fcrepo.kernel.api.Transaction;
14  import org.fcrepo.kernel.api.identifiers.FedoraId;
15  import org.fcrepo.kernel.api.models.FedoraResource;
16  import org.fcrepo.kernel.api.services.ReferenceService;
17  import org.springframework.beans.factory.annotation.Autowired;
18  import org.springframework.beans.factory.annotation.Qualifier;
19  import org.springframework.stereotype.Component;
20  
21  /**
22   * ReferenceService wrapper for collecting metrics
23   *
24   * @author pwinckles
25   */
26  @Component("referenceService")
27  public class ReferenceServiceMetrics implements ReferenceService {
28  
29      private static final String METRIC_NAME = "fcrepo.db";
30      private static final String DB = "db";
31      private static final String REFERENCE = "reference";
32      private static final String OPERATION = "operation";
33  
34      private static final Timer getInboundReferences = Metrics.timer(METRIC_NAME,
35              DB, REFERENCE, OPERATION, "getInboundReferences");
36      private static final Timer deleteAllReferencesTimer = Metrics.timer(METRIC_NAME,
37              DB, REFERENCE, OPERATION, "deleteAllReferences");
38      private static final Timer updateReferencesTimer = Metrics.timer(METRIC_NAME,
39              DB, REFERENCE, OPERATION, "updateReferences");
40      private static final Timer commitTransactionTimer = Metrics.timer(METRIC_NAME,
41              DB, REFERENCE, OPERATION, "commitTransaction");
42      private static final Timer rollbackTransactionTimer = Metrics.timer(METRIC_NAME,
43              DB, REFERENCE, OPERATION, "rollbackTransaction");
44      private static final Timer resetTimer = Metrics.timer(METRIC_NAME,
45              DB, REFERENCE, OPERATION, "reset");
46  
47      @Autowired
48      @Qualifier("referenceServiceImpl")
49      private ReferenceService referenceServiceImpl;
50  
51      @Override
52      public RdfStream getInboundReferences(final Transaction tx, final FedoraResource resource) {
53          return MetricsHelper.time(getInboundReferences, () -> {
54              return referenceServiceImpl.getInboundReferences(tx, resource);
55          });
56      }
57  
58      @Override
59      public void deleteAllReferences(final Transaction tx, final FedoraId resourceId) {
60          deleteAllReferencesTimer.record(() -> {
61              referenceServiceImpl.deleteAllReferences(tx, resourceId);
62          });
63      }
64  
65      @Override
66      public void updateReferences(final Transaction tx,
67                                   final FedoraId resourceId,
68                                   final String userPrincipal,
69                                   final RdfStream rdfStream) {
70          updateReferencesTimer.record(() -> {
71              referenceServiceImpl.updateReferences(tx, resourceId, userPrincipal, rdfStream);
72          });
73      }
74  
75      @Override
76      public void commitTransaction(final Transaction tx) {
77          commitTransactionTimer.record(() -> {
78              referenceServiceImpl.commitTransaction(tx);
79          });
80      }
81  
82      @Override
83      public void rollbackTransaction(final Transaction tx) {
84          rollbackTransactionTimer.record(() -> {
85              referenceServiceImpl.rollbackTransaction(tx);
86          });
87      }
88  
89      @Override
90      public void reset() {
91          resetTimer.record(() -> {
92              referenceServiceImpl.reset();
93          });
94      }
95  
96  }
