6   package org.fcrepo.kernel.impl.operations;
7   
8   import static org.fcrepo.kernel.api.RdfLexicon.NON_RDF_SOURCE;
9   
10  import java.io.InputStream;
11  import java.net.URI;
12  
13  import org.fcrepo.kernel.api.Transaction;
14  import org.fcrepo.kernel.api.identifiers.FedoraId;
15  import org.fcrepo.kernel.api.operations.CreateResourceOperation;
16  
17  /**
18   * Operation for creating a new non-rdf source
19   *
20   * @author bbpennel
21   */
22  public class CreateNonRdfSourceOperation extends AbstractNonRdfSourceOperation implements CreateResourceOperation {
23  
24      private FedoraId parentId;
25  
26      /**
27       * Constructor for external content.
28       *
29       * @param rescId the internal identifier.
30       * @param externalContentURI the URI of the external content.
31       * @param externalHandling the type of external content handling (REDIRECT, PROXY)
32       */
33      protected CreateNonRdfSourceOperation(final Transaction transaction, final FedoraId rescId,
34                                            final URI externalContentURI, final String externalHandling) {
35          super(transaction, rescId, externalContentURI, externalHandling);
36      }
37  
38      /**
39       * Constructor for internal binaries.
40       *
41       * @param transaction the transaction
42       * @param rescId the internal identifier.
43       * @param content the stream of the content.
44       */
45      protected CreateNonRdfSourceOperation(final Transaction transaction, final FedoraId rescId,
46                                            final InputStream content) {
47          super(transaction, rescId, content);
48      }
49  
50      @Override
51      public String getInteractionModel() {
52          return NON_RDF_SOURCE.toString();
53      }
54  
55      @Override
56      public boolean isArchivalGroup() {
57          return false;
58      }
59  
60      @Override
61      public FedoraId getParentId() {
62          return parentId;
63      }
64  
65      /**
66       * @param parentId the parentId to set
67       */
68      public void setParentId(final FedoraId parentId) {
69          this.parentId = parentId;
70      }
71  
72  }
