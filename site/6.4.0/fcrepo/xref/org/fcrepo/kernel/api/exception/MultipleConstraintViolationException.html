6   package org.fcrepo.kernel.api.exception;
7   
8   import java.util.HashSet;
9   import java.util.List;
10  import java.util.Set;
11  
12  /**
13   * Wrapper to hold multiple constraint violation exceptions for later reporting.
14   * @author whikloj
15   */
16  public class MultipleConstraintViolationException extends ConstraintViolationException {
17  
18      private Set<ConstraintViolationException> exceptionTypes = new HashSet<>();
19  
20      private String fullMessage = "";
21  
22      public Set<ConstraintViolationException> getExceptionTypes() {
23          return exceptionTypes;
24      }
25  
26      public MultipleConstraintViolationException(final List<ConstraintViolationException> exceptions) {
27          super("There are multiple exceptions");
28          for (final ConstraintViolationException exception : exceptions) {
29              exceptionTypes.add(exception);
30              fullMessage += exception.getMessage() + "\n";
31          }
32      }
33  
34      @Override
35      public String getMessage() {
36          return fullMessage;
37      }
38  
39  }
