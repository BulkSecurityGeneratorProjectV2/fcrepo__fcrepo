6   package org.fcrepo.kernel.api.rdf;
7   
8   import java.io.File;
9   import java.io.IOException;
10  import java.util.HashMap;
11  import java.util.Map;
12  
13  import org.fcrepo.kernel.api.utils.AutoReloadingConfiguration;
14  
15  import com.fasterxml.jackson.core.type.TypeReference;
16  import com.fasterxml.jackson.databind.ObjectMapper;
17  import com.fasterxml.jackson.dataformat.yaml.YAMLFactory;
18  
19  /**
20   * Registry of RDF namespaces
21   *
22   * @author bbpennel
23   */
24  public class RdfNamespaceRegistry extends AutoReloadingConfiguration {
25  
26      private Map<String, String> namespaces;
27  
28      /**
29       * Load the namespace prefix to URI configuration file
30       */
31      @Override
32      protected synchronized void loadConfiguration() throws IOException {
33          final ObjectMapper mapper = new ObjectMapper(new YAMLFactory());
34          final TypeReference<HashMap<String, String>> typeRef = new TypeReference<HashMap<String, String>>() {};
35          namespaces = mapper.readValue(new File(configPath), typeRef);
36      }
37  
38      /**
39       * Get the mapping of namespace prefixes to URIs
40       *
41       * @return map of namespace prefixes to URIs, or an empty map if no mapping was provided.
42       */
43      public Map<String, String> getNamespaces() {
44          if (namespaces == null) {
45              namespaces = new HashMap<>();
46          }
47          return namespaces;
48      }
49  
50      /**
51       * Set the mapping of namespace prefixes to URIs
52       *
53       * @param namespaces mapping of namespaces
54       */
55      public void setNamespaces(final Map<String, String> namespaces) {
56          this.namespaces = namespaces;
57      }
58  }
