142    /**
143     * List of RDF_TYPES for comparison, text/plain isn't really an RDF type but it is still accepted.
144     */
145    private static final List<MediaType> RDF_TYPES = Stream.of(TURTLE_WITH_CHARSET, JSON_LD,
146            N3_WITH_CHARSET, N3_ALT2_WITH_CHARSET, RDF_XML, NTRIPLES, TEXT_PLAIN_WITH_CHARSET
147            ).map(MediaType::valueOf).collect(Collectors.toList());
148
149    /**
150     * This predicate allows comparing a list of accept headers to a list of RDF types.
151     * It is needed to account for charset variations.
152     */
153    private static final Predicate<List<MediaType>> IS_RDF_TYPE = t -> {
154        assert t != null;
155        return t.stream()
156                .anyMatch(c -> RDF_TYPES.stream().anyMatch(c::isCompatible));
157    };
158
159    /**
160     * This predicate checks if the list does not have a mediatype that is wildcard
161     */
162    private static final Predicate<List<MediaType>> NOT_WILDCARD = t -> {
163        assert t != null;
164        return t.stream().noneMatch(MediaType::isWildcardType);
165    };
166
167    /**
168     * This predicate checks if the list does not have a mediatype that is compatible with text html
169     */
170    private static final Predicate<List<MediaType>> NOT_HTML =
171            t -> t.stream().noneMatch(TEXT_HTML_TYPE::isCompatible);

173    private static final VariantListBuilder RDF_VARIANT_BUILDER = VariantListBuilder.newInstance();
174    static {
175        RDF_TYPES.forEach(t -> RDF_VARIANT_BUILDER.mediaTypes(t).add());
176    }

205    /**
206     * Retrieve the node headers
207     *
208     * @param inlineDisposition whether to return a Content-Disposition inline header for a binary
209     * @return response
210     * @throws UnsupportedAlgorithmException if unsupported digest algorithm occurred
211     */
212    @HEAD
213    @Produces({TURTLE_WITH_CHARSET + ";qs=1.0", JSON_LD + ";qs=0.8",
214            N3_WITH_CHARSET, N3_ALT2_WITH_CHARSET, RDF_XML, NTRIPLES, TEXT_PLAIN_WITH_CHARSET,
215            TEXT_HTML_WITH_CHARSET, "*/*"})
216    public Response head(@DefaultValue("false") @QueryParam("inline") final boolean inlineDisposition)
217            throws UnsupportedAlgorithmException {
218        LOGGER.info("HEAD for: {}", externalPath);
219
220        final String datetimeHeader = headers.getHeaderString(ACCEPT_DATETIME);
221        if (!isBlank(datetimeHeader) && resource().isOriginalResource()) {
222            return getMemento(datetimeHeader, resource(), inlineDisposition);
223        }
224
225        final ImmutableList<MediaType> acceptableMediaTypes = ImmutableList.copyOf(headers
226                .getAcceptableMediaTypes());
227
228        checkCacheControlHeaders(request, servletResponse, resource(), transaction());
229
230        addResourceHttpHeaders(resource(), inlineDisposition);
231
232        Response.ResponseBuilder builder = ok();
233
234        if (resource() instanceof Binary) {
235            final Binary binary = (Binary) resource();
236            final MediaType mediaType = getBinaryResourceMediaType(binary);
237
238            if (!acceptableMediaTypes.isEmpty()) {
239                if (acceptableMediaTypes.stream().noneMatch(t -> t.isCompatible(mediaType))) {
240                    return notAcceptable(VariantListBuilder.newInstance().mediaTypes(mediaType).build()).build();
241                }
242            }
243
244            if (binary.isRedirect()) {
245                builder = temporaryRedirect(binary.getExternalURI());
246            }
247
248            // we set the content-type explicitly to avoid content-negotiation from getting in the way
249            builder.type(mediaType.toString());
250
251            // Respect the Want-Digest header with fixity check
252            final String wantDigest = headers.getHeaderString(WANT_DIGEST);
253            if (!isNullOrEmpty(wantDigest)) {
254                builder.header(DIGEST, handleWantDigestHeader(binary, wantDigest));
255            }
256        } else {
257            if (!acceptableMediaTypes.isEmpty() && NOT_WILDCARD.test(acceptableMediaTypes)) {
258                // Accept header is not empty and is not */*
259                if (!IS_RDF_TYPE.test(acceptableMediaTypes)) {
260                    return notAcceptable(VariantListBuilder.newInstance().mediaTypes().build()).build();
261                }
262            } else if (acceptableMediaTypes.isEmpty() || !NOT_WILDCARD.test(acceptableMediaTypes)) {
263                // If there is no Accept header or it is */*, so default to text/turtle
264                builder.type(TURTLE_WITH_CHARSET);
265            }
266            setVaryAndPreferenceAppliedHeaders(servletResponse, prefer, resource());
267        }
268
269
270        return builder.build();
271    }

286    /**
287     * Retrieve the node profile
288     *
289     * @param rangeValue the range value
290     * @param inlineDisposition whether to return a Content-Disposition inline header for a binary
291     * @return a binary or the triples for the specified node
292     * @throws IOException if IO exception occurred
293     * @throws UnsupportedAlgorithmException if unsupported digest algorithm occurred
294     */
295    @GET
296    @Produces({TURTLE_WITH_CHARSET + ";qs=1.0", JSON_LD + ";qs=0.8",
297            N3_WITH_CHARSET, N3_ALT2_WITH_CHARSET, RDF_XML, NTRIPLES, TEXT_PLAIN_WITH_CHARSET,
298            TEXT_HTML_WITH_CHARSET, "*/*"})
299    public Response getResource(
300            @HeaderParam("Range") final String rangeValue,
301            @DefaultValue("false") @QueryParam("inline") final boolean inlineDisposition)
302            throws IOException, UnsupportedAlgorithmException {
303
304        final String datetimeHeader = headers.getHeaderString(ACCEPT_DATETIME);
305        if (!isBlank(datetimeHeader) && resource().isOriginalResource()) {
306            return getMemento(datetimeHeader, resource(), inlineDisposition);
307        }
308
309        checkCacheControlHeaders(request, servletResponse, resource(), transaction());
310
311        final ImmutableList<MediaType> acceptableMediaTypes = ImmutableList.copyOf(headers
312                .getAcceptableMediaTypes());
313
314        LOGGER.info("GET resource '{}'", externalPath);
315        addResourceHttpHeaders(resource(), inlineDisposition);
316
317        if (resource() instanceof Binary) {
318            final Binary binary = (Binary) resource();
319            if (!acceptableMediaTypes.isEmpty()) {
320                final MediaType mediaType = getBinaryResourceMediaType(resource());
321
322                if (acceptableMediaTypes.stream().noneMatch(t -> t.isCompatible(mediaType))) {
323                    return notAcceptable(VariantListBuilder.newInstance().mediaTypes(mediaType).build()).build();
324                }
325            }
326
327            // Respect the Want-Digest header for fixity check
328            final String wantDigest = headers.getHeaderString(WANT_DIGEST);
329            if (!isNullOrEmpty(wantDigest)) {
330                servletResponse.addHeader(DIGEST, handleWantDigestHeader(binary, wantDigest));
331            }
332
333            if (binary.isRedirect()) {
334                return temporaryRedirect(binary.getExternalURI()).build();
335            } else {
336                return getBinaryContent(rangeValue, binary);
337            }
338        } else {
339            if (!acceptableMediaTypes.isEmpty() && NOT_WILDCARD.test(acceptableMediaTypes) &&
340                    NOT_HTML.test(acceptableMediaTypes) &&
341                    !IS_RDF_TYPE.test(acceptableMediaTypes)) {
342                // Accept header is not empty and is not */* and is not text/html and is not a valid RDF type.
343                return notAcceptable(RDF_VARIANT_BUILDER.build()).build();
344            }
345            return getContent(getChildrenLimit(), resource());
346        }
347    }

349    /**
350     * Return the location of a requested Memento.
351     *
352     * @param datetimeHeader The RFC datetime for the Memento.
353     * @param resource The fedora resource
354     * @param inlineDisposition whether to return binary as Content-Disposition inline
355     * @return A 302 Found response or 406 if no mementos.
356     */
357    private Response getMemento(final String datetimeHeader, final FedoraResource resource,
358                                final boolean inlineDisposition) {
359        try {
360            final Instant mementoDatetime = Instant.from(MEMENTO_RFC_1123_FORMATTER.parse(datetimeHeader));
361            final FedoraResource memento = resource.findMementoByDatetime(mementoDatetime);
362            final Response builder;
363            if (memento != null) {
364                builder =
365                    status(FOUND).header(LOCATION, getUri(memento)).build();
366            } else {
367                builder = status(NOT_ACCEPTABLE).build();
368            }
369            addResourceHttpHeaders(resource, inlineDisposition);
370            setVaryAndPreferenceAppliedHeaders(servletResponse, prefer, resource);
371            return builder;
372        } catch (final DateTimeParseException e) {
373            throw new MementoDatetimeFormatException("Invalid Accept-Datetime value: " + e.getMessage()
374                + ". Please use RFC-1123 date-time format, such as 'Tue, 3 Jun 2008 11:05:30 GMT'", e);
375        }
376    }

378    /**
379     * Deletes an object.
380     *
381     * @return response
382     */
383    @DELETE
384    public Response deleteObject() {
385        LOGGER.info("Delete resource '{}'", externalPath);
386        if (externalPath.contains("/" + FedoraTypes.FCR_VERSIONS)) {
387            handleRequestDisallowedOnMemento();
388
389            return status(METHOD_NOT_ALLOWED).build();
390        }
391
392        hasRestrictedPath(externalPath);
393        if (resource() instanceof Container) {
394            final String depth = headers.getHeaderString("Depth");
395            LOGGER.debug("Depth header value is: {}", depth);
396            if (depth != null && !depth.equalsIgnoreCase("infinity")) {
397                throw new ClientErrorException("Depth header, if present, must be set to 'infinity' for containers",
398                        SC_BAD_REQUEST);
399            }
400        }
401        if (resource() instanceof NonRdfSourceDescription && resource().isOriginalResource()) {
402            LOGGER.debug("Trying to delete binary description directly.");
403            throw new ClientErrorException(
404                "NonRDFSource descriptions are removed when their associated NonRDFSource object is removed.",
405                METHOD_NOT_ALLOWED);
406        }
407
408        try {
409            evaluateRequestPreconditions(request, servletResponse, resource(), transaction());
410
411            doInDbTxWithRetry(() -> {
412                deleteResourceService.perform(transaction(), resource(), getUserPrincipal());
413                transaction().commitIfShortLived();
414            });
415            return noContent().build();
416        } finally {
417            transaction().releaseResourceLocksIfShortLived();
418        }
419    }

421    /**
422     * Create a resource at a specified path, or replace triples with provided RDF.
423     *
424     * @param requestContentType the request content type
425     * @param requestBodyStream the request body stream
426     * @param contentDispositionRaw the content disposition value
427     * @param ifMatch the if-match value
428     * @param rawLinks the raw link values
429     * @param digest the digest header
430     * @return 204
431     * @throws InvalidChecksumException if invalid checksum exception occurred
432     * @throws MalformedRdfException if malformed rdf exception occurred
433     * @throws UnsupportedAlgorithmException if an unsupported algorithm exception occurs
434     */
435    @PUT
436    @Consumes
437    public Response createOrReplaceObjectRdf(
438            @HeaderParam(CONTENT_TYPE) final MediaType requestContentType,
439            final InputStream requestBodyStream,
440            @HeaderParam(CONTENT_DISPOSITION) final String contentDispositionRaw,
441            @HeaderParam("If-Match") final String ifMatch,
442            @HeaderParam(LINK) final List<String> rawLinks,
443            @HeaderParam("Digest") final String digest)
444            throws InvalidChecksumException, MalformedRdfException, UnsupportedAlgorithmException,
445                   PathNotFoundException {
446        LOGGER.info("PUT to create resource with ID: {}", externalPath());
447
448        if (externalPath.contains("/" + FedoraTypes.FCR_VERSIONS)) {
449            handleRequestDisallowedOnMemento();
450
451            return status(METHOD_NOT_ALLOWED).build();
452        }
453
454        hasRestrictedPath(externalPath);
455
456        final var transaction = transaction();
457
458        try {
459            final List<String> links = unpackLinks(rawLinks);
460
461            // If request is an external binary, verify link header before proceeding
462            final ExternalContent extContent = extContentHandlerFactory.createFromLinks(links);
463
464            final String interactionModel = checkInteractionModel(links);
465
466            final FedoraId fedoraId = identifierConverter().pathToInternalId(externalPath());
467            final boolean resourceExists = doesResourceExist(transaction, fedoraId, true);
468
469            if (resourceExists) {
470
471                if (httpConfiguration.putRequiresIfMatch() && StringUtils.isBlank(ifMatch)) {
472                    throw new ClientErrorException("An If-Match header is required", 428);
473                }
474
475                final String resInteractionModel = resource().getInteractionModel();
476                if (StringUtils.isNoneBlank(resInteractionModel, interactionModel) &&
477                        !Objects.equals(resInteractionModel, interactionModel)) {
478                    throw new InteractionModelViolationException("Changing the interaction model " + resInteractionModel
479                            + " to " + interactionModel + " is not allowed!");
480                }
481                evaluateRequestPreconditions(request, servletResponse, resource(), transaction);
482            }
483
484            if (isGhostNode(transaction(), fedoraId)) {
485                throw new GhostNodeException("Resource path " + externalPath() + " is an immutable resource.");
486            }
487
488            if (!resourceExists && fedoraId.isDescription()) {
489                // Can't PUT a description to a non-existant binary.
490                final String message;
491                if (fedoraId.asBaseId().isRepositoryRoot()) {
492                    message = "The root of the repository is not a binary, so /" + FCR_METADATA + " does not exist.";
493                } else {
494                    message = "Binary at path " + fedoraId.asBaseId().getFullIdPath() + " not found";
495                }
496                throw new PathNotFoundException(message);
497            }
498
499            final var providedContentType = getSimpleContentType(requestContentType);
500
501            final var created = new AtomicBoolean(false);
502
503                if ((resourceExists && resource() instanceof Binary) ||
504                        (!resourceExists && isBinary(interactionModel,
505                                providedContentType,
506                                requestBodyStream != null && providedContentType != null,
507                                extContent != null))) {
508                    ensureArchivalGroupHeaderNotPresentForBinaries(links);
509
510                    final Collection<URI> checksums = parseDigestHeader(digest);
511                    final var binaryType = requestContentType != null ?
512                            requestContentType : DEFAULT_NON_RDF_CONTENT_TYPE;
513                    final var contentType = extContent == null ?
514                            binaryType.toString() : extContent.getContentType();
515
516                    final String originalFileName;
517                    final long contentSize;
518
519                    if (StringUtils.isNotBlank(contentDispositionRaw)) {
520                        final var contentDisposition = ContentDisposition.parse(contentDispositionRaw);
521                        originalFileName = contentDisposition.getFilename();
522                        contentSize = contentDisposition.getSize() == null ? -1L : contentDisposition.getSize();
523                    } else {
524                        originalFileName = "";
525                        contentSize = -1L;
526                    }
527
528                    doInDbTx(() -> {
529                        if (resourceExists) {
530                            replaceBinariesService.perform(transaction,
531                                    getUserPrincipal(),
532                                    fedoraId,
533                                    originalFileName,
534                                    contentType,
535                                    checksums,
536                                    requestBodyStream,
537                                    contentSize,
538                                    extContent);
539                        } else {
540                            createResourceService.perform(transaction,
541                                    getUserPrincipal(),
542                                    fedoraId,
543                                    contentType,
544                                    originalFileName,
545                                    contentSize,
546                                    links,
547                                    checksums,
548                                    requestBodyStream,
549                                    extContent);
550                            created.set(true);
551                        }
552                        transaction.commitIfShortLived();
553                    });
554                } else {
555                    final var contentType = requestContentType != null ? requestContentType : DEFAULT_RDF_CONTENT_TYPE;
556                    final Model model = httpRdfService.bodyToInternalModel(fedoraId, requestBodyStream,
557                            contentType, identifierConverter(), hasLenientPreferHeader());
558
559                    doInDbTxWithRetry(() -> {
560                        if (resourceExists) {
561                            replacePropertiesService.perform(transaction,
562                                    getUserPrincipal(),
563                                    fedoraId,
564                                    model);
565                        } else {
566                            createResourceService.perform(transaction, getUserPrincipal(), fedoraId, links, model);
567                            created.set(true);
568                        }
569                        transaction.commitIfShortLived();
570                    });
571                }
572
573            LOGGER.debug("Finished creating resource with path: {}", externalPath());
574
575            return createUpdateResponse(getFedoraResource(transaction, fedoraId), created.get());
576        } finally {
577            transaction.releaseResourceLocksIfShortLived();
578        }
579    }

581    /**
582     * Update an object using SPARQL-UPDATE
583     *
584     * @param requestBodyStream the request body stream
585     * @return 201
586     * @throws IOException if IO exception occurred
587     */
588    @PATCH
589    @Consumes({contentTypeSPARQLUpdate})
590    public Response updateSparql(final InputStream requestBodyStream)
591            throws IOException {
592        if (externalPath.contains("/" + FedoraTypes.FCR_VERSIONS)) {
593            handleRequestDisallowedOnMemento();
594
595            return status(METHOD_NOT_ALLOWED).build();
596        }
597
598        hasRestrictedPath(externalPath);
599
600        if (null == requestBodyStream) {
601            throw new BadRequestException("SPARQL-UPDATE requests must have content!");
602        }
603
604        if (resource() instanceof Binary) {
605            throw new BadRequestException(resource().getFedoraId().getFullIdPath() +
606                    " is not a valid object to receive a PATCH");
607        }
608
609        final var transaction = transaction();
610
611        try {
612            final String requestBody = IOUtils.toString(requestBodyStream, UTF_8);
613            if (isBlank(requestBody)) {
614                throw new BadRequestException("SPARQL-UPDATE requests must have content!");
615            }
616
617            evaluateRequestPreconditions(request, servletResponse, resource(), transaction);
618
619            LOGGER.info("PATCH for '{}'", externalPath);
620            final String newRequest = httpRdfService.patchRequestToInternalString(resource().getFedoraId(),
621                    requestBody, identifierConverter());
622            LOGGER.debug("PATCH request translated to '{}'", newRequest);
623
624            doInDbTxWithRetry(() -> {
625                patchResourcewithSparql(resource(), newRequest);
626                transaction.commitIfShortLived();
627            });
628
629            addCacheControlHeaders(servletResponse, reloadResource(), transaction);
630
631            return noContent().build();
632        } catch (final IllegalArgumentException iae) {
633            throw new BadRequestException(iae.getMessage());
634        } catch (final AccessDeniedException e) {
635            throw e;
636        } catch ( final RuntimeException ex ) {
637            final Throwable cause = ex.getCause();
638            if (cause instanceof PathNotFoundRuntimeException) {
639                // the sparql update referred to a repository resource that doesn't exist
640                throw new BadRequestException(cause.getMessage());
641            }
642            throw ex;
643        } finally {
644            transaction.releaseResourceLocksIfShortLived();
645        }
646    }

648    /**
649     * Creates a new object.
650     *
651     * This originally used application/octet-stream;qs=1001 as a workaround
652     * for JERSEY-2636, to ensure requests without a Content-Type get routed here.
653     * This qs value does not parse with newer versions of Jersey, as qs values
654     * must be between 0 and 1. We use qs=1.000 to mark where this historical
655     * anomaly had been.
656     *
657     * @param contentDispositionRaw the content Disposition value
658     * @param requestContentType the request content type
659     * @param slug the slug value
660     * @param requestBodyStream the request body stream
661     * @param rawLinks the link values
662     * @param digest the digest header
663     * @return 201
664     * @throws InvalidChecksumException if invalid checksum exception occurred
665     * @throws MalformedRdfException if malformed rdf exception occurred
666     * @throws UnsupportedAlgorithmException if an unsupported algorithm exception occurs
667     */
668    @POST
669    @Consumes({MediaType.APPLICATION_OCTET_STREAM + ";qs=1.000", WILDCARD})
670    @Produces({TURTLE_WITH_CHARSET + ";qs=1.0", JSON_LD + ";qs=0.8",
671            N3_WITH_CHARSET, N3_ALT2_WITH_CHARSET, RDF_XML, NTRIPLES, TEXT_PLAIN_WITH_CHARSET,
672            TEXT_HTML_WITH_CHARSET, "*/*"})
673    public Response createObject(@HeaderParam(CONTENT_DISPOSITION) final String contentDispositionRaw,
674                                 @HeaderParam(CONTENT_TYPE) final MediaType requestContentType,
675                                 @HeaderParam("Slug") final String slug,
676                                 final InputStream requestBodyStream,
677                                 @HeaderParam(LINK) final List<String> rawLinks,
678                                 @HeaderParam("Digest") final String digest)
679            throws InvalidChecksumException, MalformedRdfException, UnsupportedAlgorithmException {
680
681        final var decodedSlug = slug != null ? URLDecoder.decode(slug, UTF_8) : null;
682        final var transaction = transaction();
683
684        try {
685            final List<String> links = unpackLinks(rawLinks);
686
687            if (externalPath.contains("/" + FedoraTypes.FCR_VERSIONS)) {
688                handleRequestDisallowedOnMemento();
689
690                return status(METHOD_NOT_ALLOWED).build();
691            }
692
693            // If request is an external binary, verify link header before proceeding
694            final ExternalContent extContent = extContentHandlerFactory.createFromLinks(links);
695
696            final String interactionModel = checkInteractionModel(links);
697
698            final FedoraId fedoraId = identifierConverter().pathToInternalId(externalPath());
699            // If the resource doesn't exist and it's not a ghost node, throw an exception.
700            // Ghost node checking is done further down in the code and returns a 400 Bad Request error.
701            if (!doesResourceExist(transaction, fedoraId, false) && !isGhostNode(transaction, fedoraId)) {
702                throw new PathNotFoundRuntimeException(String.format("Path %s not found", fedoraId.getFullIdPath()));
703            }
704            final FedoraId newFedoraId = mintNewPid(fedoraId, decodedSlug);
705            final var providedContentType = getSimpleContentType(requestContentType);
706
707            LOGGER.info("POST to create resource with ID: {}, slug: {}", newFedoraId.getFullIdPath(), decodedSlug);
708
709            if (isBinary(interactionModel,
710                    providedContentType,
711                    requestBodyStream != null && providedContentType != null,
712                    extContent != null)) {
713                ensureArchivalGroupHeaderNotPresentForBinaries(links);
714
715                final Collection<URI> checksums = parseDigestHeader(digest);
716
717                final String originalFileName;
718                final long contentSize;
719
720                if (StringUtils.isNotBlank(contentDispositionRaw)) {
721                    final var contentDisposition = ContentDisposition.parse(contentDispositionRaw);
722                    originalFileName = contentDisposition.getFilename();
723                    contentSize = contentDisposition.getSize() == null ? -1L : contentDisposition.getSize();
724                } else {
725                    originalFileName = "";
726                    contentSize = -1L;
727                }
728
729                final var binaryType = requestContentType != null ?
730                        requestContentType : DEFAULT_NON_RDF_CONTENT_TYPE;
731                final var contentType = extContent == null ? binaryType.toString() : extContent.getContentType();
732
733                doInDbTx(() -> {
734                    createResourceService.perform(transaction,
735                            getUserPrincipal(),
736                            newFedoraId,
737                            contentType,
738                            originalFileName,
739                            contentSize,
740                            links,
741                            checksums,
742                            requestBodyStream,
743                            extContent);
744
745                    transaction.commitIfShortLived();
746                });
747            } else {
748                final var contentType = requestContentType != null ? requestContentType : DEFAULT_RDF_CONTENT_TYPE;
749                final Model model = httpRdfService.bodyToInternalModel(newFedoraId, requestBodyStream,
750                        contentType, identifierConverter(), hasLenientPreferHeader());
751
752                doInDbTxWithRetry(() -> {
753                    createResourceService.perform(transaction,
754                            getUserPrincipal(),
755                            newFedoraId,
756                            links,
757                            model);
758
759                    transaction.commitIfShortLived();
760                });
761            }
762
763            LOGGER.debug("Finished creating resource with path: {}", externalPath());
764
765            try {
766                final var resource = getFedoraResource(transaction, newFedoraId);
767                return createUpdateResponse(resource, true);
768            } catch (final PathNotFoundException e) {
769                throw new PathNotFoundRuntimeException(e.getMessage(), e);
770            }
771        } finally {
772            transaction.releaseResourceLocksIfShortLived();
773        }
774    }

801    /**
802     * Determine based on several factors whether the interaction model should be ldp:NonRdfSource
803     * @param interactionModel the interaction model from the links.
804     * @param contentType the content type.
805     * @param contentPresent is there a request body.
806     * @param contentExternal is there an external content header.
807     * @return Use ldp:NonRdfSource as the interaction model.
808     */
809    private boolean isBinary(final String interactionModel, final String contentType,
810                             final boolean contentPresent, final boolean contentExternal) {
811        final String simpleContentType = contentPresent ? contentType : null;
812        final boolean isRdfContent = isRdfContentType(simpleContentType);
813        return NON_RDF_SOURCE.getURI().equals(interactionModel) || contentExternal ||
814                (contentPresent && interactionModel == null && !isRdfContent);
815    }

817    private String handleWantDigestHeader(final Binary binary, final String wantDigest)
818            throws UnsupportedAlgorithmException {
819        // handle the Want-Digest header with fixity check
820        final Collection<String> preferredDigests = parseWantDigestHeader(wantDigest);
821        if (preferredDigests.isEmpty()) {
822            throw new UnsupportedAlgorithmException(
823                    "Unsupported digest algorithm provided in 'Want-Digest' header: " + wantDigest);
824        }
825
826        final Collection<URI> checksumResults = fixityService.getFixity(binary, preferredDigests);
827        return checksumResults.stream().map(uri -> uri.toString().replaceFirst("urn:", "")
828                .replaceFirst(":", "=").replaceFirst("sha1=", "sha=")).collect(Collectors.joining(","));
829    }

831    private static void ensureArchivalGroupHeaderNotPresentForBinaries(final List<String> links) {
832        if (links == null) {
833            return;
834        }
835
836        if (links.stream().map(Link::valueOf)
837                      .filter(l -> l.getUri().toString().equals(ARCHIVAL_GROUP.getURI()))
838                      .anyMatch(l -> l.getRel().equals("type"))) {
839            throw new ClientErrorException("Binary resources cannot be created as an" +
840                    " ArchiveGroup. Please remove the ArchiveGroup link header and try again", BAD_REQUEST);
841        }
842    }
843
844    private static String checkInteractionModel(final List<String> links) {
845        if (links == null) {
846            return null;
847        }
848
849        try {
850            for (final String link : links) {
851                final Link linq = Link.valueOf(link);
852                if ("type".equals(linq.getRel())) {
853                    //skip ArchivalGroup types
854                    if (linq.getUri().toString().equals(ARCHIVAL_GROUP.getURI())) {
855                        continue;
856                    }
857                    final Resource type = createResource(linq.getUri().toString());
858                    if (INTERACTION_MODEL_RESOURCES.contains(type)) {
859                        return type.getURI();
860                    } else if (type.equals(VERSIONED_RESOURCE)) {
861                        // skip if versioned resource link header
862                        // NB: the versioned resource header is used for enabling
863                        // versioning on a resource and is thus orthogonal to
864                        // issue of interaction models. Nevertheless, it is
865                        // a possible link header and, therefore, must be ignored.
866                    } else {
867                        LOGGER.info("Invalid interaction model: {}", type);
868                        throw new CannotCreateResourceException("Invalid interaction model: " + type);
869                    }
870                }
871            }
872        } catch (final RuntimeException e) {
873            if (e instanceof IllegalArgumentException || e instanceof UriBuilderException) {
874                throw new ClientErrorException("Invalid link specified: " + String.join(", ", links), BAD_REQUEST);
875            }
876            throw e;
877        }
878
879        return null;
880    }

882    /**
883     * Parse the RFC-3230 Want-Digest header value.
884     * @param wantDigest The Want-Digest header value with optional q value in format:
885     *    'md5', 'md5, sha', 'MD5;q=0.3, sha;q=1' etc.
886     * @return Digest algorithms that are supported
887     */
888    private static Collection<String> parseWantDigestHeader(final String wantDigest) {
889        final Map<String, Double> digestPairs = new HashMap<>();
890        try {
891            final List<String> algs = Splitter.on(',').omitEmptyStrings().trimResults().splitToList(wantDigest);
892            // Parse the optional q value with default 1.0, and 0 ignore. Format could be: SHA-1;qvalue=0.1
893            for (final String alg : algs) {
894                final String[] tokens = alg.split(";", 2);
895                final double qValue = tokens.length == 1 || !tokens[1].contains("=") ?
896                        1.0 : Double.parseDouble(tokens[1].split("=", 2)[1]);
897                digestPairs.put(tokens[0], qValue);
898            }
899
900            return digestPairs.entrySet().stream().filter(entry -> entry.getValue() > 0)
901                    .map(Map.Entry::getKey)
902                    .filter(DigestAlgorithm::isSupportedAlgorithm)
903                    .collect(Collectors.toSet());
904        } catch (final NumberFormatException e) {
905            throw new ClientErrorException("Invalid 'Want-Digest' header value: " + wantDigest, SC_BAD_REQUEST, e);
906        } catch (final RuntimeException e) {
907            if (e instanceof IllegalArgumentException) {
908                throw new ClientErrorException("Invalid 'Want-Digest' header value: " + wantDigest + "\n", BAD_REQUEST);
909            }
910            throw e;
911        }
912    }

921        }

924            FedoraTypes.FCR_VERSIONS, externalPath);
925    }

927    private FedoraId mintNewPid(final FedoraId fedoraId, final String slug) {
928        final String pid;
929
930        if (isGhostNode(transaction(), fedoraId)) {
931            LOGGER.debug("Resource with path {} is an immutable resource; it cannot be POSTed to.", fedoraId);
932            throw new CannotCreateResourceException("Cannot create resource as child of the immutable resource at " +
933                    fedoraId.getFullIdPath());
934        }
935        if (!isBlank(slug)) {
936            pid = slug;
937        } else if (pidMinter != null) {
938            pid = pidMinter.get();
939        } else {
940            pid = defaultPidMinter.get();
941        }
942
943        final FedoraId fullTestPath = fedoraId.resolve(pid);
944        hasRestrictedPath(fullTestPath.getFullIdPath());
945
946        if (doesResourceExist(transaction(), fullTestPath, true) || isGhostNode(transaction(), fullTestPath)) {
947            LOGGER.debug("Resource with path {} already exists or is an immutable resource; minting new path instead",
948                    fullTestPath);
949            return mintNewPid(fedoraId, null);
950        }
951
952        return fullTestPath;
953    }
