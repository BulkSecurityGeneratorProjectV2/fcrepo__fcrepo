59  /**
60   * A service that will translate the resourceURI to Fedora ID in the Rdf InputStream

62   * @author bseeger

64   * @since 2019-11-07

74      /**
75       * Convert internal IDs to external URIs
76       * @param extResourceId The external URI of the resource.
77       * @param stream The RDF stream to be translated.
78       * @param idTranslator The identifier converter.
79       * @return a converted RDF Stream.
80       */
81      public RdfStream bodyToExternalStream(final String extResourceId, final RdfStream stream,
82                                       final HttpIdentifierConverter idTranslator) {
83          return new DefaultRdfStream(NodeFactory.createURI(extResourceId), stream.map(t -> {
84              final Node subject = makeExternalNode(t.getSubject(), idTranslator);
85              final Node object = makeExternalNode(t.getObject(), idTranslator);
86              return new Triple(subject, t.getPredicate(), object);
87          }));
88      }
89  
90      /**
91       * Return a converted or original resource.
92       * @param resource The Node to be checked.
93       * @param identifierConverter A identifier converter.
94       * @return The resulting node.
95       */
96      private Node makeExternalNode(final Node resource, final HttpIdentifierConverter identifierConverter) {
97          if (resource.isURI() && identifierConverter.inInternalDomain(resource.toString())) {
98              return NodeFactory.createURI(identifierConverter.toExternalId(resource.toString()));
99          } else {
100             return resource;
101         }
102     }

104     /**
105      * Converts binary description IDs to their associated binary ID.
106      * @param uri
107      *   The URI of an external resource or the ID of an internal one.
108      * @param idTranslator
109      *   An identifier translator.
110      * @return
111      *   The same as input unless it is an internal binary description ID, then return the binary ID.
112      */
113     private String binaryConversion(final String uri, final HttpIdentifierConverter idTranslator) {
114         if (idTranslator.inInternalDomain(uri)) {
115             final var id = FedoraId.create(uri);
116             if (id.isDescription()) {
117                 return id.getFullDescribedId();
118             }
119         }
120         return uri;
121     }

123     /**
124      * Parse the request body to a Model, with the URI to Fedora ID translations done.
125      *
126      * @param extResourceId the external ID of the Fedora resource
127      * @param stream the input stream containing the RDF
128      * @param contentType the media type of the RDF
129      * @param idTranslator the identifier convert
130      * @param lenientHandling whether the request included a handling=lenient prefer header.
131      * @return RdfStream containing triples from request body, with Fedora IDs in them
132      * @throws MalformedRdfException in case rdf json cannot be parsed
133      * @throws BadRequestException in the case where the RDF syntax is bad
134      */
135     public Model bodyToInternalModel(final FedoraId extResourceId, final InputStream stream,
136                                      final MediaType contentType, final HttpIdentifierConverter idTranslator,
137                                      final boolean lenientHandling)
138                                      throws RepositoryRuntimeException, BadRequestException {
139         final List<ConstraintViolationException> exceptions = new ArrayList<>();
140         final String externalURI = idTranslator.toExternalId(extResourceId.getFullDescribedId());
141         final Model model = parseBodyAsModel(stream, contentType, externalURI);
142         final List<Statement> insertStatements = new ArrayList<>();
143         final StmtIterator stmtIterator = model.listStatements();
144 
145         while (stmtIterator.hasNext()) {
146             final Statement stmt = stmtIterator.nextStatement();
147             if (lenientHandling && stmtIsServerManaged(stmt) &&
148                     fedoraPropsConfig.getServerManagedPropsMode().equals(STRICT)) {
149                 // Remove any statement that touches a server managed property or namespace.
150                 stmtIterator.remove();
151             } else {
152                 try {
153                     checkForDisallowedRdf(stmt);
154                 } catch (final RelaxableServerManagedPropertyException exc) {
155                     if (fedoraPropsConfig.getServerManagedPropsMode().equals(STRICT)) {
156                         exceptions.add(exc);
157                         continue;
158                     }
159                 } catch (final ServerManagedTypeException | ServerManagedPropertyException exc) {
160                     if (lenientHandling) {
161                         // Remove the invalid statement because client specified lenient handling.
162                         stmtIterator.remove();
163                     } else {
164                         exceptions.add(exc);
165                     }
166                     continue;
167                 }
168                 if (stmt.getSubject().isURIResource()) {
169                     final String originalSubj = stmt.getSubject().getURI();
170                     final String subj = binaryConversion(idTranslator.translateUri(originalSubj), idTranslator);
171 
172                     RDFNode obj = stmt.getObject();
173                     if (stmt.getObject().isURIResource()) {
174                         final String objString = stmt.getObject().asResource().getURI();
175                         final String objUri = binaryConversion(idTranslator.translateUri(objString), idTranslator);
176                         obj = model.createResource(objUri);
177                     }
178 
179                     if (!subj.equals(originalSubj) || !obj.equals(stmt.getObject())) {
180                         insertStatements.add(new StatementImpl(model.createResource(subj), stmt.getPredicate(), obj));
181 
182                         stmtIterator.remove();
183                     }
184                 } else {
185                     log.debug("Subject {} is not a URI resource, skipping", stmt.getSubject());
186                 }
187             }
188         }
189 
190         if (!exceptions.isEmpty()) {
191             throw new MultipleConstraintViolationException(exceptions);
192         }
193 
194         model.add(insertStatements);
195 
196         log.debug("Model: {}", model);
197         return model;
198     }

200     /**
201      * Takes a PATCH request body and translates any subjects and objects that are in the domain of the repository
202      * to use internal IDs.
203      * @param resourceId the internal ID of the current resource.
204      * @param requestBody the request body.
205      * @param idTranslator an ID converter for the current context.
206      * @return the converted PATCH request.
207      */
208     public String patchRequestToInternalString(final FedoraId resourceId, final String requestBody,
209                                                final HttpIdentifierConverter idTranslator) {
210         final String externalURI = idTranslator.toExternalId(resourceId.getFullDescribedId());
211         final UpdateRequest request = UpdateFactory.create(requestBody, externalURI);
212         final List<Update> updates = request.getOperations();
213         final SparqlTranslateVisitor visitor = new SparqlTranslateVisitor(idTranslator, fedoraPropsConfig);
214         for (final Update update : updates) {
215             update.visit(visitor);
216         }
217         return visitor.getTranslatedRequest().toString();
218     }

220     /**
221      * Parse the request body as a Model.
222      *
223      * @param requestBodyStream rdf request body
224      * @param contentType content type of body
225      * @param extResourceId the external ID of the Fedora resource
226      * @return Model containing triples from request body
227      * @throws MalformedRdfException in case rdf json cannot be parsed
228      * @throws BadRequestException in the case where the RDF syntax is bad
229      */
230     protected static Model parseBodyAsModel(final InputStream requestBodyStream,
231                                          final MediaType contentType,
232                                          final String extResourceId) throws BadRequestException,
233                                          RepositoryRuntimeException {
234 
235         if (requestBodyStream == null) {
236             return null;
237         }
238 
239         // The 'contentTypeToLang()' method will not accept 'charset' parameters
240         String contentTypeWithoutCharset = contentType.toString();
241         if (contentType.getParameters().containsKey("charset")) {
242             contentTypeWithoutCharset = contentType.getType() + "/" + contentType.getSubtype();
243         }
244 
245         final Lang format = contentTypeToLang(contentTypeWithoutCharset);
246         if (format == null) {
247             // No valid RDF format for the mimeType.
248             throw new UnsupportedMediaTypeException("Media type " + contentTypeWithoutCharset + " is not a valid RDF " +
249                     "format");
250         }
251         try {
252             final Model inputModel = createDefaultModel();
253             inputModel.read(requestBodyStream, extResourceId, format.getName().toUpperCase());
254             return inputModel;
255         } catch (final RiotException e) {
256             throw new BadRequestException("RDF was not parsable: " + e.getMessage(), e);
257 
258         } catch (final RuntimeIOException e) {
259             if (e.getCause() instanceof JsonParseException) {
260                 final var cause = e.getCause();
261                 throw new MalformedRdfException(cause.getMessage(), cause);
262             }
263             throw new RepositoryRuntimeException(e.getMessage(), e);
264         }
265     }

267     /**
268      * Checks if the RDF contains any disallowed statements.
269      * @param statement a statement from the incoming RDF.
270      */
271     private static void checkForDisallowedRdf(final Statement statement) {
272         checkTripleForDisallowed(statement.asTriple());
273     }
274 
275     /**
276      * Does the statement's triple touch any server managed properties / namespaces.
277      * i.e.
278      * - has a rdf:type with an object which is in a managed namespace
279      * - has a predicate which is in a managed namespace.
280      *
281      * @param statement the statement to check
282      * @return Return true if this does touch a server managed property or namespace.
283      */
284     private boolean stmtIsServerManaged(final Statement statement) {
285         final Triple triple = statement.asTriple();
286         return restrictedType.test(triple) || isManagedPredicate.test(createProperty(triple.getPredicate().getURI()));
287     }
