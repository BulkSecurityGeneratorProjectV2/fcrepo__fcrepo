6   package org.fcrepo.event.serialization;
7   
8   import static com.fasterxml.jackson.databind.SerializationFeature.WRITE_DATES_AS_TIMESTAMPS;
9   import static org.fcrepo.event.serialization.JsonLDEventMessage.from;
10  import static org.slf4j.LoggerFactory.getLogger;
11  
12  import com.fasterxml.jackson.core.JsonProcessingException;
13  import com.fasterxml.jackson.databind.ObjectMapper;
14  import com.fasterxml.jackson.datatype.jsr310.JavaTimeModule;
15  import org.fcrepo.kernel.api.observer.Event;
16  import org.slf4j.Logger;
17  
18  /**
19   * Some serialization utilities for Event objects
20   * @author acoburn
21   */
22  public class JsonLDSerializer implements EventSerializer {
23  
24      private static final Logger LOGGER = getLogger(JsonLDSerializer.class);
25  
26      private static final ObjectMapper MAPPER = new ObjectMapper();
27  
28      /**
29       * Create a new JSON-LD Event Serializer
30       */
31      public JsonLDSerializer() {
32          // newer versions of jackson rename this to `JavaTimeModule`
33          MAPPER.registerModule(new JavaTimeModule());
34          MAPPER.configure(WRITE_DATES_AS_TIMESTAMPS, false);
35      }
36  
37      /**
38       * Serialize a Event into a JSON String
39       * @param evt the Fedora event
40       * @return a JSON string
41       */
42      @Override
43      public String serialize(final Event evt) {
44          try {
45              return MAPPER.writeValueAsString(from(evt));
46          } catch (final JsonProcessingException ex) {
47              LOGGER.error("Error processing JSON: {}", ex.getMessage());
48              return null;
49          }
50      }
51  }
