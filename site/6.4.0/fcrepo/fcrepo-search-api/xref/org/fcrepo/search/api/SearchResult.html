6   package org.fcrepo.search.api;
7   
8   import com.fasterxml.jackson.annotation.JsonProperty;
9   
10  import java.util.List;
11  import java.util.Map;
12  
13  import static com.google.common.base.Preconditions.checkNotNull;
14  
15  /**
16   * This class holds the search result data for a single page.
17   * @author dbernstein
18   */
19  public class SearchResult {
20      @JsonProperty
21      private PaginationInfo pagination;
22      @JsonProperty
23      private List<Map<String, Object>> items;
24  
25      /**
26       * Default Constructor
27       */
28      public SearchResult() { }
29      /**
30       * Constructor
31       *
32       * @param items      The individual search result items
33       * @param pagination The pagination info
34       */
35      public SearchResult(final List<Map<String, Object>> items, final PaginationInfo pagination) {
36          checkNotNull(items, "items cannot be null");
37          checkNotNull(pagination, "pagination cannot be null");
38          this.items = items;
39          this.pagination = pagination;
40      }
41  
42      /**
43       * The pagination information.
44       * @return The pagination info
45       */
46      public PaginationInfo getPagination() {
47          return this.pagination;
48      }
49  
50      /**
51       * The list of items returned by the search operation associated with the page indicated by the pagination info.
52       * @return The list of items
53       */
54      public List<Map<String, Object>> getItems() {
55          return this.items;
56      }
57  }
