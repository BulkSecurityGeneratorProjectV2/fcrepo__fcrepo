6   package org.fcrepo.auth.common;
7   
8   import static org.slf4j.LoggerFactory.getLogger;
9   
10  import java.util.Set;
11  
12  import org.apache.shiro.authc.AuthenticationException;
13  import org.apache.shiro.authc.AuthenticationInfo;
14  import org.apache.shiro.authc.AuthenticationToken;
15  import org.apache.shiro.authc.SimpleAuthenticationInfo;
16  import org.apache.shiro.realm.AuthenticatingRealm;
17  import org.apache.shiro.subject.SimplePrincipalCollection;
18  import org.fcrepo.auth.common.ContainerRolesPrincipalProvider.ContainerRolesPrincipal;
19  import org.slf4j.Logger;
20  
21  /**
22   * @author peichman
23   */
24  public class ServletContainerAuthenticatingRealm extends AuthenticatingRealm {
25  
26      private static final Logger log = getLogger(ServletContainerAuthenticatingRealm.class);
27  
28      @Override
29      public String getName() {
30          return "servlet container authentication";
31      }
32  
33      @Override
34      protected AuthenticationInfo doGetAuthenticationInfo(final AuthenticationToken token)
35              throws AuthenticationException {
36          final ContainerAuthToken authToken = (ContainerAuthToken) token;
37          final SimplePrincipalCollection principals = new SimplePrincipalCollection();
38          log.debug("Creating principals from servlet container principal and roles");
39          // container-managed auth username
40          principals.add(authToken.getPrincipal(), getName());
41          // container-managed auth roles
42          final Set<ContainerRolesPrincipal> roles = authToken.getRoles();
43          if (!roles.isEmpty()) {
44              principals.addAll(roles, getName());
45          }
46          return new SimpleAuthenticationInfo(principals, ContainerAuthToken.AUTHORIZED);
47      }
48  
49      @Override
50      public boolean supports(final AuthenticationToken token) {
51          return token instanceof ContainerAuthToken;
52      }
53  
54  }
