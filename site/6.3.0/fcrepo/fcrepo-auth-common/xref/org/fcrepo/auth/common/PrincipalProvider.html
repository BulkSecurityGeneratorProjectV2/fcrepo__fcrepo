6   package org.fcrepo.auth.common;
7   
8   import javax.servlet.Filter;
9   import javax.servlet.http.HttpServletRequest;
10  
11  import java.security.Principal;
12  import java.util.Set;
13  
14  /**
15   * This interface provides a way for authentication code to communicate generic
16   * credentials to authorization delegates. An implementation of this interface
17   * could perform a query to determine group membership, for example.
18   * <p>
19   * The ServletContainerAuthenticationProvider's principalProviders set may be
20   * configured with zero or more instances of implementations of this interface,
21   * which it will consult during authentication. The union of the results will be
22   * assigned to the FEDORA_ALL_PRINCIPALS session attribute.
23   * </p>
24   *
25   * @author Gregory Jansen
26   * @see HttpHeaderPrincipalProvider
27   */
28  public interface PrincipalProvider extends Filter {
29  
30      /**
31       * Extract principals from the provided HttpServletRequest.
32       * <p>
33       * If no principals can be extracted, implementations of this method
34       * should return the empty set rather than null.
35       * </p>
36       *
37       * @param request the request
38       * @return a set of security principals
39       */
40      Set<Principal> getPrincipals(HttpServletRequest request);
41  
42  }
