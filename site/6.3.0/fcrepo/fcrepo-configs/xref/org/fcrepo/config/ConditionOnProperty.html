7   package org.fcrepo.config;
8   
9   import java.util.Objects;
10  
11  import org.slf4j.Logger;
12  import org.slf4j.LoggerFactory;
13  import org.springframework.context.annotation.ConditionContext;
14  import org.springframework.context.annotation.ConfigurationCondition;
15  import org.springframework.core.type.AnnotatedTypeMetadata;
16  
17  /**
18   * This condition enables a bean/configuration when the specified property matches the expected value
19   *
20   * Implementations must provide a no-arg constructor.
21   *
22   * @author pwinckles
23   */
24  public abstract class ConditionOnProperty<T> implements ConfigurationCondition {
25  
26      private static final Logger LOGGER = LoggerFactory.getLogger(ConditionOnProperty.class);
27  
28      private final String name;
29      private final T expected;
30      private final T defaultValue;
31      private final Class<T> clazz;
32  
33      public ConditionOnProperty(final String name, final T expected, final T defaultValue, final Class<T> clazz) {
34          this.name = name;
35          this.expected = expected;
36          this.defaultValue = defaultValue;
37          this.clazz = clazz;
38      }
39  
40      @Override
41      public boolean matches(final ConditionContext context, final AnnotatedTypeMetadata metadata) {
42          LOGGER.debug("Prop {}: {}", name, context.getEnvironment().getProperty(name));
43          return Objects.equals(expected, context.getEnvironment().getProperty(name, clazz, defaultValue));
44      }
45  
46      @Override
47      public ConfigurationPhase getConfigurationPhase() {
48          // This forces spring to not evaluate these conditions until after it has loaded other @Configuration classes,
49          // ensuring that the properties have been loaded.
50          return ConfigurationPhase.REGISTER_BEAN;
51      }
52  }
