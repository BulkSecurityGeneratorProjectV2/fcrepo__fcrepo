007package org.fcrepo.config;
008
009import java.nio.file.Path;
010import java.util.List;
011
012import org.springframework.beans.factory.annotation.Value;
013import org.springframework.context.annotation.Configuration;
014
015/**
016 * Auth related configuration properties
017 *
018 * @author pwinckles
019 */
020@Configuration
021public class AuthPropsConfig extends BasePropsConfig {
022
023    public static final String FCREPO_AUTH_ENABLED = "fcrepo.auth.enabled";
024    public static final String FCREPO_AUTH_PRINCIPAL_HEADER_ENABLED = "fcrepo.auth.principal.header.enabled";
025    private static final String FCREPO_AUTH_PRINCIPAL_HEADER_NAME = "fcrepo.auth.principal.header.name";
026    private static final String FCREPO_AUTH_PRINCIPAL_HEADER_SEPARATOR = "fcrepo.auth.principal.header.separator";
027    public static final String FCREPO_AUTH_PRINCIPAL_ROLES_ENABLED = "fcrepo.auth.principal.roles.enabled";
028    private static final String FCREPO_AUTH_PRINCIPAL_ROLES_LIST = "fcrepo.auth.principal.roles.list";
029    public static final String FCREPO_AUTH_PRINCIPAL_DELEGATE_ENABLED = "fcrepo.auth.principal.delegate.enabled";
030    private static final String FCREPO_GROUP_AGENT_BASE_URI = "fcrepo.auth.webac.groupAgent.baseUri";
031    private static final String FCREPO_USER_AGENT_BASE_URI = "fcrepo.auth.webac.userAgent.baseUri";
032    private static final String FCREPO_ROOT_AUTH_ACL = "fcrepo.auth.webac.authorization";
033
034    @Value("${" + FCREPO_ROOT_AUTH_ACL + ":#{null}}")
035    private Path rootAuthAclPath;
036
037    @Value("${" + FCREPO_USER_AGENT_BASE_URI + ":#{null}}")
038    private String userAgentBaseUri;
039    @Value("${" + FCREPO_GROUP_AGENT_BASE_URI + ":#{null}}")
040    private String groupAgentBaseUri;
041
042    @Value("${" + FCREPO_AUTH_PRINCIPAL_DELEGATE_ENABLED + ":true}")
043    private boolean authPrincipalDelegateEnabled;
044
045    @Value("${" + FCREPO_AUTH_PRINCIPAL_HEADER_ENABLED + ":false}")
046    private boolean authPrincipalHeaderEnabled;
047    @Value("${" + FCREPO_AUTH_PRINCIPAL_HEADER_NAME + ":some-header}")
048    private String authPrincipalHeaderName;
049    @Value("${" + FCREPO_AUTH_PRINCIPAL_HEADER_SEPARATOR + ":,}")
050    private String authPrincipalHeaderSeparator;
051
052    @Value("${" + FCREPO_AUTH_PRINCIPAL_ROLES_ENABLED + ":false}")
053    private boolean authPrincipalRolesEnabled;
054    @Value("#{'${" + FCREPO_AUTH_PRINCIPAL_ROLES_LIST + ":tomcat-role-1,tomcat-role-2}'.split(',')}")
055    private List<String> authPrincipalRolesList;
056
057    /**
058     * @return the path to the root auth acl to use instead of the default
059     */
060    public Path getRootAuthAclPath() {
061        return rootAuthAclPath;
062    }
063
064    /**
065     * @param rootAuthAclPath path to custom root auth acl
066     */
067    public void setRootAuthAclPath(final Path rootAuthAclPath) {
068        this.rootAuthAclPath = rootAuthAclPath;
069    }
070
071    /**
072     * @return the user agent base uri, if specified
073     */
074    public String getUserAgentBaseUri() {
075        return userAgentBaseUri;
076    }
077
078    /**
079     * @return the user agent base uri, if specified
080     */
081    public String getGroupAgentBaseUri() {
082        return groupAgentBaseUri;
083    }
084
085    /**
086     * @return the header name for the auth principal header provider
087     */
088    public String getAuthPrincipalHeaderName() {
089        return authPrincipalHeaderName;
090    }
091
092    /**
093     * @return the separator for the auth principal header provider
094     */
095    public String getAuthPrincipalHeaderSeparator() {
096        return authPrincipalHeaderSeparator;
097    }
098
099    /**
100     * @return the list of auth roles
101     */
102    public List<String> getAuthPrincipalRolesList() {
103        return authPrincipalRolesList;
104    }
105
106    /**
107     * @return header principal provider enabled
108     */
109    public boolean isAuthPrincipalHeaderEnabled() {
110        return authPrincipalHeaderEnabled;
111    }
112
113    /**
114     * @return roles principal provider enabled
115     */
116    public boolean isAuthPrincipalRolesEnabled() {
117        return authPrincipalRolesEnabled;
118    }
119
120    /**
121     * @return delegate principal provider enabled
122     */
123    public boolean isAuthPrincipalDelegateEnabled() {
124        return authPrincipalDelegateEnabled;
125    }
126
127}
