007package org.fcrepo.config;
008
009/**
010 * Indicates what storage backend to use.
011 *
012 * @author pwinckles
013 */
014public enum Storage {
015
016    OCFL_FILESYSTEM("ocfl-fs"),
017    OCFL_S3("ocfl-s3");
018
019    private final String value;
020
021    Storage(final String value) {
022        this.value = value;
023    }
024
025    public String getValue() {
026        return value;
027    }
028
029    public static Storage fromString(final String value) {
030        for (final var storage : values()) {
031            if (storage.value.equalsIgnoreCase(value)) {
032                return storage;
033            }
034        }
035        throw new IllegalArgumentException("Unknown storage: " + value);
036    }
037
038    @Override
039    public String toString() {
040        return value;
041    }
042}
