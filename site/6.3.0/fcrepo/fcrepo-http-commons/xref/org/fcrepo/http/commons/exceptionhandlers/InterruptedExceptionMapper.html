6   package org.fcrepo.http.commons.exceptionhandlers;
7   
8   import org.fcrepo.kernel.api.exception.InterruptedRuntimeException;
9   import org.slf4j.Logger;
10  
11  import javax.ws.rs.core.Response;
12  import javax.ws.rs.ext.ExceptionMapper;
13  import javax.ws.rs.ext.Provider;
14  
15  import static javax.ws.rs.core.Response.Status.SERVICE_UNAVAILABLE;
16  import static javax.ws.rs.core.Response.status;
17  import static org.slf4j.LoggerFactory.getLogger;
18  import static org.fcrepo.http.commons.domain.RDFMediaType.TEXT_PLAIN_WITH_CHARSET;
19  
20  /**
21   * If an HTTP request's processing is interrupted, return an HTTP 503 Service Unavailable.
22   *
23   * @author Mike Durbin
24   */
25  @Provider
26  public class InterruptedExceptionMapper implements
27          ExceptionMapper<InterruptedRuntimeException>, ExceptionDebugLogging {
28  
29      private static final Logger LOGGER = getLogger(InterruptedExceptionMapper.class);
30  
31      @Override
32      public Response toResponse(final InterruptedRuntimeException e) {
33          LOGGER.warn("Service interrupted", e);
34          return status(SERVICE_UNAVAILABLE).entity(e.getMessage()).type(TEXT_PLAIN_WITH_CHARSET).build();
35      }
36  }
