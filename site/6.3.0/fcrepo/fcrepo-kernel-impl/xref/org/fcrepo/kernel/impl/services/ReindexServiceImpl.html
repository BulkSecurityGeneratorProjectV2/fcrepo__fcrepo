6   package org.fcrepo.kernel.impl.services;
7   
8   import org.fcrepo.kernel.api.Transaction;
9   import org.fcrepo.kernel.api.TransactionManager;
10  import org.fcrepo.kernel.api.identifiers.FedoraId;
11  import org.fcrepo.kernel.api.operations.ReindexResourceOperationFactory;
12  import org.fcrepo.kernel.api.services.ReindexService;
13  import org.fcrepo.persistence.api.PersistentStorageSessionManager;
14  import org.springframework.stereotype.Component;
15  
16  import javax.inject.Inject;
17  
18  /**
19   * Implementation of {@link org.fcrepo.kernel.api.services.ReindexService}
20   *
21   * @author dbernstein
22   */
23  @Component
24  public class ReindexServiceImpl extends AbstractService implements ReindexService {
25  
26      @Inject
27      private TransactionManager transactionManager;
28  
29      @Inject
30      private PersistentStorageSessionManager persistentStorageSessionManager;
31  
32      @Inject
33      private ReindexResourceOperationFactory resourceOperationFactory;
34  
35      @Override
36      public void reindexByFedoraId(final Transaction transaction, final String principal, final FedoraId fedoraId) {
37          final var tx = transactionManager.get(transaction.getId());
38          final var psession = persistentStorageSessionManager.getSession(transaction);
39          final var operation = resourceOperationFactory.create(transaction, fedoraId)
40                  .userPrincipal(principal).build();
41          tx.lockResource(fedoraId);
42  
43          psession.persist(operation);
44      }
45  }
