30  /**
31   * Implementation of the ResourceTripleService
32   * @author whikloj
33   * @since 6.0.0
34   */
35  @Component
36  public class ResourceTripleServiceImpl implements ResourceTripleService {
37  
38      @Inject
39      private ManagedPropertiesService managedPropertiesService;
40  
41      @Inject
42      private ContainmentTriplesService containmentTriplesService;
43  
44      @Autowired
45      @Qualifier("referenceService")
46      private ReferenceService referenceService;
47  
48      @Inject
49      private MembershipService membershipService;
50  
51      @Override
52      public Stream<Triple> getResourceTriples(final Transaction tx, final FedoraResource resource,
53                                               final LdpTriplePreferences preferences, final int limit) {
54          final List<Stream<Triple>> streams = new ArrayList<>();
55  
56          // Provide user RDF if we didn't ask for omit=ldp:PreferMinimalContainer.
57          if (preferences.displayUserRdf()) {
58              streams.add(resource.getTriples());
59          }
60          // Provide server-managed triples if we didn't ask for omit=fedora:ServerManaged or
61          // omit=ldp:PreferMinimalContainer
62          if (preferences.displayServerManaged()) {
63              streams.add(this.managedPropertiesService.get(resource));
64          }
65          // containment triples about this resource, return by default. Containment is server managed so also
66          // check for that prefer tag.
67          if (preferences.displayContainment()) {
68              if (limit == -1) {
69                  streams.add(this.containmentTriplesService.get(tx, resource));
70              } else {
71                  streams.add(this.containmentTriplesService.get(tx, resource).limit(limit));
72              }
73          }
74  
75          // LDP container membership triples for this resource, returned by default. Membership is server managed so
76          // also check that tag.
77          if (preferences.displayMembership()) {
78              streams.add(membershipService.getMembership(tx, resource.getFedoraId()));
79          }
80  
81          // Include inbound references to this object, NOT returned by default.
82          if (preferences.displayReferences()) {
83              streams.add(referenceService.getInboundReferences(tx, resource));
84          }
85  
86          return streams.stream().reduce(empty(), Stream::concat);
87      }
88  
89  }
