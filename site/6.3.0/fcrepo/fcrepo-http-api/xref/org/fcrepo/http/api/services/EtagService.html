8   import static org.slf4j.LoggerFactory.getLogger;
9   
10  import java.util.Collection;
11  import java.util.stream.Collectors;
12  
13  import javax.inject.Inject;
14  import javax.ws.rs.core.MediaType;
15  
16  import org.fcrepo.kernel.api.ContainmentIndex;
17  import org.fcrepo.kernel.api.Transaction;
18  import org.fcrepo.kernel.api.models.FedoraResource;
19  import org.fcrepo.kernel.api.rdf.LdpTriplePreferences;
20  import org.fcrepo.kernel.api.services.MembershipService;
21  
22  import org.apache.commons.codec.digest.DigestUtils;
23  import org.slf4j.Logger;
24  import org.springframework.stereotype.Component;
25  
26  /**
27   * Service for computing etags for request responses
28   *
29   * @author bbpennel
30   */
31  @Component
32  public class EtagService {
33      private static final Logger LOGGER = getLogger(EtagService.class);
34  
35      @Inject
36      private ContainmentIndex containmentIndex;
37  
38      @Inject
39      private MembershipService membershipService;
40  
41      /**
42       * Produces etag for a request for an RDF resource. It is based on factors related to the
43       * current state of the resource, as well as request options which change the
44       * representation of the resource.
45       *
46       * @param transaction transaction
47       * @param resource resource
48       * @param prefers LDP preference headers for the request
49       * @param acceptableMediaTypes collection of acceptable media types for the response
50       * @return etag for the request
51       */
52      public String getRdfResourceEtag(final Transaction transaction, final FedoraResource resource,
53                                       final LdpTriplePreferences prefers,
54                                       final Collection<MediaType> acceptableMediaTypes) {
55          // Start etag based on the current state of the fedora resource, using the state token
56          final StringBuilder etag = new StringBuilder(resource.getStateToken());
57  
58          // Factor in the requested mimetype(s)
59          final String mimetype = acceptableMediaTypes.stream()
60                  .map(MediaType::toString)
61                  .sorted()
62                  .collect(Collectors.joining(";"));
63          addComponent(etag, mimetype);
64  
65          // Factor in preferences which change which triples are included in the response
66          etag.append('|');
67          if (prefers.displayContainment()) {
68              final var lastUpdated = containmentIndex.containmentLastUpdated(transaction, resource.getFedoraId());
69              if (lastUpdated != null) {
70                  etag.append(lastUpdated);
71              }
72          }
73          etag.append('|');
74          if (prefers.displayMembership()) {
75              final var lastUpdated = membershipService.getLastUpdatedTimestamp(transaction, resource.getFedoraId());
76              if (lastUpdated != null) {
77                  etag.append(lastUpdated);
78              }
79          }
80          addComponent(etag, prefers.displayEmbed());
81          addComponent(etag, prefers.displayUserRdf());
82          addComponent(etag, prefers.displayReferences());
83          addComponent(etag, prefers.displayServerManaged());
84  
85          // Compute a digest of all these components to use as the etag
86          final String etagMd5 = DigestUtils.md5Hex(etag.toString()).toUpperCase();
87          LOGGER.debug("Produced etag {} for {} from {}", etagMd5, resource.getId(), etag);
88  
89          return etagMd5;
90      }
91  
92      private void addComponent(final StringBuilder etag, final Object component) {
93          etag.append('|').append(component);
94      }
95  }
