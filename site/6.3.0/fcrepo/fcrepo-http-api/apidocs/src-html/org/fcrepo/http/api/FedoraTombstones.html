006package org.fcrepo.http.api;
007
008import com.google.common.annotations.VisibleForTesting;
009import io.micrometer.core.annotation.Timed;
010
011import org.fcrepo.kernel.api.exception.PathNotFoundException;
012import org.fcrepo.kernel.api.exception.PathNotFoundRuntimeException;
013import org.fcrepo.kernel.api.identifiers.FedoraId;
014import org.fcrepo.kernel.api.models.FedoraResource;
015import org.fcrepo.kernel.api.models.Tombstone;
016import org.fcrepo.kernel.api.services.PurgeResourceService;
017import org.slf4j.Logger;
018import org.springframework.context.annotation.Scope;
019
020import javax.inject.Inject;
021import javax.ws.rs.DELETE;
022import javax.ws.rs.GET;
023import javax.ws.rs.OPTIONS;
024import javax.ws.rs.POST;
025import javax.ws.rs.PUT;
026import javax.ws.rs.Path;
027import javax.ws.rs.PathParam;
028import javax.ws.rs.core.Response;
029
030import static javax.ws.rs.core.HttpHeaders.ALLOW;
031import static javax.ws.rs.core.Response.noContent;
032import static org.slf4j.LoggerFactory.getLogger;
033
034/**
035 * CRUD operations on Fedora tombstones
036 *
037 * @author cbeer
038 */
039@Timed
040@Scope("request")
041@Path("/{path: .*}/fcr:tombstone")
042public class FedoraTombstones extends ContentExposingResource {
043
044    private static final Logger LOGGER = getLogger(FedoraTombstones.class);
045
046    @PathParam("path") protected String externalPath;
047
048    @Inject
049    private PurgeResourceService purgeResourceService;
050
051    /**
052     * Default JAX-RS entry point
053     */
054    public FedoraTombstones() {
055        super();
056    }
057
058    /**
059     * Create a new FedoraTombstones instance for a given path
060     * @param externalPath the external path
061     */
062    @VisibleForTesting
063    public FedoraTombstones(final String externalPath) {
064        this.externalPath = externalPath;
065    }
066
067    /**
068     * Delete a tombstone resource (freeing the original resource to be reused)
069     * @return the free resource
070     */
071    @DELETE
072    public Response delete() {
073        final FedoraResource resource = resource();
074        if (!(resource instanceof Tombstone)) {
075            // If the resource is not deleted there is no tombstone.
076            return Response.status(Response.Status.NOT_FOUND).build();
077        }
078        try {
079            final Tombstone tombstone = (Tombstone) resource;
080            LOGGER.info("Delete tombstone: {}", resource.getFedoraId());
081            doInDbTxWithRetry(() -> {
082                purgeResourceService.perform(transaction(), tombstone.getDeletedObject(), getUserPrincipal());
083                transaction().commitIfShortLived();
084            });
085            return noContent().build();
086        } finally {
087            transaction().releaseResourceLocksIfShortLived();
088        }
089    }
090
091    /*
092     * These methods are disallowed, but need to exist here or the path gets caught by the FedoraLdp path matcher.
093     */
094    @GET
095    public Response get() {
096        return methodNotAllowed();
097    }
098
099    @POST
100    public Response post() {
101        return methodNotAllowed();
102    }
103    @PUT
104    public Response put() {
105        return methodNotAllowed();
106    }
107
108    @OPTIONS
109    public Response options() {
110        return Response.ok().header(ALLOW, "DELETE").build();
111    }
112
113    @Override
114    protected FedoraResource resource() {
115        final FedoraId resourceId = identifierConverter().pathToInternalId(externalPath);
116        try {
117            return getFedoraResource(transaction(), resourceId);
118        } catch (final PathNotFoundException e) {
119            throw new PathNotFoundRuntimeException(e.getMessage(), e);
120        }
121    }
122
123    @Override
124    protected String externalPath() {
125        return null;
126    }
127
128    private Response methodNotAllowed() {
129        return Response.status(Response.Status.METHOD_NOT_ALLOWED).header(ALLOW, "DELETE").build();
130    }
131}
