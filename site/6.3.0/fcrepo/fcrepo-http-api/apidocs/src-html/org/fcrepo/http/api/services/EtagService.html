006package org.fcrepo.http.api.services;
007
008import static org.slf4j.LoggerFactory.getLogger;
009
010import java.util.Collection;
011import java.util.stream.Collectors;
012
013import javax.inject.Inject;
014import javax.ws.rs.core.MediaType;
015
016import org.fcrepo.kernel.api.ContainmentIndex;
017import org.fcrepo.kernel.api.Transaction;
018import org.fcrepo.kernel.api.models.FedoraResource;
019import org.fcrepo.kernel.api.rdf.LdpTriplePreferences;
020import org.fcrepo.kernel.api.services.MembershipService;
021
022import org.apache.commons.codec.digest.DigestUtils;
023import org.slf4j.Logger;
024import org.springframework.stereotype.Component;
025
026/**
027 * Service for computing etags for request responses
028 *
029 * @author bbpennel
030 */
031@Component
032public class EtagService {
033    private static final Logger LOGGER = getLogger(EtagService.class);
034
035    @Inject
036    private ContainmentIndex containmentIndex;
037
038    @Inject
039    private MembershipService membershipService;
040
041    /**
042     * Produces etag for a request for an RDF resource. It is based on factors related to the
043     * current state of the resource, as well as request options which change the
044     * representation of the resource.
045     *
046     * @param transaction transaction
047     * @param resource resource
048     * @param prefers LDP preference headers for the request
049     * @param acceptableMediaTypes collection of acceptable media types for the response
050     * @return etag for the request
051     */
052    public String getRdfResourceEtag(final Transaction transaction, final FedoraResource resource,
053                                     final LdpTriplePreferences prefers,
054                                     final Collection<MediaType> acceptableMediaTypes) {
055        // Start etag based on the current state of the fedora resource, using the state token
056        final StringBuilder etag = new StringBuilder(resource.getStateToken());
057
058        // Factor in the requested mimetype(s)
059        final String mimetype = acceptableMediaTypes.stream()
060                .map(MediaType::toString)
061                .sorted()
062                .collect(Collectors.joining(";"));
063        addComponent(etag, mimetype);
064
065        // Factor in preferences which change which triples are included in the response
066        etag.append('|');
067        if (prefers.displayContainment()) {
068            final var lastUpdated = containmentIndex.containmentLastUpdated(transaction, resource.getFedoraId());
069            if (lastUpdated != null) {
070                etag.append(lastUpdated);
071            }
072        }
073        etag.append('|');
074        if (prefers.displayMembership()) {
075            final var lastUpdated = membershipService.getLastUpdatedTimestamp(transaction, resource.getFedoraId());
076            if (lastUpdated != null) {
077                etag.append(lastUpdated);
078            }
079        }
080        addComponent(etag, prefers.displayEmbed());
081        addComponent(etag, prefers.displayUserRdf());
082        addComponent(etag, prefers.displayReferences());
083        addComponent(etag, prefers.displayServerManaged());
084
085        // Compute a digest of all these components to use as the etag
086        final String etagMd5 = DigestUtils.md5Hex(etag.toString()).toUpperCase();
087        LOGGER.debug("Produced etag {} for {} from {}", etagMd5, resource.getId(), etag);
088
089        return etagMd5;
090    }
091
092    private void addComponent(final StringBuilder etag, final Object component) {
093        etag.append('|').append(component);
094    }
095}
