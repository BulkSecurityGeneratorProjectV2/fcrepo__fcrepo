048/**
049 * A special UpdateVisitor to translate Fedora URIs to internal FedoraIDs.
050 * @author whikloj
051 */
052public class SparqlTranslateVisitor extends UpdateVisitorBase {
053
054    private static final Logger LOGGER = getLogger(SparqlTranslateVisitor.class);
055
056    private List<Update> newUpdates = new ArrayList<>();
057
058    private HttpIdentifierConverter idTranslator;
059
060    private boolean isRelaxedMode;
061
062    public SparqlTranslateVisitor(final HttpIdentifierConverter identifierConverter, final FedoraPropsConfig config) {
063        idTranslator = identifierConverter;
064        isRelaxedMode = config.getServerManagedPropsMode().equals(RELAXED);
065    }
066
067    private List<ConstraintViolationException> exceptions = new ArrayList<>();
068
069    @Override
070    public void visit(final UpdateDataInsert update) {
071        translateUpdate(update);
072    }
073
074    @Override
075    public void visit(final UpdateDataDelete update) {
076        translateUpdate(update);
077    }
078
079    @Override
080    public void visit(final UpdateDeleteWhere update) {
081        translateUpdate(update);
082    }
083
084    @Override
085    public void visit(final UpdateModify update) {
086        translateUpdate(update);
087    }
088
089    /**
090     * Get the new UpdateRequest based on the parsed Updates.
091     * @return the new update request object.
092     */
093    public UpdateRequest getTranslatedRequest() {
094        final UpdateRequest newRequest = UpdateFactory.create();
095        newUpdates.forEach(newRequest::add);
096        return newRequest;
097    }
098
099    /**
100     * Perform a translation of all the triples in an Update adding them to the internal list.
101     * @param update the update request to translate.
102     */
103    private void translateUpdate(final Update update) {
104        final List<Quad> sourceQuads;
105        if (update instanceof UpdateDeleteWhere) {
106            sourceQuads = ((UpdateDeleteWhere)update).getQuads();
107        } else {
108            sourceQuads = ((UpdateData) update).getQuads();
109        }
110        final List<Quad> newQuads = translateQuads(sourceQuads);
111        assertNoExceptions();
112        final Update newUpdate = makeUpdate(update.getClass(), newQuads);
113        newUpdates.add(newUpdate);
114    }
115
116    /**
117     * Perform a translation of all the triples in an UpdateModify request.
118     * @param update the update request to translate
119     */
120    private void translateUpdate(final UpdateModify update) {
121        final UpdateModify newUpdate = new UpdateModify();
122        final List<Quad> insertQuads = (update.hasInsertClause() ? translateQuads(update.getInsertQuads()) :
123                Collections.emptyList());
124        final List<Quad> deleteQuads = (update.hasDeleteClause() ? translateQuads(update.getDeleteQuads()) :
125                Collections.emptyList());
126        assertNoExceptions();
127
128        insertQuads.forEach(q -> newUpdate.getInsertAcc().addQuad(q));
129        deleteQuads.forEach(q -> newUpdate.getDeleteAcc().addQuad(q));
130
131        final Element where = update.getWherePattern();
132        final Element newElement = processElements(where);
133        newUpdate.setElement(newElement);
134        newUpdates.add(newUpdate);
135    }
136
137    private void assertNoExceptions() {
138        if (!exceptions.isEmpty()) {
139            throw new MultipleConstraintViolationException(exceptions);
140        }
141    }
142
143    /**
144     * Process triples inside the Element or return the element.
145     * @param element the element to translate.
146     * @return the translated or original element.
147     */
148    private Element processElements(final Element element) {
149        if (element instanceof ElementGroup) {
150            final ElementGroup group = new ElementGroup();
151            ((ElementGroup) element).getElements().forEach(e -> group.addElement(processElements(e)));
152            return group;
153        } else if (element instanceof ElementPathBlock) {
154            final BasicPattern basicPattern = new BasicPattern();
155            final var tripleIter = ((ElementPathBlock) element).patternElts();
156            tripleIter.forEachRemaining(t -> {
157                if (t.isTriple()) {
158                    try {
159                        checkTripleForDisallowed(t.asTriple());
160                    } catch (final ServerManagedPropertyException | ServerManagedTypeException exc) {
161                        if (!isRelaxedMode) {
162                            exceptions.add(exc);
163                            return;
164                        }
165                    }
166                    basicPattern.add(translateTriple(t.asTriple()));
167                }
168            });
169            return new ElementPathBlock(basicPattern);
170        }
171        return element;
172    }
173
174    /**
175     * Perform the translation to a list of quads.
176     * @param quadsList the quads
177     * @return the translated list of quads.
178     */
179    private List<Quad> translateQuads(final List<Quad> quadsList) {
180        final List<Quad> newQuads = new ArrayList<>();
181        for (final Quad q : quadsList) {
182            try {
183                checkTripleForDisallowed(q.asTriple());
184            } catch (final RelaxableServerManagedPropertyException exc) {
185                if (!isRelaxedMode) {
186                    // Swallow these exceptions to throw together later.
187                    exceptions.add(exc);
188                    continue;
189                }
190            } catch (final ServerManagedTypeException | ServerManagedPropertyException exc) {
191                exceptions.add(exc);
192                continue;
193            }
194            final Node subject = translateId(q.getSubject());
195            final Node object = translateId(q.getObject());
196            final Quad quad = Quad.create(q.getGraph(), subject, q.getPredicate(), object);
197            LOGGER.trace("Translated quad is: {}", quad);
198            newQuads.add(quad);
199        }
200        return newQuads;
201    }
202
203    /**
204     * Translate the subject and object of a triple from external URIs to internal IDs.
205     * @param triple the triple to translate
206     * @return the translated triple.
207     */
208    private Triple translateTriple(final Triple triple) {
209        final Node subject = translateId(triple.getSubject());
210        final Node object = translateId(triple.getObject());
211        return Triple.create(subject, triple.getPredicate(), object);
212    }
213
214    /**
215     * Quads insert/delete data statements don't contain variables and use QuadDataAcc to accumulate,
216     * insert {} delete {} where {} statements can't contain variables and use QuadAcc to accumulate. This function
217     * simplifies the creation of the eventual Update.
218     * @param updateClass the class of Update we are starting with.
219     * @param quadList the list of Quads to generate the above class with.
220     * @return a subclass of Update with the provided Quads.
221     */
222    private Update makeUpdate(final Class<? extends Update> updateClass, final List<Quad> quadList) {
223        try {
224            if (updateClass.equals(UpdateDeleteWhere.class)) {
225                final QuadAcc quadAcc = new QuadAcc();
226                quadList.forEach(quadAcc::addQuad);
227                return new UpdateDeleteWhere(quadAcc);
228            } else {
229                final QuadDataAcc quadsAcc = new QuadDataAcc();
230                quadList.forEach(quadsAcc::addQuad);
231                final Constructor<? extends Update> update = updateClass.getConstructor(QuadDataAcc.class);
232                return update.newInstance(quadsAcc);
233            }
234        } catch (final ReflectiveOperationException exc) {
235            LOGGER.warn("Could not find constructor UpdateRequest");
236            throw new RepositoryRuntimeException("Could not find constructor UpdateRequest", exc);
237        }
238    }
239
240    /**
241     * If the node is a URI with the external domain translate it, otherwise leave it alone
242     * @param externalNode the node to translate
243     * @return the original or translated node.
244     */
245    private Node translateId(final Node externalNode) {
246        if (externalNode.isURI()) {
247            final String externalId = externalNode.getURI();
248            final String newUri = idTranslator.translateUri(externalId);
249            if (idTranslator.inInternalDomain(newUri)) {
250                // If this was converted to an internal ID, make a FedoraId out of it.
251                final var id = FedoraId.create(newUri);
252                if (id.isDescription()) {
253                    // If we are dealing with a binary description ID convert it to the binary ID.
254                    return NodeFactory.createURI(id.getFullDescribedId());
255                }
256            }
257            return NodeFactory.createURI(newUri);
258        }
259        return externalNode;
260    }
