214     * This method returns an HTTP response with content body appropriate to the following arguments.
215     *
216     * @param limit is the number of child resources returned in the response, -1 for all
217     * @param resource the fedora resource
218     * @return HTTP response
219     * @throws IOException in case of error extracting content
220     */
221    protected Response getContent(final int limit, final FedoraResource resource) throws IOException {
222        final RdfStream rdfStream = httpRdfService.bodyToExternalStream(getUri(resource).toString(),
223                getResourceTriples(limit, resource), identifierConverter());
224        final var outputStream = new RdfNamespacedStream(
225                    rdfStream, namespaceRegistry.getNamespaces());
226        setVaryAndPreferenceAppliedHeaders(servletResponse, prefer, resource);
227        return ok(outputStream).build();
228    }

230    protected void setVaryAndPreferenceAppliedHeaders(final HttpServletResponse servletResponse,
231            final MultiPrefer prefer, final FedoraResource resource) {
232        if (prefer != null) {
233            prefer.getReturn().addResponseHeaders(servletResponse);
234        }
235
236        // add vary headers
237        final List<String> varyValues = new ArrayList<>(VARY_HEADERS);
238
239        if (resource.isOriginalResource()) {
240            varyValues.add(ACCEPT_DATETIME);
241        }
242
243        varyValues.forEach(x -> servletResponse.addHeader("Vary", x));
244    }

258    /**
259     * This method returns a stream of RDF triples associated with this target resource
260     *
261     * @param limit is the number of child resources returned in the response, -1 for all
262     * @param resource the fedora resource
263     * @return {@link RdfStream}
264     */
265    private RdfStream getResourceTriples(final int limit, final FedoraResource resource) {
266
267        final LdpPreferTag ldpPreferences = getLdpPreferTag();
268
269        final List<Stream<Triple>> embedStreams = new ArrayList<>();
270
271        embedStreams.add(resourceTripleService.getResourceTriples(
272                transaction(), resource, ldpPreferences, limit));
273
274        // Embed the children of this object
275        if (ldpPreferences.displayEmbed()) {
276            final var containedResources = resourceFactory.getChildren(
277                    transaction(),
278                    resource.getFedoraId());
279            embedStreams.add(containedResources.flatMap(child -> resourceTripleService.getResourceTriples(
280                    transaction(), child, ldpPreferences, limit)));
281        }
282
283        final var rdfStream = new DefaultRdfStream(
284                asNode(resource),
285                embedStreams.stream().reduce(empty(), Stream::concat)
286        );
287
288        if (httpTripleUtil != null && ldpPreferences.displayServerManaged()) {
289            // Adds fixity service triple to all resources and transaction triple to repo root.
290            return httpTripleUtil.addHttpComponentModelsForResourceToStream(rdfStream, resource, uriInfo);
291        }
292
293        return rdfStream;
294    }

297        final PreferTag returnPreference;

310    /**
311     * Get the binary content of a datastream
312     *
313     * @param rangeValue the range value
314     * @param resource the fedora resource
315     * @return Binary blob
316     * @throws IOException if io exception occurred
317     */
318    protected Response getBinaryContent(final String rangeValue, final FedoraResource resource)
319            throws IOException {
320            final Binary binary = (Binary)resource;
321            final CacheControl cc = new CacheControl();
322            cc.setMaxAge(0);
323            cc.setMustRevalidate(true);
324            final Response.ResponseBuilder builder;
325
326            if (rangeValue != null && rangeValue.startsWith("bytes")) {
327
328                final Range range = Range.convert(rangeValue);
329
330                final long contentSize = binary.getContentSize();
331
332                final String endAsString;
333
334                if (range.end() == -1) {
335                    endAsString = Long.toString(contentSize - 1);
336                } else {
337                    endAsString = Long.toString(range.end());
338                }
339
340                final String contentRangeValue =
341                        String.format("bytes %s-%s/%s", range.start(),
342                                endAsString, contentSize);
343
344                if (range.end() > contentSize ||
345                        (range.end() == -1 && range.start() > contentSize)) {
346
347                    builder = status(REQUESTED_RANGE_NOT_SATISFIABLE)
348                            .header("Content-Range", contentRangeValue);
349                } else {
350                    @SuppressWarnings("resource")
351                    final RangeRequestInputStream rangeInputStream =
352                            new RangeRequestInputStream(binary.getContent(), range.start(), range.size());
353
354                    builder = status(PARTIAL_CONTENT).entity(rangeInputStream)
355                            .header("Content-Range", contentRangeValue)
356                            .header(CONTENT_LENGTH, range.size());
357                }
358
359            } else {
360                @SuppressWarnings("resource")
361                final InputStream content = binary.getContent();
362                builder = ok(content);
363            }
364
365
366            // we set the content-type explicitly to avoid content-negotiation from getting in the way
367            // getBinaryResourceMediaType will try to use the mime type on the resource, falling back on
368            // 'application/octet-stream' if the mime type is syntactically invalid
369            return builder.type(getBinaryResourceMediaType(resource).toString())
370                    .cacheControl(cc)
371                    .build();
372
373        }

459    private void addResourceLinkHeaders(final FedoraResource resource, final boolean includeAnchor) {
460        if (resource instanceof NonRdfSourceDescription) {
461            // Link to the original described resource
462            final FedoraResource described = resource.getOriginalResource().getDescribedResource();
463            final URI uri = getUri(described);
464            final Link link = Link.fromUri(uri).rel("describes").build();
465            servletResponse.addHeader(LINK, link.toString());
466        } else if (resource instanceof Binary) {
467            // Link to the original description
468            final FedoraResource description = resource.getOriginalResource().getDescription();
469            final URI uri = getUri(description);
470            final Link.Builder builder = Link.fromUri(uri).rel("describedby");
471
472            if (includeAnchor) {
473                builder.param("anchor", getUri(resource).toString());
474            }
475            servletResponse.addHeader(LINK, builder.build().toString());
476
477            if (resource.isMemento()) {
478                final URI mementoDescription = getUri(resource.getDescription());
479                servletResponse.addHeader(LINK, buildLink(mementoDescription, "describedby"));
480            }
481        }
482
483        final boolean isOriginal = resource.isOriginalResource();
484        // Add versioning headers for versioned originals and mementos
485        if (isOriginal || resource.isMemento() || resource instanceof TimeMap) {
486            final URI originalUri = getUri(resource.getOriginalResource());
487            try {
488                final URI timemapUri = getUri(resource.getTimeMap());
489                servletResponse.addHeader(LINK, buildLink(originalUri, "timegate"));
490                servletResponse.addHeader(LINK, buildLink(originalUri, "original"));
491                servletResponse.addHeader(LINK, buildLink(timemapUri, "timemap"));
492            } catch (final PathNotFoundRuntimeException e) {
493                LOGGER.debug("TimeMap not found for {}, resource not versioned", getUri(resource));
494            }
495        }
496        // Add all system and user types as Link headers.
497        for (final var type : resource.getTypes()) {
498            servletResponse.addHeader(LINK, buildLink(type, "type"));
499        }
500    }

502    /**
503     * Add Link and Option headers
504     *
505     * @param resource the resource to generate headers for
506     */
507    protected void addLinkAndOptionsHttpHeaders(final FedoraResource resource) {
508        // Add Link headers
509        addResourceLinkHeaders(resource);
510        addArchiveGroupLinkHeader(resource);
511        addAcceptExternalHeader();
512
513        // Add Options headers
514        final String options;
515        if (resource.isMemento()) {
516            options = "GET,HEAD,OPTIONS";
517        } else if (resource instanceof TimeMap) {
518            options = "POST,HEAD,GET,OPTIONS";
519            addAcceptPostHeader();
520        } else if (resource instanceof Binary) {
521            options = "DELETE,HEAD,GET,PUT,OPTIONS";
522        } else if (resource instanceof NonRdfSourceDescription) {
523            options = "HEAD,GET,DELETE,PUT,PATCH,OPTIONS";
524            servletResponse.addHeader(HTTP_HEADER_ACCEPT_PATCH, contentTypeSPARQLUpdate);
525        } else if (resource instanceof Container) {
526            options = "DELETE,POST,HEAD,GET,PUT,PATCH,OPTIONS";
527            servletResponse.addHeader(HTTP_HEADER_ACCEPT_PATCH, contentTypeSPARQLUpdate);
528            addAcceptPostHeader();
529        } else {
530            options = "";
531        }
532
533        servletResponse.addHeader("Allow", options);
534    }

536    protected void addTransactionHeaders(final FedoraResource resource) {
537        final var tx = transaction();
538        if (!tx.isShortLived()) {
539            final var externalId = identifierConverter()
540                    .toExternalId(FEDORA_ID_PREFIX + "/" + TX_PREFIX + tx.getId());

543        if (resource.getFedoraId().isRepositoryRoot()) {

545                    .toExternalId(FEDORA_ID_PREFIX + "/" + TX_PREFIX);
546            final Link link = Link.fromUri(txEndpointUri).rel(TX_ENDPOINT_REL).build();
547            servletResponse.addHeader(LINK, link.toString());
548        }

573    /**
574     * Multi-value Link header values parsed by the javax.ws.rs.core are not split out by the framework Therefore we
575     * must do this ourselves.
576     *
577     * @param rawLinks the list of unprocessed links
578     * @return List of strings containing one link value per string.

580    protected List<String> unpackLinks(final List<String> rawLinks) {
581        if (rawLinks == null) {
582            return null;
583        }
584
585        return rawLinks.stream()
586                .flatMap(x -> Arrays.stream(x.split(",")))
587                .collect(Collectors.toList());
588    }

594    /**
595     * Add any resource-specific headers to the response
596     * @param resource the resource
597     * @param dispositionInline whether to return a binary as Content-Disposition inline
598     */
599    protected void addResourceHttpHeaders(final FedoraResource resource, final boolean dispositionInline) {
600        if (resource instanceof Binary) {
601            final Binary binary = (Binary)resource;
602
603            final ContentDisposition.Builder dispositionBuilder = (dispositionInline ? ContentDisposition.inline() :
604                    ContentDisposition.attachment());
605
606            // TODO: Size, created-date and modification-date have been deprecated by Spring and the RFC-6266 Appendix B
607            dispositionBuilder.size(binary.getContentSize());
608
609            if (binary.getCreatedDate() != null) {
610                dispositionBuilder.creationDate(binary.getCreatedDate().atZone(ZoneOffset.UTC));
611            }
612            if (binary.getLastModifiedDate() != null) {
613                dispositionBuilder.modificationDate(binary.getLastModifiedDate().atZone(ZoneOffset.UTC));
614            }
615
616            if (StringUtils.isNotBlank(binary.getFilename())) {
617                final var encoder = StandardCharsets.ISO_8859_1.newEncoder();
618
619                if (encoder.canEncode(binary.getFilename())) {
620                    dispositionBuilder.filename(binary.getFilename());
621                } else {
622                    dispositionBuilder.filename(binary.getFilename(), StandardCharsets.UTF_8);
623                }
624            }
625
626            servletResponse.addHeader(CONTENT_TYPE, binary.getMimeType());
627            // Returning content-length > 0 causes the client to wait for additional data before following the redirect.
628            if (!binary.isRedirect()) {
629                servletResponse.addHeader(CONTENT_LENGTH, String.valueOf(binary.getContentSize()));
630            }
631            servletResponse.addHeader("Accept-Ranges", "bytes");
632            servletResponse.addHeader(CONTENT_DISPOSITION, dispositionBuilder.build().toString());
633        }
634
635        addLinkAndOptionsHttpHeaders(resource);
636        addAclHeader(resource);
637        addMementoHeaders(resource);
638    }

657    /**
658     * Add ETag and Last-Modified cache control headers to the response
659     * <p>
660     * Note: In this implementation, the HTTP headers for ETags and Last-Modified dates are swapped
661     * for fedora:Binary resources and their descriptions. Here, we are drawing a distinction between
662     * the HTTP resource and the LDP resource. As an HTTP resource, the last-modified header should
663     * reflect when the resource at the given URL was last changed. With fedora:Binary resources and
664     * their descriptions, this is a little complicated, for the descriptions have, as their subjects,
665     * the binary itself. And the fedora:lastModified property produced by that NonRdfSourceDescription
666     * refers to the last-modified date of the binary -- not the last-modified date of the
667     * NonRdfSourceDescription.
668     * </p>
669     * @param servletResponse the servlet response
670     * @param resource the fedora resource
671     * @param transaction the transaction
672     */
673    protected void addCacheControlHeaders(final HttpServletResponse servletResponse,
674                                                 final FedoraResource resource,
675                                                 final Transaction transaction) {
676
677        final EntityTag etag;
678        final Instant date;
679
680        // See note about this code in the javadoc above.
681        if (resource instanceof Binary) {
682            // Use a strong ETag for LDP-NR
683            etag = new EntityTag(resource.getEtagValue());
684        } else {
685            // Use a weak ETag for the LDP-RS
686            etag = new EntityTag(etagService.getRdfResourceEtag(transaction, resource, getLdpPreferTag(),
687                    headers.getAcceptableMediaTypes()), true);
688        }
689
690        date = resource.getLastModifiedDate();
691
692        if (!etag.getValue().isEmpty()) {
693            servletResponse.addHeader("ETag", etag.toString());
694        }
695
696        if (!resource.getStateToken().isEmpty()) {
697            //State Tokens, while not used for caching per se,  nevertheless belong
698            //here since we can conveniently reuse the value of the etag for
699            //our state token
700            servletResponse.addHeader("X-State-Token", resource.getStateToken());
701        }
702
703        if (date != null) {
704            servletResponse.addDateHeader("Last-Modified", date.toEpochMilli());
705        }
706    }

709     * Evaluate request preconditions to ensure the resource is the expected state
710     * @param request the request
711     * @param servletResponse the servlet response

713     * @param transaction the transaction

715    protected void evaluateRequestPreconditions(final Request request,
716                                                       final HttpServletResponse servletResponse,
717                                                       final FedoraResource resource,
718                                                       final Transaction transaction) {

724    @VisibleForTesting
725    void evaluateRequestPreconditions(final Request request,
726                                                     final HttpServletResponse servletResponse,
727                                                     final FedoraResource resource,
728                                                     final Transaction transaction,
729                                                     final boolean cacheControl) {
730
731        if (!transaction.isShortLived()) {
732            // Force cache revalidation if in a transaction
733            servletResponse.addHeader(CACHE_CONTROL, "must-revalidate");
734            servletResponse.addHeader(CACHE_CONTROL, "max-age=0");
735        }
736
737        final EntityTag etag;
738        final Instant date;
739        Instant roundedDate = Instant.now();
740
741        // See the related note about the next block of code in the
742        // ContentExposingResource::addCacheControlHeaders method
743        if (resource instanceof Binary) {
744            // Use a strong ETag for the LDP-NR
745            etag = new EntityTag(resource.getEtagValue());
746        } else {
747            // Use a strong ETag for the LDP-RS when validating If-(None)-Match headers
748            etag = new EntityTag(etagService.getRdfResourceEtag(transaction, resource, getLdpPreferTag(),
749                    headers.getAcceptableMediaTypes()), false);
750        }
751
752        date = resource.getLastModifiedDate();
753
754        if (date != null) {
755            roundedDate = date.minusMillis(date.toEpochMilli() % 1000);
756        }
757
758        Response.ResponseBuilder builder = request.evaluatePreconditions(etag);
759        if ( builder == null ) {
760            builder = request.evaluatePreconditions(Date.from(roundedDate));
761        }
762
763        if (builder != null && cacheControl ) {
764            final CacheControl cc = new CacheControl();
765            cc.setMaxAge(0);
766            cc.setMustRevalidate(true);
767            // here we are implicitly emitting a 304
768            // the exception is not an error, it's genuinely
769            // an exceptional condition
770            builder = builder.cacheControl(cc).lastModified(Date.from(roundedDate)).tag(etag);
771        }
772
773        if (builder != null) {
774            final Response response = builder.build();
775            final Object message = response.getEntity();
776            throw new PreconditionException(message != null ? message.toString()
777                    : "Request failed due to unspecified failed precondition.", response.getStatus());
778        }
779
780        final String method = request.getMethod();
781        if (method.equals(HttpPut.METHOD_NAME) || method.equals(HttpPatch.METHOD_NAME)) {
782            final String stateToken = resource.getStateToken();
783            final String clientSuppliedStateToken = headers.getHeaderString("X-If-State-Token");
784            if (clientSuppliedStateToken != null && !stateToken.equals(clientSuppliedStateToken)) {
785                throw new PreconditionException(format(
786                    "The client-supplied value ({0}) does not match the current state token ({1}).",
787                    clientSuppliedStateToken, stateToken), 412);
788            }
789        }
790    }

792    /**
793     * Returns an acceptable plain text media type if possible, or null if not.
794     * @return an acceptable plain-text media type, or null
795     */
796    private MediaType acceptablePlainTextMediaType() {
797        final List<MediaType> acceptable = headers.getAcceptableMediaTypes();
798        if (acceptable == null || acceptable.size() == 0) {
799            return TEXT_PLAIN_TYPE;
800        }
801        for (final MediaType type : acceptable) {
802            if (type.isWildcardType() || (type.isCompatible(TEXT_PLAIN_TYPE) && type.isWildcardSubtype())) {
803                return TEXT_PLAIN_TYPE;
804            } else if (type.isCompatible(TEXT_PLAIN_TYPE)) {
805                return type;
806            }
807        }
808        return null;
809    }

811    /**
812     * Create the appropriate response after a create or update request is processed. When a resource is created,
813     * examine the Prefer and Accept headers to determine whether to include a representation. By default, the URI for
814     * the created resource is return as plain text. If a minimal response is requested, then no body is returned. If a
815     * non-minimal return is requested, return the RDF for the created resource in the appropriate RDF serialization.
816     *
817     * @param resource The created or updated Fedora resource.
818     * @param created True for a newly-created resource, false for an updated resource.
819     * @return 204 No Content (for updated resources), 201 Created (for created resources) including the resource URI or
820     *         content depending on Prefer headers.
821     */
822    @SuppressWarnings("resource")
823    protected Response createUpdateResponse(final FedoraResource resource, final boolean created) {
824        addCacheControlHeaders(servletResponse, resource, transaction());
825        addResourceLinkHeaders(resource, created);
826        addExternalContentHeaders(resource);
827        addAclHeader(resource);
828        addMementoHeaders(resource);
829        addTransactionHeaders(resource);
830
831        if (!created) {
832            return noContent().build();
833        }
834
835        final URI location = getUri(resource);
836        final Response.ResponseBuilder builder = created(location);
837
838        if (prefer == null || !prefer.hasReturn()) {
839            final MediaType acceptablePlainText = acceptablePlainTextMediaType();
840            if (acceptablePlainText != null) {
841                return builder.type(acceptablePlainText).entity(location.toString()).build();
842            }
843            return notAcceptable(mediaTypes(TEXT_PLAIN_TYPE).build()).build();
844        } else if (prefer.getReturn().getValue().equals("minimal")) {
845            return builder.build();
846        } else {
847            prefer.getReturn().addResponseHeaders(servletResponse);
848            final RdfNamespacedStream rdfStream = new RdfNamespacedStream(
849                new DefaultRdfStream(asNode(resource), getResourceTriples(resource)),
850                namespaceRegistry.getNamespaces());
851            return builder.entity(rdfStream).build();
852        }
853    }

910    /**
911     * Calculate the max number of children to display at once.
912     *
913     * @return the limit of children to display.
914     */
915    protected int getChildrenLimit() {
916        final List<String> acceptHeaders = headers.getRequestHeader(ACCEPT);
917        if (acceptHeaders != null && acceptHeaders.size() > 0) {
918            final List<String> accept = Arrays.asList(acceptHeaders.get(0).split(","));
919            if (accept.contains(TEXT_HTML)) {
920                // Magic number '100' is tied to common-metadata.vsl display of ellipses
921                return 100;
922            }
923        }
924
925        final List<String> limits = headers.getRequestHeader("Limit");
926        if (null != limits && limits.size() > 0) {
927            try {
928                return Integer.parseInt(limits.get(0));
929
930            } catch (final NumberFormatException e) {
931                LOGGER.warn("Invalid 'Limit' header value: {}", limits.get(0));
932                throw new ClientErrorException("Invalid 'Limit' header value: " + limits.get(0), SC_BAD_REQUEST, e);
933            }
934        }
935        return -1;
936    }

952    /**
953     * Parse the RFC-3230 Digest response header value. Look for a sha1 checksum and return it as a urn, if missing or
954     * malformed an empty string is returned.
955     *
956     * @param digest The Digest header value
957     * @return the sha1 checksum value
958     * @throws UnsupportedAlgorithmException if an unsupported digest is used
959     */
960    protected static Collection<URI> parseDigestHeader(final String digest) throws UnsupportedAlgorithmException {
961        try {
962            final var digestPairs = RFC3230_SPLITTER.split(nullToEmpty(digest));
963            final var unsupportedAlgs = digestPairs.keySet().stream()
964                    .filter(Predicate.not(DigestAlgorithm::isSupportedAlgorithm))
965                    .collect(Collectors.toSet());
966
967            // If you have one or more digests that are all valid or no digests.
968            if (digestPairs.isEmpty() || unsupportedAlgs.isEmpty()) {
969                return digestPairs.entrySet().stream()
970                    .map(entry -> ContentDigest.asURI(entry.getKey(), entry.getValue()))
971                    .collect(toSet());
972            } else {
973                throw new UnsupportedAlgorithmException(String.format("Unsupported Digest Algorithm%1$s: %2$s",
974                        unsupportedAlgs.size() > 1 ? 's' : "", String.join(",", unsupportedAlgs)));
975            }
976        } catch (final IllegalArgumentException e) {
977            throw new ClientErrorException("Invalid Digest header: " + digest + "\n", BAD_REQUEST);
978        }
979    }

981    /**
982     * @param rootThrowable The original throwable
983     * @param throwable The throwable under direct scrutiny.
984     * @throws InvalidChecksumException in case there was a checksum mismatch
985     */
986    protected void checkForInsufficientStorageException(final Throwable rootThrowable, final Throwable throwable)
987        throws InvalidChecksumException {
988        final String message = throwable.getMessage();
989        if (throwable instanceof IOException && message != null && message.contains(
990            INSUFFICIENT_SPACE_IDENTIFYING_MESSAGE)) {
991            throw new InsufficientStorageException(throwable.getMessage(), rootThrowable);
992        }
993
994        if (throwable.getCause() != null) {
995            checkForInsufficientStorageException(rootThrowable, throwable.getCause());
996        }
997
998        if (rootThrowable instanceof InvalidChecksumException) {
999            throw (InvalidChecksumException) rootThrowable;
1000        } else if (rootThrowable instanceof RuntimeException) {
1001            throw (RuntimeException) rootThrowable;
1002        } else {
1003            throw new RepositoryRuntimeException(rootThrowable.getMessage(), rootThrowable);
1004        }
1005    }
