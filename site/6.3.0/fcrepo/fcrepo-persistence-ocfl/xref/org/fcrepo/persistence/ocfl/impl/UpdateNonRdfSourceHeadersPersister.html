6   package org.fcrepo.persistence.ocfl.impl;
7   
8   import static org.fcrepo.kernel.api.operations.ResourceOperationType.UPDATE_HEADERS;
9   
10  import org.fcrepo.kernel.api.operations.ResourceOperation;
11  import org.fcrepo.kernel.api.operations.UpdateNonRdfSourceHeadersOperation;
12  import org.fcrepo.persistence.api.exceptions.PersistentStorageException;
13  import org.fcrepo.persistence.ocfl.api.FedoraToOcflObjectIndex;
14  import org.fcrepo.storage.ocfl.OcflObjectSession;
15  import org.slf4j.Logger;
16  import org.slf4j.LoggerFactory;
17  
18  /**
19   * This class implements the persistence of the headers for an RDFSource to bring in new system managed properties
20   *
21   * @author mikejritter
22   * @since 6.0.0
23   */
24  public class UpdateNonRdfSourceHeadersPersister extends AbstractRdfSourcePersister {
25  
26      private static final Logger log = LoggerFactory.getLogger(UpdateRdfSourcePersister.class);
27  
28      /**
29       * Constructor
30       *
31       * @param fedoraOcflIndex the FedoraToOcflObjectIndex
32       */
33      public UpdateNonRdfSourceHeadersPersister(final FedoraToOcflObjectIndex fedoraOcflIndex) {
34          super(UpdateNonRdfSourceHeadersOperation.class, UPDATE_HEADERS, fedoraOcflIndex);
35      }
36  
37      @Override
38      public void persist(final OcflPersistentStorageSession session, final ResourceOperation operation)
39          throws PersistentStorageException {
40          final var resourceId = operation.getResourceId();
41          log.debug("persisting {} headers to {}", resourceId, session);
42  
43          final var fedoraOcflMapping = getMapping(operation.getTransaction(), resourceId);
44          final var ocflId = fedoraOcflMapping.getOcflObjectId();
45          final OcflObjectSession objSession = session.findOrCreateSession(ocflId);
46  
47          // unlike with normal updates we don't want to clear the digests/content-size, just the server managed headers
48          // in the event the server managed mode is strict, all values will be null
49          final var headers = new ResourceHeadersAdapter(objSession.readHeaders(resourceId.getResourceId()));
50  
51          final var updateHeadersOp = (UpdateNonRdfSourceHeadersOperation) operation;
52          final var createdDate = updateHeadersOp.getCreatedDate();
53          final var lastModifiedDate = updateHeadersOp.getLastModifiedDate();
54          final var createdBy = updateHeadersOp.getCreatedBy();
55          final var lastModifiedBy = updateHeadersOp.getLastModifiedBy();
56          final var mimetype = updateHeadersOp.getMimeType();
57          final var filename = updateHeadersOp.getFilename();
58          if (createdDate != null) {
59              headers.setCreatedDate(createdDate);
60          }
61          if (lastModifiedDate != null) {
62              headers.setLastModifiedDate(lastModifiedDate);
63          }
64          if (createdBy != null) {
65              headers.setCreatedBy(createdBy);
66          }
67          if (lastModifiedBy != null) {
68              headers.setLastModifiedBy(lastModifiedBy);
69          }
70          if (mimetype != null) {
71              headers.setMimeType(mimetype);
72          }
73          if (filename != null) {
74              headers.setFilename(filename);
75          }
76  
77          objSession.writeHeaders(headers.asStorageHeaders());
78      }
79  
80  }
