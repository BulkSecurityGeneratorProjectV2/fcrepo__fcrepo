6   package org.fcrepo.kernel.api.services;
7   
8   import org.fcrepo.kernel.api.Transaction;
9   import org.fcrepo.kernel.api.identifiers.FedoraId;
10  
11  import java.time.format.DateTimeFormatter;
12  
13  import static java.time.ZoneOffset.UTC;
14  import static java.time.format.DateTimeFormatter.RFC_1123_DATE_TIME;
15  
16  /**
17   * Service for creating versions of resources
18   *
19   * @author bbpennel
20   * @author whikloj
21   * @since Feb 19, 2014
22   */
23  public interface VersionService {
24  
25      /**
26       * To format a datetime for use as a Memento path.
27       */
28      DateTimeFormatter MEMENTO_LABEL_FORMATTER = DateTimeFormatter.ofPattern("yyyyMMddHHmmss")
29              .withZone(UTC);
30  
31      /**
32       * To format a datetime as RFC-1123 with correct timezone.
33       */
34      DateTimeFormatter MEMENTO_RFC_1123_FORMATTER = RFC_1123_DATE_TIME.withZone(UTC);
35  
36      /**
37       * Explicitly creates a version for the resource at the path provided.
38       *
39       * @param transaction the transaction in which the resource resides
40       * @param fedoraId the internal resource id
41       * @param userPrincipal the user principal
42       */
43      void createVersion(Transaction transaction, FedoraId fedoraId, String userPrincipal);
44  
45  }
