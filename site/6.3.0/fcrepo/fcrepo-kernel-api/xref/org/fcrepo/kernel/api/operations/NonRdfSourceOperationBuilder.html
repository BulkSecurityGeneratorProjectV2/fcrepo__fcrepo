6   package org.fcrepo.kernel.api.operations;
7   
8   import java.net.URI;
9   import java.util.Collection;
10  
11  /**
12   * Builder for an operation for interacting with a non-rdf source
13   *
14   * @author bbpennel
15   */
16  public interface NonRdfSourceOperationBuilder extends ResourceOperationBuilder {
17  
18      /**
19       * Set the mimetype for content in this resource
20       *
21       * @param mimetype the mime-type.
22       * @return the builder.
23       */
24      NonRdfSourceOperationBuilder mimeType(String mimetype);
25  
26      /**
27       * Set the filename
28       *
29       * @param filename name of the file.
30       * @return the builder.
31       */
32      NonRdfSourceOperationBuilder filename(String filename);
33  
34      /**
35       * Collection of digests for content in this resource
36       *
37       * @param digests collection of digests
38       * @return the builder.
39       */
40      NonRdfSourceOperationBuilder contentDigests(Collection<URI> digests);
41  
42      /**
43       * Set the number of bytes for the content
44       *
45       * @param size size of the content in bytes
46       * @return the builder
47       */
48      NonRdfSourceOperationBuilder contentSize(long size);
49  
50      @Override
51      NonRdfSourceOperationBuilder userPrincipal(String userPrincipal);
52  
53      @Override
54      NonRdfSourceOperation build();
55  }
