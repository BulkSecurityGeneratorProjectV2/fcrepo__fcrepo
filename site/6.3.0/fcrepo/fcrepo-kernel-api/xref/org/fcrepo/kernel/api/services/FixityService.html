6   package org.fcrepo.kernel.api.services;
7   
8   import org.fcrepo.kernel.api.RdfStream;
9   import org.fcrepo.kernel.api.exception.UnsupportedAlgorithmException;
10  import org.fcrepo.kernel.api.models.Binary;
11  
12  import java.net.URI;
13  import java.util.Collection;
14  
15  /**
16   * Service which calculates and compares digests for binary objects
17   *
18   * @author peichman
19   */
20  public interface FixityService {
21    /**
22     * Calculate the requested set of digests for the provided binary
23     * @param binary the binary resource to
24     * @param algorithms set of digest algorithms to calculate
25     * @return list of calculated digests
26     * @throws UnsupportedAlgorithmException if unsupported digest algorithms were provided
27     */
28    Collection<URI> getFixity(Binary binary, Collection<String> algorithms) throws UnsupportedAlgorithmException;
29  
30    /**
31     * Digest this binary with the digest algorithms provided
32     * @param binary the binary resource to digest
33     * @return The result of the fixity check.
34     */
35    RdfStream checkFixity(Binary binary);
36  }
