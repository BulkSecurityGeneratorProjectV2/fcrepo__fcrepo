8   import java.io.InputStream;
9   import java.net.URI;
10  import java.util.Collection;
11  
12  import org.fcrepo.kernel.api.Transaction;
13  import org.fcrepo.kernel.api.identifiers.FedoraId;
14  import org.fcrepo.kernel.api.models.ExternalContent;

16  /**
17   * Interface for service to replace existing binaries
18   *
19   * @author mohideen
20   */
21  public interface ReplaceBinariesService {
22  
23      /**
24       * Replace an existing binary.
25       *
26       * @param tx The transaction for the request.
27       * @param userPrincipal the user performing the service
28       * @param fedoraId The internal identifier of the parent.
29       * @param filename The filename of the binary.
30       * @param contentType The content-type header or null if none.
31       * @param digests The binary digest or null if none.
32       * @param size The binary size.
33       * @param contentBody The request body or null if none.
34       * @param externalContent The external content handler or null if none.
35       */
36      void perform(Transaction tx,
37                   String userPrincipal,
38                   FedoraId fedoraId,
39                   String filename,
40                   String contentType,
41                   Collection<URI> digests,
42                   InputStream contentBody,
43                   long size,
44                   ExternalContent externalContent);
45  }
