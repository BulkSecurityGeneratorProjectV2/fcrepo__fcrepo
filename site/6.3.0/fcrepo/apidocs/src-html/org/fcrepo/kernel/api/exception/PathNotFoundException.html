006package org.fcrepo.kernel.api.exception;
007
008
009/**
010 * Indicates the path was not found in the repository.
011 *
012 * @author dbernstein
013 * @author whikloj
014 */
015public class PathNotFoundException extends Exception {
016
017    private static final long serialVersionUID = 1L;
018
019    /**
020     * Ordinary constructor
021     *
022     * @param msg the message
023     */
024    public PathNotFoundException(final String msg) {
025        super(msg);
026    }
027
028    /**
029     * Constructor for wrapping exception.
030     *
031     * @param message the original message.
032     * @param cause the root cause.
033     */
034    public PathNotFoundException(final String message, final Throwable cause) {
035        super(message, cause);
036    }
037}
