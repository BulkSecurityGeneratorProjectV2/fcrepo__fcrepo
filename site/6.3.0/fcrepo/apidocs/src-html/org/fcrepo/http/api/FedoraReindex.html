006package org.fcrepo.http.api;
007
008import static java.lang.String.format;
009import static javax.ws.rs.core.MediaType.APPLICATION_JSON;
010import static javax.ws.rs.core.Response.status;
011import static org.fcrepo.http.commons.domain.RDFMediaType.TEXT_PLAIN_WITH_CHARSET;
012import static org.slf4j.LoggerFactory.getLogger;
013
014import javax.inject.Inject;
015import javax.ws.rs.BadRequestException;
016import javax.ws.rs.DELETE;
017import javax.ws.rs.GET;
018import javax.ws.rs.POST;
019import javax.ws.rs.PUT;
020import javax.ws.rs.Path;
021import javax.ws.rs.PathParam;
022import javax.ws.rs.Produces;
023import javax.ws.rs.core.Response;
024
025import org.fcrepo.kernel.api.identifiers.FedoraId;
026import org.fcrepo.kernel.api.services.ReindexService;
027
028import org.apache.http.HttpStatus;
029import org.slf4j.Logger;
030import org.springframework.context.annotation.Scope;
031
032import io.micrometer.core.annotation.Timed;
033
034/**
035 * @author dbernstein
036 * @since 12/01/20
037 */
038@Timed
039@Scope("request")
040@Path("/{path: (.+/)?}fcr:reindex")
041public class FedoraReindex extends FedoraBaseResource {
042
043    private static final Logger LOGGER = getLogger(FedoraReindex.class);
044
045    @Inject
046    private ReindexService reindexService;
047
048    @PathParam("path")
049    protected String externalPath;
050
051    /**
052     * Default JAX-RS entry point
053     */
054    public FedoraReindex() {
055        super();
056    }
057
058    /**
059     * Reindex a fedora resource.
060     *
061     * @return A response
062     */
063    @POST
064    @Produces({APPLICATION_JSON + ";qs=1.0",
065            TEXT_PLAIN_WITH_CHARSET})
066    public Response reindexObject() {
067        LOGGER.info("receiving reindex request for fedora_id = {}", externalPath);
068        try {
069            final var transaction = transaction();
070            final var id = FedoraId.create(externalPath);
071            final var baseId = id.asBaseId();
072
073            doInDbTxWithRetry(() -> {
074                this.reindexService.reindexByFedoraId(transaction(), getUserPrincipal(), baseId);
075                transaction.commitIfShortLived();
076            });
077            final var message = format("successfully reindexed %s", id.getBaseId());
078            LOGGER.info(message);
079            return status(HttpStatus.SC_NO_CONTENT).entity(message).build();
080        } catch (final Exception ex) {
081            throw new BadRequestException(ex.getMessage(), ex);
082        } finally {
083            transaction().releaseResourceLocksIfShortLived();
084        }
085    }
086
087    @DELETE
088    public Response delete() {
089        return methodNotAllowed();
090    }
091
092    @PUT
093    public Response put() {
094        return methodNotAllowed();
095    }
096
097    @GET
098    public Response get() {
099        return methodNotAllowed();
100    }
101
102    private Response methodNotAllowed() {
103        return status(HttpStatus.SC_METHOD_NOT_ALLOWED).build();
104    }
105
106
107}
