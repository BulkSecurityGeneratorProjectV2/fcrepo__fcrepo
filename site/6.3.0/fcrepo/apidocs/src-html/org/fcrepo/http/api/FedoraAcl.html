111    /**
112     * PUT to create FedoraWebacACL resource.
113     * @param requestContentType The content type of the resource body
114     * @param requestBodyStream  The request body as stream
115     * @return the response for a request to create a Fedora WebAc acl
116     */
117    @PUT
118    public Response createFedoraWebacAcl(@HeaderParam(CONTENT_TYPE) final MediaType requestContentType,
119                                         final InputStream requestBodyStream) {
120
121        if (resource().isAcl() || resource().isMemento()) {
122            throw new BadRequestException("ACL resource creation is not allowed for resource " + resource().getId());
123        }
124        LOGGER.info("PUT acl resource '{}'", externalPath());
125
126        final FedoraId aclId = identifierConverter().pathToInternalId(externalPath()).asAcl();
127        final boolean exists = doesResourceExist(transaction(), aclId, false);
128
129        try {
130            final MediaType contentType =
131                    requestContentType == null ?
132                            RDFMediaType.TURTLE_TYPE : valueOf(getSimpleContentType(requestContentType));
133
134            if (isRdfContentType(contentType.toString())) {
135                final Model model = httpRdfService.bodyToInternalModel(aclId,
136                        requestBodyStream, contentType, identifierConverter(), hasLenientPreferHeader());
137                doInDbTxWithRetry(() -> {
138                    if (exists) {
139                        replacePropertiesService.perform(transaction(), getUserPrincipal(), aclId, model);
140                    } else {
141                        webacAclService.create(transaction(), aclId, getUserPrincipal(), model);
142                    }
143                    transaction().commitIfShortLived();
144                });
145            } else {
146                throw new BadRequestException("Content-Type (" + requestContentType + ") is invalid." +
147                        " Try text/turtle or other RDF compatible type.");
148            }
149
150            try {
151                final var aclResource = getFedoraResource(transaction(), aclId);
152                addCacheControlHeaders(servletResponse, aclResource, transaction());
153                final URI location = getUri(aclResource);
154                if (!exists) {
155                    return created(location).build();
156                } else {
157                    return noContent().location(location).build();
158                }
159            } catch (final PathNotFoundException e) {
160                throw new PathNotFoundRuntimeException(e.getMessage(), e);
161            }
162        } finally {
163            transaction().releaseResourceLocksIfShortLived();
164        }
165    }

167    /**
168     * PATCH to update an FedoraWebacACL resource using SPARQL-UPDATE
169     *
170     * @param requestBodyStream the request body stream
171     * @return 204
172     * @throws IOException if IO exception occurred
173     */
174    @PATCH
175    @Consumes({ contentTypeSPARQLUpdate })
176    public Response updateSparql(final InputStream requestBodyStream)
177        throws IOException, ItemNotFoundException {
178        hasRestrictedPath(externalPath);
179
180        if (null == requestBodyStream) {
181            throw new BadRequestException("SPARQL-UPDATE requests must have content!");
182        }
183
184        final FedoraId originalId = identifierConverter().pathToInternalId(externalPath());
185        final FedoraId aclId = originalId.asAcl();
186        final FedoraResource aclResource;
187        try {
188             aclResource = getFedoraResource(transaction(), aclId);
189        } catch (final PathNotFoundException exc) {
190            if (originalId.isRepositoryRoot()) {
191                throw new ClientErrorException("The default root ACL is system generated and cannot be modified. " +
192                        "To override the default root ACL you must PUT a user-defined ACL to this endpoint.",
193                        CONFLICT);
194            }
195            throw new ItemNotFoundException("not found");
196        }
197
198        try {
199            final String requestBody = IOUtils.toString(requestBodyStream, UTF_8);
200            if (isBlank(requestBody)) {
201                throw new BadRequestException("SPARQL-UPDATE requests must have content!");
202            }
203
204            evaluateRequestPreconditions(request, servletResponse, aclResource, transaction());
205
206            LOGGER.info("PATCH for '{}'", externalPath);
207            final String newRequest = httpRdfService.patchRequestToInternalString(aclResource.getFedoraId(),
208                    requestBody, identifierConverter());
209            LOGGER.debug("PATCH request translated to '{}'", newRequest);
210
211            doInDbTxWithRetry(() -> {
212                patchResourcewithSparql(aclResource, newRequest);
213                transaction().commitIfShortLived();
214            });
215
216            addCacheControlHeaders(servletResponse, aclResource, transaction());
217
218            return noContent().build();
219        } catch (final IllegalArgumentException iae) {
220            throw new BadRequestException(iae.getMessage());
221        } catch (final AccessDeniedException e) {
222            throw e;
223        } catch (final RuntimeException ex) {
224            final Throwable cause = ex.getCause();
225            if (cause instanceof PathNotFoundRuntimeException) {
226                // the sparql update referred to a repository resource that doesn't exist
227                throw new BadRequestException(cause.getMessage());
228            }
229            throw ex;
230        } finally {
231            transaction().releaseResourceLocksIfShortLived();
232        }
233    }

240    /**
241     * GET to retrieve the ACL resource.
242     *
243     * @return a binary or the triples for the specified node
244     * @throws IOException if IO exception occurred
245     */
246    @GET
247    @Produces({ TURTLE_WITH_CHARSET + ";qs=1.0", JSON_LD + ";qs=0.8",
248                N3_WITH_CHARSET, N3_ALT2_WITH_CHARSET, RDF_XML, NTRIPLES, TEXT_PLAIN_WITH_CHARSET,
249                TEXT_HTML_WITH_CHARSET })
250    public Response getResource()
251            throws IOException, ItemNotFoundException {
252
253        LOGGER.info("GET resource '{}'", externalPath());
254
255        final FedoraId originalId = identifierConverter().pathToInternalId(externalPath());
256        final FedoraId aclId = originalId.asAcl();
257        final boolean exists = doesResourceExist(transaction(), aclId, false);
258
259        if (!exists) {
260            if (originalId.isRepositoryRoot()) {
261                final String aclUri = identifierConverter().toExternalId(aclId.getFullId());
262
263                final RdfStream defaultRdfStream = DefaultRdfStream.fromModel(createResource(aclUri).asNode(),
264                    getDefaultAcl(aclUri, authPropsConfig.getRootAuthAclPath()));
265                final RdfStream rdfStream = httpRdfService.bodyToExternalStream(aclUri,
266                        defaultRdfStream, identifierConverter());
267                final var output = new RdfNamespacedStream(
268                        rdfStream, namespaceRegistry.getNamespaces());
269                return ok(output).build();
270            }
271
272            throw new ItemNotFoundException(String.format("No ACL found at %s", externalPath));
273        }
274
275        final WebacAcl aclResource = webacAclService.find(transaction(), aclId);
276        checkCacheControlHeaders(request, servletResponse, aclResource, transaction());
277
278        LOGGER.info("GET resource '{}'", externalPath);
279        addResourceHttpHeaders(aclResource);
280        return getContent(getChildrenLimit(), aclResource);
281
282    }

291
292        hasRestrictedPath(externalPath);

294
295        final FedoraId originalId = identifierConverter().pathToInternalId(externalPath());
296        final FedoraId aclId = originalId.asAcl();
297        try {
298            final var aclResource = getFedoraResource(transaction(), aclId);
299            doInDbTxWithRetry(() -> {
300                deleteResourceService.perform(transaction(), aclResource, getUserPrincipal());
301                transaction().commitIfShortLived();
302            });
303        } catch (final PathNotFoundException exc) {
304            if (originalId.isRepositoryRoot()) {
305                throw new ClientErrorException("The default root ACL is system generated and cannot be deleted. " +
306                        "To override the default root ACL you must PUT a user-defined ACL to this endpoint.",
307                        CONFLICT);
308            }
309            throw new PathNotFoundRuntimeException(exc.getMessage(), exc);
310        } finally {
311            transaction().releaseResourceLocksIfShortLived();
312        }
313        return noContent().build();
314    }

316    /**
317     * Retrieve the default root ACL from a user specified location if it exists,
318     * otherwise the one provided by Fedora will be used.
319     * @param baseUri the URI of the default ACL
320     * @param customRootAcl the path to a custom root acl to use, optional
321     * @return Model the rdf model of the default root ACL
322     */
323    public static Model getDefaultAcl(final String baseUri, final java.nio.file.Path customRootAcl) {
324        final Model model = createDefaultModel();
325
326        if (customRootAcl != null && Files.isRegularFile(customRootAcl)) {
327            try {
328                LOGGER.debug("Getting root authorization from file: {}", customRootAcl);
329
330                try (final var stream = new BufferedInputStream(Files.newInputStream(customRootAcl))) {
331                    RDFDataMgr.read(model, stream, baseUri, RDFLanguages.pathnameToLang(customRootAcl.toString()));
332                }
333
334                return model;
335            } catch (final JenaException | IOException ex) {
336                throw new RuntimeException("Error parsing the default root ACL " + customRootAcl + ".", ex);
337            }
338        }
339
340        try (final InputStream is = FedoraAcl.class.getResourceAsStream(ROOT_AUTHORIZATION_LOCATION)) {
341            LOGGER.debug("Getting root ACL from classpath: {}", ROOT_AUTHORIZATION_LOCATION);
342
343            return model.read(is, baseUri, TTL.getName());
344        } catch (final IOException ex) {
345            throw new RuntimeException("Error reading the default root Acl " + ROOT_AUTHORIZATION_LOCATION + ".", ex);
346        } catch (final JenaException ex) {
347            throw new RuntimeException("Error parsing the default root ACL " + ROOT_AUTHORIZATION_LOCATION + ".", ex);
348        }
349    }
