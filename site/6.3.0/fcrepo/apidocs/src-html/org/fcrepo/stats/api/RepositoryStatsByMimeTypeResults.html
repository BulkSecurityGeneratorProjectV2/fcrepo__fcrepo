006package org.fcrepo.stats.api;
007
008import static com.fasterxml.jackson.annotation.JsonProperty.Access.READ_ONLY;
009import static org.apache.commons.lang3.ObjectUtils.isEmpty;
010
011import java.util.ArrayList;
012import java.util.List;
013
014import com.fasterxml.jackson.annotation.JsonProperty;
015import org.apache.commons.lang3.builder.ToStringBuilder;
016
017/**
018 * A repository stats object for binary results by mime type.
019 * @author dbernstein
020 */
021public class RepositoryStatsByMimeTypeResults {
022
023    @JsonProperty("mime_types")
024    private List<MimeTypeStatsResult> mimeTypes = new ArrayList<>();
025
026    public List<MimeTypeStatsResult> getMimeTypes() {
027        return mimeTypes;
028    }
029
030    public void setMimeTypes(final List<MimeTypeStatsResult> mimeTypes) {
031        this.mimeTypes = mimeTypes;
032    }
033
034    @JsonProperty(value = "resource_count", access = READ_ONLY)
035    public Long getResourceCount() {
036        return isEmpty(this.mimeTypes) ? 0 :
037                this.mimeTypes.stream().mapToLong(x -> x.getResourceCount()).sum();
038    }
039
040    @JsonProperty(value = "byte_count", access = READ_ONLY)
041    public Long getByteCount() {
042        return isEmpty(this.mimeTypes) ? 0 :
043                this.mimeTypes.stream().mapToLong(x -> x.getByteCount()).sum();
044    }
045
046    @Override
047    public String toString() {
048        return ToStringBuilder.reflectionToString(this).toString();
049    }
050}
