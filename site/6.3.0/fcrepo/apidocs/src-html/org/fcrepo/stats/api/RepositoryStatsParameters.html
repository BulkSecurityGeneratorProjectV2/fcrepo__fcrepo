006package org.fcrepo.stats.api;
007
008import java.util.List;
009
010/**
011 * A parameter object that allows you to narrow your statistics query
012 * @author dbernstein
013 */
014public class RepositoryStatsParameters {
015
016    private List<String> mimeTypes;
017    private List<String> rdfTypes;
018
019    public List<String> getMimeTypes() {
020        return mimeTypes;
021    }
022
023    public void setMimeTypes(final List<String> mimeTypes) {
024        this.mimeTypes = mimeTypes;
025    }
026
027    public RepositoryStatsParameters() {
028    }
029
030    public void setRdfTypes(final List<String> rdfTypes) {
031        this.rdfTypes = rdfTypes;
032    }
033
034    public List<String> getRdfTypes() {
035        return rdfTypes;
036    }
037}
