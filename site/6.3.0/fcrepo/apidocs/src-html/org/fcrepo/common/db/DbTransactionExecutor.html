007package org.fcrepo.common.db;
008
009import static org.slf4j.LoggerFactory.getLogger;
010
011import java.time.temporal.ChronoUnit;
012
013import org.slf4j.Logger;
014import org.springframework.beans.factory.annotation.Autowired;
015import org.springframework.dao.DeadlockLoserDataAccessException;
016import org.springframework.stereotype.Component;
017import org.springframework.transaction.support.TransactionTemplate;
018
019import net.jodah.failsafe.Failsafe;
020import net.jodah.failsafe.RetryPolicy;
021
022/**
023 * Wrapper around Spring's db transaction management
024 *
025 * @author pwinckles
026 */
027@Component
028public class DbTransactionExecutor {
029
030    private static final Logger LOGGER = getLogger(DbTransactionExecutor.class);
031
032    private static final RetryPolicy<Object> DB_RETRY = new RetryPolicy<>()
033            .handleIf(e -> {
034                return e instanceof DeadlockLoserDataAccessException
035                        || (e.getCause() != null && e.getCause() instanceof DeadlockLoserDataAccessException);
036            })
037            .onRetry(event -> {
038                LOGGER.debug("Retrying operation that failed with the following exception", event.getLastFailure());
039            })
040            .withBackoff(10, 100, ChronoUnit.MILLIS, 1.5)
041            .withJitter(0.1)
042            .withMaxRetries(10);
043
044    @Autowired
045    private TransactionTemplate transactionTemplate;
046
047    public DbTransactionExecutor() {
048
049    }
050
051    public DbTransactionExecutor(final TransactionTemplate transactionTemplate) {
052        this.transactionTemplate = transactionTemplate;
053    }
054
055    /**
056     * Executes the runnable within a DB transaction that will retry entire block on MySQL deadlock exceptions.
057     *
058     * @param action the code to execute
059     */
060    public void doInTxWithRetry(final Runnable action) {
061        Failsafe.with(DB_RETRY).run(() -> {
062            doInTx(action);
063        });
064    }
065
066    /**
067     * Executes the runnable within a DB transaction. MySQL deadlock exceptions are NOT retried.
068     *
069     * @param action the code to execute
070     */
071    public void doInTx(final Runnable action) {
072        if (transactionTemplate == null) {
073            // If the transaction template is not set, just execute the code without a tx.
074            // This will never happen when configured by Spring, but is useful when unit testing
075            LOGGER.warn("Executing outside of a DB transaction");
076            action.run();
077        } else {
078            transactionTemplate.executeWithoutResult(status -> {
079                action.run();
080            });
081        }
082    }
083
084}
