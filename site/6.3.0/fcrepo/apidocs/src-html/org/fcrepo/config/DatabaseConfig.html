007package org.fcrepo.config;
008
009import java.time.Instant;
010import java.time.LocalDateTime;
011import java.time.ZoneOffset;
012import java.util.Map;
013
014import javax.annotation.PostConstruct;
015import javax.sql.DataSource;
016
017import org.flywaydb.core.Flyway;
018import org.slf4j.Logger;
019import org.slf4j.LoggerFactory;
020import org.springframework.beans.factory.annotation.Value;
021import org.springframework.context.annotation.Bean;
022import org.springframework.context.annotation.Configuration;
023import org.springframework.core.convert.converter.Converter;
024import org.springframework.core.convert.converter.ConverterRegistry;
025import org.springframework.core.convert.support.DefaultConversionService;
026import org.springframework.jdbc.datasource.DataSourceTransactionManager;
027import org.springframework.transaction.PlatformTransactionManager;
028import org.springframework.transaction.TransactionDefinition;
029import org.springframework.transaction.annotation.EnableTransactionManagement;
030import org.springframework.transaction.support.DefaultTransactionDefinition;
031import org.springframework.transaction.support.TransactionTemplate;
032
033import com.zaxxer.hikari.HikariDataSource;
034
035/**
036 * @author pwinckles
037 */
038@EnableTransactionManagement
039@Configuration
040public class DatabaseConfig extends BasePropsConfig {
041
042    private static final Logger LOGGER = LoggerFactory.getLogger(DatabaseConfig.class);
043
044    private static final String H2_FILE = "fcrepo-h2";
045
046    @Value("${fcrepo.db.url:#{'jdbc:h2:'" +
047            " + fedoraPropsConfig.fedoraData.resolve('" + H2_FILE + "').toAbsolutePath().toString()" +
048            " + ';FILE_LOCK=SOCKET'}}")
049    private String dbUrl;
050
051    @Value("${fcrepo.db.user:}")
052    private String dbUser;
053
054    @Value("${fcrepo.db.password:}")
055    private String dbPassword;
056
057    @Value("${fcrepo.db.max.pool.size:10}")
058    private Integer maxPoolSize;
059
060    @Value("${fcrepo.db.connection.checkout.timeout:30000}")
061    private Integer checkoutTimeout;
062
063    private static final Map<String, String> DB_DRIVER_MAP = Map.of(
064            "h2", "org.h2.Driver",
065            "postgresql", "org.postgresql.Driver",
066            "mariadb", "org.mariadb.jdbc.Driver",
067            "mysql", "com.mysql.cj.jdbc.Driver"
068    );
069
070    @PostConstruct
071    public void setup() {
072        ((ConverterRegistry) DefaultConversionService.getSharedInstance())
073                // Adds a converter for mapping local datetimes to instants. This is dubious and not supported
074                // by default because you must make an assumption about the timezone
075                .addConverter(new Converter<LocalDateTime, Instant>() {
076                    @Override
077                    public Instant convert(final LocalDateTime source) {
078                        return source.toInstant(ZoneOffset.UTC);
079                    }
080                });
081    }
082
083    @Bean
084    public DataSource dataSource() throws Exception {
085        final var driver = identifyDbDriver();
086
087        LOGGER.info("JDBC URL: {}", dbUrl);
088        LOGGER.info("JDBC User: {}", dbUser);
089        LOGGER.info("JDBC Password length: {}", dbPassword == null ? 0 : dbPassword.length());
090        LOGGER.info("Using database driver: {}", driver);
091
092        final var dataSource = new HikariDataSource();
093        dataSource.setDriverClassName(driver);
094        dataSource.setJdbcUrl(dbUrl);
095        dataSource.setUsername(dbUser);
096        dataSource.setPassword(dbPassword);
097        dataSource.setConnectionTimeout(checkoutTimeout);
098        dataSource.setMaximumPoolSize(maxPoolSize);
099
100        flyway(dataSource);
101
102        return dataSource;
103    }
104
105    /**
106     * Get the database type in use
107     * @return database type from the connect url.
108     */
109    private String getDbType() {
110        final var parts = dbUrl.split(":");
111
112        if (parts.length < 2) {
113            throw new IllegalArgumentException("Invalid DB url: " + dbUrl);
114        }
115        return parts[1].toLowerCase();
116    }
117
118    private String identifyDbDriver() {
119        final var driver = DB_DRIVER_MAP.get(getDbType());
120
121        if (driver == null) {
122            throw new IllegalStateException("No database driver found for: " + dbUrl);
123        }
124
125        return driver;
126    }
127
128    @Bean
129    public DataSourceTransactionManager txManager(final DataSource dataSource) {
130        final var txManager = new DataSourceTransactionManager();
131        txManager.setDataSource(dataSource);
132        return txManager;
133    }
134
135    @Bean
136    public TransactionTemplate txTemplate(final PlatformTransactionManager txManager) {
137        final var txDefinition = new DefaultTransactionDefinition();
138        txDefinition.setIsolationLevel(TransactionDefinition.ISOLATION_READ_COMMITTED);
139        return new TransactionTemplate(txManager, txDefinition);
140    }
141
142    @Bean
143    public Flyway flyway(final DataSource source) throws Exception {
144        LOGGER.debug("Instantiating a new flyway bean");
145        return FlywayFactory.create().setDataSource(source).setDatabaseType(getDbType()).getObject();
146    }
147
148}
