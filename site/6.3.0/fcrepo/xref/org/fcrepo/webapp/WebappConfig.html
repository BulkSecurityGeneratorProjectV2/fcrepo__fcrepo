7   package org.fcrepo.webapp;
8   
9   import java.util.Optional;
10  import java.util.concurrent.ExecutorService;
11  import java.util.concurrent.Executors;
12  import java.util.concurrent.TimeUnit;
13  
14  import javax.inject.Inject;
15  import javax.servlet.Filter;
16  
17  import org.fcrepo.config.FedoraPropsConfig;
18  import org.fcrepo.http.api.ExternalContentHandlerFactory;
19  import org.fcrepo.http.api.ExternalContentPathValidator;
20  import org.fcrepo.kernel.api.auth.ACLHandle;
21  import org.fcrepo.kernel.api.rdf.RdfNamespaceRegistry;
22  
23  import org.apache.http.conn.HttpClientConnectionManager;
24  import org.apache.http.impl.conn.PoolingHttpClientConnectionManager;
25  import org.fcrepo.persistence.ocfl.RepositoryInitializationFilter;
26  import org.slf4j.Logger;
27  import org.slf4j.LoggerFactory;
28  import org.springframework.context.annotation.Bean;
29  import org.springframework.context.annotation.Configuration;
30  import org.springframework.scheduling.annotation.EnableAsync;
31  import org.springframework.scheduling.annotation.EnableScheduling;
32  import org.springframework.scheduling.concurrent.ThreadPoolTaskScheduler;
33  
34  import com.github.benmanes.caffeine.cache.Cache;
35  import com.github.benmanes.caffeine.cache.Caffeine;
36  import com.google.common.eventbus.AsyncEventBus;
37  import com.google.common.eventbus.EventBus;
38  
39  /**
40   * Spring config for the webapp
41   *
42   * @author pwinckles
43   */
44  @Configuration
45  @EnableAsync
46  @EnableScheduling
47  public class WebappConfig {
48  
49      private static final Logger LOGGER = LoggerFactory.getLogger(WebappConfig.class);
50  
51      @Inject
52      private FedoraPropsConfig fedoraPropsConfig;
53  
54      /**
55       * Task scheduler used for cleaning up transactions
56       *
57       * @return scheduler
58       */
59      @Bean
60      public ThreadPoolTaskScheduler taskScheduler() {
61          final var scheduler = new ThreadPoolTaskScheduler();
62          scheduler.setPoolSize(1);
63          scheduler.setThreadNamePrefix("ScheduledTask");
64          return scheduler;
65      }
66  
67      /**
68       * HTTP connection manager
69       *
70       * @return connection manager
71       */
72      @Bean
73      public HttpClientConnectionManager connectionManager() {
74          return new PoolingHttpClientConnectionManager();
75      }
76  
77      /**
78       * Fedora's lightweight internal event bus. Currently memory-resident.
79       *
80       * @param propsConfig config
81       * @return event bus
82       */
83      @Bean
84      public EventBus eventBus(final FedoraPropsConfig propsConfig) {
85          return new AsyncEventBus(eventBusExecutor(propsConfig));
86      }
87  
88      /**
89       * @param propsConfig config
90       * @return executor intended to be used by the Guava event bus
91       */
92      @Bean
93      public ExecutorService eventBusExecutor(final FedoraPropsConfig propsConfig) {
94          LOGGER.debug("Event bus threads: {}", propsConfig);
95          return Executors.newFixedThreadPool(propsConfig.getEventBusThreads());
96      }
97  
98      /**
99       * Configuration of namespace prefixes
100      *
101      * @param propsConfig config properties
102      * @return rdf namespace registry
103      */
104     @Bean(initMethod = "init", destroyMethod = "shutdown")
105     public RdfNamespaceRegistry rdfNamespaceRegistry(final FedoraPropsConfig propsConfig) {
106         final var registry = new RdfNamespaceRegistry();
107         registry.setConfigPath(propsConfig.getNamespaceRegistry());
108         registry.setMonitorForChanges(true);
109         return registry;
110     }
111 
112     /**
113      * External content configuration
114      *
115      * @param propsConfig config properties
116      * @return external content path validator
117      */
118     @Bean(initMethod = "init", destroyMethod = "shutdown")
119     public ExternalContentPathValidator externalContentPathValidator(final FedoraPropsConfig propsConfig) {
120         final var validator = new ExternalContentPathValidator();
121         validator.setConfigPath(propsConfig.getExternalContentAllowed());
122         validator.setMonitorForChanges(true);
123         return validator;
124     }
125 
126     @Bean
127     public ExternalContentHandlerFactory externalContentHandlerFactory(final ExternalContentPathValidator validator) {
128         final var factory = new ExternalContentHandlerFactory();
129         factory.setValidator(validator);
130         return factory;
131     }
132 
133     /**
134      * Used to cache the effective ACL location and authorizations for a given resource.
135      *
136      * @return the cache
137      */
138     @Bean
139     public Cache<String, Optional<ACLHandle>> authHandleCache() {
140         return Caffeine.newBuilder().weakValues()
141                 .expireAfterAccess(fedoraPropsConfig.getWebacCacheTimeout(), TimeUnit.MINUTES)
142                 .maximumSize(fedoraPropsConfig.getWebacCacheSize()).build();
143     }
144 
145     /**
146      * Filter to prevent http requests during repo init
147      *
148      * @return the filter
149      */
150     @Bean
151     public Filter repositoryInitializationFilter() {
152         return new RepositoryInitializationFilter();
153     }
154 
155 }
