6   package org.fcrepo.persistence.api;
7   
8   import org.fcrepo.kernel.api.Transaction;
9   
10  /**
11   * Interface to access PersistentStorageSessions.
12   *
13   * @author whikloj
14   * @author dbernstein
15   * @since 2019-09-19
16   */
17  public interface PersistentStorageSessionManager {
18  
19      /**
20       * Retrieve a PersistentStorageSession.
21       *
22       * @param transaction the externally generated transaction.
23       * @return the PersistentStorageSession instance.
24       */
25      PersistentStorageSession getSession(final Transaction transaction);
26  
27      /**
28       * Retrieve a read-only PersistentStorageSession. Clients should expect
29       * invocation on storage modifying methods to throw exception.
30       *
31       * @return the PersistentStorageSession instance.
32       */
33      PersistentStorageSession getReadOnlySession();
34  
35      /**
36       * Removes the indicated session. If the session does not exist, null is returned.
37       *
38       * @param sessionId the id of the session to remove
39       * @return the session, if it exists
40       */
41      PersistentStorageSession removeSession(final String sessionId);
42  
43  }
