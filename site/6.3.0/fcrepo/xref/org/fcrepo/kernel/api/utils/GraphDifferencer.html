6   package org.fcrepo.kernel.api.utils;
7   
8   import java.util.stream.Stream;
9   
10  import org.apache.jena.graph.Graph;
11  import org.apache.jena.graph.Triple;
12  import org.apache.jena.rdf.model.Model;
13  
14  import static java.util.Spliterator.IMMUTABLE;
15  import static java.util.Spliterators.spliteratorUnknownSize;
16  import static java.util.stream.StreamSupport.stream;
17  import static org.apache.jena.graph.Node.ANY;
18  import static org.apache.jena.sparql.graph.GraphFactory.createDefaultGraph;
19  
20  /**
21   * A wrapping {@link Stream} that calculates two differences between a
22   * {@link Graph} A and a source Stream B. The differences are (A - (A ∩ B)) and
23   * (B - (A ∩ B)). The ordinary output of this stream is (B - (A ∩ B)), and
24   * after exhaustion, sets containing (A - (A ∩ B)) and (A ∩ B) are available.
25   *
26   * @author ajs6f
27   * @author acoburn
28   * @since Oct 24, 2013
29   */
30  public class GraphDifferencer {
31  
32      private final Graph notCommon;
33  
34      private final Graph common;
35  
36      private final Stream.Builder<Triple> source = Stream.builder();
37  
38      /**
39       * Diff a Model against a stream of triples
40       *
41       * @param replacement the replacement
42       * @param original the original
43       */
44      public GraphDifferencer(final Model replacement,
45                                       final Stream<Triple> original) {
46          this(replacement.getGraph(), original);
47      }
48  
49      /**
50       * Diff a graph against a stream of triples
51       *
52       * @param replacement the replacement
53       * @param original the original
54       */
55      public GraphDifferencer(final Graph replacement,
56                                       final Stream<Triple> original) {
57          notCommon = replacement;
58          common = createDefaultGraph();
59          original.forEach(x -> {
60              synchronized (this) {
61                  if (notCommon.contains(x)) {
62                      notCommon.remove(x.getSubject(), x.getPredicate(), x.getObject());
63                      common.add(x);
64                  } else if (!common.contains(x)) {
65                      source.accept(x);
66                  }
67              }
68          });
69      }
70  
71      /**
72       * This method returns the difference between the two input sources.
73       *
74       * @return The differences between the two inputs.
75       */
76      public Stream<Triple> difference() {
77          return source.build();
78      }
79  
80      /**
81       * This method will return null until the source iterator is exhausted.
82       *
83       * @return The elements that turned out to be common to the two inputs.
84       */
85      public Stream<Triple> common() {
86          return stream(spliteratorUnknownSize(common.find(ANY, ANY, ANY), IMMUTABLE), false);
87      }
88  
89      /**
90       * This method will return null until the source iterator is exhausted.
91       *
92       * @return The elements that turned out not to be common to the two inputs.
93       */
94      public Stream<Triple> notCommon() {
95          return stream(spliteratorUnknownSize(notCommon.find(ANY, ANY, ANY), IMMUTABLE), false);
96      }
97  }
