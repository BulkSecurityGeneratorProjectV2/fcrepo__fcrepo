6   package org.fcrepo.kernel.api.exception;
7   
8   import static java.time.ZoneOffset.UTC;
9   import static java.time.format.DateTimeFormatter.ISO_INSTANT;
10  
11  import java.time.format.DateTimeFormatter;
12  import java.util.Objects;
13  
14  import org.fcrepo.kernel.api.models.FedoraResource;
15  
16  /**
17   * Exception when a Tombstone {@link org.fcrepo.kernel.api.models.FedoraResource}
18   * is used where a real object is expected
19   *
20   * @author cabeer
21   * @since 10/16/14
22   */
23  public class TombstoneException extends RepositoryRuntimeException {
24  
25      private static final long serialVersionUID = 1L;
26  
27      private final String uri;
28  
29      private static DateTimeFormatter isoFormatter = ISO_INSTANT.withZone(UTC);
30  
31      /**
32       * Construct a new tombstone exception for a resource
33       * @param resource the fedora resource
34       */
35      public TombstoneException(final FedoraResource resource) {
36          this(resource, null);
37      }
38  
39      /**
40       * Create a new tombstone exception with a URI to the tombstone resource
41       * @param resource the fedora resource
42       * @param tombstoneUri the uri to the tombstone resource for the Link header.
43       */
44      public TombstoneException(final FedoraResource resource, final String tombstoneUri) {
45          super("Discovered tombstone resource at " + resource.getFedoraId().getFullIdPath() +
46                  (Objects.nonNull(resource.getLastModifiedDate()) ? ", departed at: " +
47                  isoFormatter.format(resource.getLastModifiedDate()) : ""));
48          this.uri = tombstoneUri;
49      }
50  
51      /**
52       * Get a URI to the tombstone resource
53       * @return the URI to the tombstone resource
54       */
55      public String getURI() {
56          return uri;
57      }
58  }
