6   package org.fcrepo.kernel.api.models;
7   
8   import java.util.stream.Stream;
9   
10  import org.fcrepo.kernel.api.Transaction;
11  import org.fcrepo.kernel.api.exception.PathNotFoundException;
12  import org.fcrepo.kernel.api.identifiers.FedoraId;
13  
14  /**
15   * Interface to a factory to instantiate FedoraResources
16   *
17   * @author whikloj
18   * @since 2019-09-23
19   */
20  public interface ResourceFactory {
21  
22      /**
23       * Get a FedoraResource for existing resource
24       *
25       * @param transaction The transaction associated with this request or null if not in a transaction.
26       * @param fedoraID The identifier for the resource.
27       * @return The resource.
28       * @throws PathNotFoundException If the identifier cannot be found.
29       */
30      public FedoraResource getResource(final Transaction transaction, final FedoraId fedoraID)
31              throws PathNotFoundException;
32  
33      /**
34       * Get a resource as a particular type
35       *
36       * @param <T> type for the resource
37       * @param transaction The transaction associated with this request or null
38       * @param fedoraID The identifier for the resource.
39       * @param clazz class the resource will be cast to
40       * @return The resource.
41       * @throws PathNotFoundException If the identifier cannot be found.
42       */
43      public <T extends FedoraResource> T getResource(final Transaction transaction, final FedoraId fedoraID,
44                                                      final Class<T> clazz) throws PathNotFoundException;
45  
46      /**
47       * Get the containing resource (if exists).
48       * @param transaction The current transaction
49       * @param resourceId The internal identifier
50       * @return The containing resource or null if none.
51       */
52      public FedoraResource getContainer(final Transaction transaction, final FedoraId resourceId);
53  
54      /**
55       * Get immediate children of the resource
56       * @param transaction The transaction
57       * @param resourceId Identifier of the resource
58       * @return Stream of child resources
59       */
60      public Stream<FedoraResource> getChildren(final Transaction transaction, final FedoraId resourceId);
61  }
