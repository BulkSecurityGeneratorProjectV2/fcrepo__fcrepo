29  /**
30   * Utility class for helper methods.
31   * @author whikloj
32   * @since 6.0.0
33   */
34  @Component
35  public class ResourceHelperImpl implements ResourceHelper {
36  
37      private static final Logger LOGGER = getLogger(ResourceHeadersImpl.class);
38  
39      @Inject
40      private PersistentStorageSessionManager persistentStorageSessionManager;
41  
42      @Autowired
43      @Qualifier("containmentIndex")
44      private ContainmentIndex containmentIndex;
45  
46      @Override
47      public boolean isGhostNode(final Transaction transaction, final FedoraId resourceId) {
48          if (!doesResourceExist(transaction, resourceId, true)) {
49              return containmentIndex.hasResourcesStartingWith(transaction, resourceId);
50          }
51          return false;
52      }
53  
54      @Override
55      public boolean doesResourceExist(final Transaction transaction, final FedoraId fedoraId,
56                                       final boolean includeDeleted) {
57          if (fedoraId.isRepositoryRoot()) {
58              // Root always exists.
59              return true;
60          }
61          if (!(fedoraId.isMemento() || fedoraId.isAcl())) {
62              // containment index doesn't handle versions and only tells us if the resource (not acl) is there,
63              // so don't bother checking for them.
64              return containmentIndex.resourceExists(transaction, fedoraId, includeDeleted);
65          } else {
66  
67              final PersistentStorageSession psSession = getSession(transaction);
68  
69              try {
70                  // Resource ID for metadata or ACL contains their individual endopoints (ie. fcr:metadata, fcr:acl)
71                  final ResourceHeaders headers = psSession.getHeaders(fedoraId, fedoraId.getMementoInstant());
72                  return !headers.isDeleted();
73              } catch (final PersistentItemNotFoundException e) {
74                  // Object doesn't exist.
75                  return false;
76              } catch (final PersistentStorageException e) {
77                  // Other error, pass along.
78                  throw new RepositoryRuntimeException(e.getMessage(), e);
79              }
80          }
81      }
82  
83      /**
84       * Get a session for this interaction.
85       *
86       * @param transaction The supplied transaction.
87       * @return a storage session.
88       */
89      private PersistentStorageSession getSession(final Transaction transaction) {
90          final PersistentStorageSession session;
91          if (transaction.isReadOnly() || !transaction.isOpen()) {
92              session = persistentStorageSessionManager.getReadOnlySession();
93          } else {
94              session = persistentStorageSessionManager.getSession(transaction);
95          }
96          return session;
97      }
98  }
