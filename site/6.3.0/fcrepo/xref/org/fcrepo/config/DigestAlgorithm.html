6   package org.fcrepo.config;
7   
8   import static java.util.stream.Collectors.toSet;
9   
10  import java.util.Arrays;
11  import java.util.Set;
12  import java.util.stream.Stream;
13  
14  /**
15   * Digest Algorith enum
16   * @author cbeer
17   *
18   * Moved to its own class for Fedora 6.0.0
19   */
20  public enum DigestAlgorithm {
21      SHA1("SHA", "urn:sha1", "sha-1", "sha1"),
22      SHA256("SHA-256", "urn:sha-256", "sha256"),
23      SHA512("SHA-512", "urn:sha-512", "sha512"),
24      SHA512256("SHA-512/256", "urn:sha-512/256", "sha512/256"),
25      MD5("MD5", "urn:md5"),
26      MISSING("NONE", "missing");
27  
28      final private String algorithm;
29      final private String scheme;
30      final private Set<String> aliases;
31  
32      DigestAlgorithm(final String alg, final String scheme, final String... aliases) {
33          this.algorithm = alg;
34          this.scheme = scheme;
35          this.aliases = Stream.concat(Arrays.stream(aliases), Stream.of(algorithm)).map(String::valueOf)
36                          .map(String::toLowerCase).collect(toSet());
37      }
38  
39      /**
40       * Return the scheme associated with the provided algorithm (e.g. SHA-1 returns urn:sha1)
41       *
42       * @param alg for which scheme is requested
43       * @return scheme
44       */
45      public static String getScheme(final String alg) {
46          return Arrays.stream(values()).filter(value ->
47                  value.algorithm.equalsIgnoreCase(alg) || value.algorithm.replace("-", "").equalsIgnoreCase(alg)
48          ).findFirst().orElse(MISSING).scheme;
49      }
50  
51      /**
52       * Return enum value for the provided scheme (e.g. urn:sha1 returns SHA-1)
53       *
54       * @param argScheme for which enum is requested
55       * @return enum value associated with the arg scheme
56       */
57      public static DigestAlgorithm fromScheme(final String argScheme) {
58          return Arrays.stream(values()).filter(value -> value.scheme.equalsIgnoreCase(argScheme)
59          ).findFirst().orElse(MISSING);
60      }
61  
62      /**
63       * Return enum value for the provided algorithm
64       *
65       * @param alg algorithm name to seek
66       * @return enum value associated with the algorithm name, or missing if not found
67       */
68      public static DigestAlgorithm fromAlgorithm(final String alg) {
69          final String seek = alg.toLowerCase();
70          return Arrays.stream(values())
71                  .filter(value -> value.aliases.contains(seek))
72                  .findFirst()
73                  .orElse(MISSING);
74      }
75  
76      /**
77       * Return true if the provided algorithm is included in this enum
78       *
79       * @param alg to test
80       * @return true if arg algorithm is supported
81       */
82      public static boolean isSupportedAlgorithm(final String alg) {
83          return !getScheme(alg).equals(MISSING.scheme);
84      }
85  
86      /**
87       * @return the aliases
88       */
89      public Set<String> getAliases() {
90          return aliases;
91      }
92  
93      /**
94       * @return the algorithm
95       */
96      public String getAlgorithm() {
97          return algorithm;
98      }
99  }
