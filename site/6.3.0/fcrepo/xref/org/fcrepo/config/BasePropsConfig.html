7   package org.fcrepo.config;
8   
9   import java.io.IOException;
10  import java.nio.file.FileAlreadyExistsException;
11  import java.nio.file.Files;
12  import java.nio.file.Path;
13  
14  import org.springframework.context.annotation.PropertySource;
15  import org.springframework.context.annotation.PropertySources;
16  
17  /**
18   * A base class for property configs
19   *
20   * @author dbernstein
21   * @since 6.0.0
22   */
23  @PropertySources({
24          @PropertySource(value = BasePropsConfig.DEFAULT_FCREPO_CONFIG_FILE_PROP_SOURCE,
25                  ignoreResourceNotFound = true),
26          @PropertySource(value = BasePropsConfig.FCREPO_CONFIG_FILE_PROP_SOURCE, ignoreResourceNotFound = true)
27  })
28  abstract class BasePropsConfig {
29  
30      public static final String FCREPO_HOME_PROPERTY = "fcrepo.home";
31      public static final String DEFAULT_FCREPO_HOME_VALUE = "fcrepo-home";
32      public static final String DEFAULT_FCREPO_CONFIG_FILE_PROP_SOURCE =
33              "file:${" + FCREPO_HOME_PROPERTY + ":" + DEFAULT_FCREPO_HOME_VALUE + "}/config/fcrepo.properties";
34      public static final String FCREPO_CONFIG_FILE_PROP_SOURCE = "file:${fcrepo.config.file}";
35  
36      protected Path createDirectories(final Path path) throws IOException {
37          try {
38              return Files.createDirectories(path);
39          } catch (final FileAlreadyExistsException e) {
40              // Ignore. This only happens with the path is a symlink
41              return path;
42          }
43      }
44  
45  }
