30  /**
31   * Provides serialization for streaming RDF results.
32   *
33   * @author ajs6f
34   * @since Nov 19, 2013
35   */
36  @Provider
37  @Produces({TURTLE, N3, N3_ALT2, RDF_XML, NTRIPLES, TEXT_PLAIN, JSON_LD})
38  public class RdfStreamProvider implements MessageBodyWriter<RdfNamespacedStream> {
39  
40      private static final Logger LOGGER = getLogger(RdfStreamProvider.class);
41  
42      @Override
43      public boolean isWriteable(final Class<?> type, final Type genericType,
44              final Annotation[] annotations, final MediaType mediaType) {
45          LOGGER.debug(
46                  "Checking to see if we can serialize type: {} to mimeType: {}",
47                  type.getName(), mediaType.toString());
48          if (!RdfNamespacedStream.class.isAssignableFrom(type)) {
49              return false;
50          }
51          if (mediaType.equals(TEXT_HTML_TYPE)
52                  || (mediaType.getType().equals("application") && mediaType
53                          .getSubtype().equals("html"))) {
54              LOGGER.debug("Was asked for an HTML mimeType, returning false.");
55              return false;
56          }
57          LOGGER.debug("Assuming that this is an attempt to retrieve RDF, returning true.");
58          return true;
59      }
60  
61      @Override
62      public long getSize(final RdfNamespacedStream t, final Class<?> type,
63              final Type genericType, final Annotation[] annotations,
64              final MediaType mediaType) {
65          // We do not know how long the stream is
66          return -1;
67      }
68  
69      @Override
70      public void writeTo(final RdfNamespacedStream nsStream, final Class<?> type,
71          final Type genericType, final Annotation[] annotations,
72          final MediaType mediaType,
73          final MultivaluedMap<String, Object> httpHeaders,
74          final OutputStream entityStream) {
75  
76          LOGGER.debug("Serializing an RdfStream to mimeType: {}", mediaType);
77          final RdfStreamStreamingOutput streamOutput = new RdfStreamStreamingOutput(nsStream.stream,
78                  nsStream.namespaces, mediaType);
79          streamOutput.write(entityStream);
80      }
81  }
