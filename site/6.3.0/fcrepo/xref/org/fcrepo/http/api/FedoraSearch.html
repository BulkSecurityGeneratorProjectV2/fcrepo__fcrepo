52  
53      @Autowired
54      @Qualifier("searchIndex")
55      private SearchIndex searchIndex;
56  
57      /**
58       * Default JAX-RS entry point
59       */
60      public FedoraSearch() {
61          super();
62      }
63  
64      /**
65       * Perform simple search on the repository
66       *
67       * @param conditions The conditions constraining the query
68       * @param fields     The fields to return in results
69       * @param maxResults The max number of results to return
70       * @param offset     The zero-based offset of the first result to be returned
71       * @param order      The order: ie "asc" or "desc"
72       * @param orderBy    The field by which to order the results
73       * @param includeTotalResultCount A flag for including total result count (false by default)
74       * @return A response object with the search results
75       */
76      @GET
77      @Produces({APPLICATION_JSON + ";qs=1.0",
78              TEXT_PLAIN_WITH_CHARSET,
79              TEXT_HTML_WITH_CHARSET})
80      public Response doSearch(@QueryParam(value = "condition") final List<String> conditions,
81                               @QueryParam(value = "fields") final String fields,
82                               @DefaultValue("100") @QueryParam("max_results") final int maxResults,
83                               @DefaultValue("0") @QueryParam("offset") final int offset,
84                               @DefaultValue("asc") @QueryParam("order") final String order,
85                               @QueryParam("order_by") final String orderBy,
86                               @DefaultValue("false") @QueryParam("include_total_result_count")
87                                           final boolean includeTotalResultCount) {
88  
89          LOGGER.info("GET on search with conditions: {}, and fields: {}", conditions, fields);
90          try {
91              final var conditionList = new ArrayList<Condition>();
92              for (final String condition : conditions) {
93                  final var parsedCondition = parse(condition, identifierConverter());
94                  conditionList.add(parsedCondition);
95              }
96  
97              List<Condition.Field> parsedFields = null;
98              if (StringUtils.isBlank(fields) || fields.equals("*")) {
99                  parsedFields = Arrays.asList(Condition.Field.values());
100             } else {
101                 parsedFields = new ArrayList<>();
102                 for (final String field : fields.split(",")) {
103                     try {
104                         parsedFields.add(Condition.Field.fromString(field));
105                     } catch (final Exception e) {
106                         throw new InvalidQueryException("The field \"" + field + "\" is not a valid output field.");
107                     }
108                 }
109             }
110 
111             final Condition.Field orderByField;
112             try {
113                 if (!StringUtils.isBlank(orderBy)) {
114                     orderByField = Condition.Field.fromString(orderBy);
115                 } else {
116                     orderByField = null;
117                 }
118             } catch (final Exception e) {
119                 throw new InvalidQueryException("The order_by field must contain a valid value such as " +
120                         StringUtils.join(Condition.Field.values(), ","));
121             }
122 
123             if (!(order.equalsIgnoreCase("asc") || order.equalsIgnoreCase("desc"))) {
124                 throw new InvalidQueryException("The order field is invalid:  valid values are \"asc\" and \"desc\"");
125             }
126 
127             final var params = new SearchParameters(parsedFields, conditionList, maxResults, offset, orderByField,
128                     order, includeTotalResultCount);
129             final Response.ResponseBuilder builder = ok();
130             final var result = this.searchIndex.doSearch(params);
131             final var translatedResults = translateResults(result);
132 
133             builder.entity(translatedResults);
134             return builder.build();
135         } catch (final InvalidConditionExpressionException | InvalidQueryException ex) {
136             throw new BadRequestException(ex.getMessage(), ex);
137         }
138     }
139 
140     private SearchResult translateResults(final SearchResult result) {
141         result.getItems().forEach(item -> {
142             final var key = Condition.Field.FEDORA_ID.toString();
143             final var fedoraId = item.get(key);
144             if (fedoraId != null) {
145                 item.put(key, identifierConverter().toExternalId(fedoraId.toString()));
146             }
147         });
148         return result;
149     }
150 
151     /**
152      * Parses the url decoded value of a single parameter passed by the
153      * http layer into a {@link Condition}.
154      *
155      * @param expression The url decoded value of the condition parameter.
156      * @return the parsed {@link Condition} object.
157      */
158     protected static Condition parse(final String expression, final HttpIdentifierConverter converter)
159             throws InvalidConditionExpressionException {
160         final Condition condition = Condition.fromExpression(expression);
161         if (condition.getField().equals(Condition.Field.FEDORA_ID)) {
162             //convert the object value to an internal identifier stem where appropriate
163             final var object = condition.getObject();
164             final var field = condition.getField();
165             final var operator = condition.getOperator();
166             if (!object.startsWith(FEDORA_ID_PREFIX) && isExternalUrl(object)) {
167                 return Condition.fromEnums(field, operator, converter.toInternalId(object));
168             } else if (object.startsWith("/")) {
169                 return Condition.fromEnums(field, operator, FEDORA_ID_PREFIX + object);
170             } else if (!object.startsWith(FEDORA_ID_PREFIX) && !object.equals("*")) {
171                 return Condition.fromEnums(field, operator, FEDORA_ID_PREFIX + "/" + object);
172             }
173         }
174 
175         return condition;
176     }
177 
178     private static boolean isExternalUrl(final String str) {
179         try {
180             new URL(str);
181             return true;
182         } catch (final Exception ex) {
183             return false;
184         }
185     }
