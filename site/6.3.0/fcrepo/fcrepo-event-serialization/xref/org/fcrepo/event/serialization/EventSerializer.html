6   package org.fcrepo.event.serialization;
7   
8   import static java.nio.charset.StandardCharsets.UTF_8;
9   import static org.apache.jena.rdf.model.ModelFactory.createDefaultModel;
10  
11  import java.io.ByteArrayInputStream;
12  
13  import org.apache.jena.rdf.model.Model;
14  import org.fcrepo.kernel.api.observer.Event;
15  
16  /**
17   * A basic serialization API for Fedora events
18   * @author acoburn
19   * @author dbernstein
20   */
21  public interface EventSerializer {
22  
23      /**
24       * Convert an event to an Rdf Model
25       * @param evt the Fedora event
26       * @return an RDF model representing the event
27       */
28      static Model toModel(final Event evt) {
29          final EventSerializer serializer = new JsonLDSerializer();
30          final String json = serializer.serialize(evt);
31          final Model model = createDefaultModel();
32          final String baseUrl = evt.getBaseUrl();
33          model.read(new ByteArrayInputStream(json.getBytes(UTF_8)), baseUrl + evt.getPath(), "JSON-LD");
34          return model;
35      }
36  
37      /**
38       * Serialize a Event into a JSON String
39       * @param evt the Fedora event
40       * @return a JSON string
41       */
42      String serialize(final Event evt);
43  }
