9   import static org.slf4j.LoggerFactory.getLogger;

11  import static javax.ws.rs.core.Response.Status.CONFLICT;
12  import static javax.ws.rs.core.Response.status;

15  import javax.ws.rs.core.Context;
16  import javax.ws.rs.core.Link;
17  import javax.ws.rs.core.Response;
18  import javax.ws.rs.core.UriInfo;
19  import javax.ws.rs.ext.Provider;

21  import org.fcrepo.kernel.api.exception.ConstraintViolationException;
22  import org.fcrepo.kernel.api.exception.MultipleConstraintViolationException;
23  import org.fcrepo.kernel.api.exception.RelaxableServerManagedPropertyException;
24  import org.fcrepo.kernel.api.exception.ServerManagedPropertyException;
25  
26  import org.slf4j.Logger;

28  /**
29   * Mapper to display all the various constrainedby links and messages.
30   * @author whikloj
31   */
32  @Provider
33  public class MultipleConstraintViolationExceptionMapper extends
34          ConstraintExceptionMapper<MultipleConstraintViolationException> implements ExceptionDebugLogging {
35  
36      private static final Logger LOGGER = getLogger(MultipleConstraintViolationExceptionMapper.class);
37  
38      @Context
39      private UriInfo uriInfo;
40  
41      @Context
42      private ServletContext context;
43  
44      @Override
45      public Response toResponse(final MultipleConstraintViolationException e) {
46          debugException(this, e, LOGGER);
47  
48          final String msg = e.getMessage();
49          final Response.ResponseBuilder response = status(CONFLICT).entity(msg).type(TEXT_PLAIN_WITH_CHARSET);
50          final Link[] constraintLinks = e.getExceptionTypes().stream().map(ConstraintViolationException::getClass)
51                  .distinct().map(c -> {
52                      // Avoid building a link with the relaxable sub-class which would require another constraint RDF
53                      // file.
54                      if (c.equals(RelaxableServerManagedPropertyException.class)) {
55                          return ServerManagedPropertyException.class;
56                      }
57                      return c;
58                  }).map(c -> buildConstraintLink(c,
59                          context,
60                          uriInfo))
61                  .toArray(Link[]::new);
62          return response.links(constraintLinks).build();
63      }
64  
65  }
