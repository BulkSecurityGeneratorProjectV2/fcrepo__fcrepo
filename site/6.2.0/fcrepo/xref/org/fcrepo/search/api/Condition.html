6   package org.fcrepo.search.api;
7   
8   import java.util.regex.Pattern;
9   
10  /**
11   * A data structure representing a search condition.
12   *
13   * @author dbernstein
14   */
15  public class Condition {
16      public enum Operator {
17          LTE("<="),
18          GTE(">="),
19          EQ("="),
20          GT(">"),
21          LT("<");
22  
23          private String value;
24  
25          Operator(final String value) {
26              this.value = value;
27          }
28  
29          public String getStringValue() {
30              return this.value;
31          }
32  
33          public static Operator fromString(final String str) {
34              for (final Operator o : Operator.values()) {
35                  if (o.value.equals(str)) {
36                      return o;
37                  }
38              }
39  
40              throw new IllegalArgumentException("Value " + str + " not recognized.");
41          }
42  
43      }
44  
45      public enum Field {
46          FEDORA_ID,
47          MODIFIED,
48          CREATED,
49          CONTENT_SIZE,
50          MIME_TYPE,
51          RDF_TYPE;
52  
53          @Override
54          public String toString() {
55              return super.toString().toLowerCase();
56          }
57  
58          public static Field fromString(final String fieldStr) {
59              return Field.valueOf(fieldStr.toUpperCase());
60          }
61      }
62  
63      /* A regex for parsing the value of a "condition" query  parameter which follows the format
64       * [field_name][operation][object]
65       * The field name is composed of at least one character and can contain alpha number characters and underscores.
66       * The operation can equal "=", "<", ">", "<=" or ">="
67       * The object can be anything but cannot start with >, <, and =.
68       */
69      final static Pattern CONDITION_REGEX = Pattern.compile("([a-zA-Z0-9_]+)([><=]|<=|>=)([^><=].*)");
70  
71  
72      private Field field;
73      private Operator operator;
74      private String object;
75  
76      /**
77       * Internal constructor
78       *
79       * @param field    The search field (condition subject)
80       * @param operator The operator (condition predicate)
81       * @param object   The object (condition object)
82       */
83      private Condition(final Field field, final Operator operator, final String object) {
84          this.field = field;
85          this.operator = operator;
86          this.object = object;
87      }
88  
89      /**
90       * Field accessor
91       *
92       * @return the field
93       */
94      public Field getField() {
95          return field;
96      }
97  
98      /**
99       * Operator accessor
100      * @return the operator
101      */
102     public Operator getOperator() {
103         return operator;
104     }
105 
106     /**
107      * @return the object portion of the condition
108      */
109     public String getObject() {
110         return object;
111     }
112 
113     @Override
114     public String toString() {
115         return this.field.toString().toLowerCase() + operator + object;
116     }
117 
118     /**
119      * Parses a string expression into a Condition object.
120      * @param expression The condition as a string expression.
121      * @return The condition
122      * @throws InvalidConditionExpressionException if we can't parse the string into a Condition.
123      */
124     public static Condition fromExpression(final String expression) throws InvalidConditionExpressionException {
125         final var m = CONDITION_REGEX.matcher(expression);
126         if (m.matches()) {
127             final var field = Field.fromString(m.group(1));
128             final var operation = Operator.fromString(m.group(2));
129             final var object = m.group(3);
130             return fromEnums(field, operation, object);
131         }
132 
133         throw new InvalidConditionExpressionException(expression);
134     }
135 
136     public static Condition fromEnums(final Field field, final Operator operator, final String expression) {
137         return new Condition(field, operator, expression);
138     }
139 }
