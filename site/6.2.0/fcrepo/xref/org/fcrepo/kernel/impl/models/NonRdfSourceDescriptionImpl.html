29  /**
30   * Implementation of a non-rdf source description
31   *
32   * @author bbpennel
33   */
34  public class NonRdfSourceDescriptionImpl extends FedoraResourceImpl implements NonRdfSourceDescription {
35  
36      private static final URI RDF_SOURCE_URI = URI.create(RDF_SOURCE.getURI());
37  
38      /**
39       * Construct a description resource
40       *
41       * @param fedoraID internal identifier
42       * @param transaction transaction
43       * @param pSessionManager session manager
44       * @param resourceFactory resource factory
45       * @param userTypesCache the user types cache
46       */
47      public NonRdfSourceDescriptionImpl(final FedoraId fedoraID,
48                                         final Transaction transaction,
49                                         final PersistentStorageSessionManager pSessionManager,
50                                         final ResourceFactory resourceFactory,
51                                         final UserTypesCache userTypesCache) {
52          super(fedoraID, transaction, pSessionManager, resourceFactory, userTypesCache);
53      }
54  
55      @Override
56      public String getId() {
57          return getFedoraId().getResourceId();
58      }
59  
60      @Override
61      public FedoraResource getDescribedResource() {
62          // Get a FedoraId for the binary
63          FedoraId describedId = getFedoraId().asBaseId();
64          if (getFedoraId().isMemento()) {
65              describedId = describedId.asMemento(getFedoraId().getMementoInstant());
66          }
67          try {
68              return this.resourceFactory.getResource(transaction, describedId);
69          } catch (final PathNotFoundException e) {
70              throw new PathNotFoundRuntimeException(e.getMessage(), e);
71          }
72      }
73  
74      @Override
75      public List<URI> getSystemTypes(final boolean forRdf) {
76          var types = resolveSystemTypes(forRdf);
77  
78          if (types == null) {
79              types = super.getSystemTypes(forRdf);
80              // NonRdfSource gets the ldp:Resource and adds ldp:RDFSource types.
81              types.add(RDF_SOURCE_URI);
82          }
83  
84          return types;
85      }
86  
87      @Override
88      public RdfStream getTriples() {
89          // Remap the subject to the described resource
90          // TODO: With FCREPO-3819 and FCREPO-3820, this will no longer be necessary.
91          //       But existing sites might have RDF with NonRdfSourceDescription subjects, so leave it for now.
92          final Node describedID = createURI(this.getDescribedResource().getId());
93          final Stream<Triple> triples = super.getTriples().map(t ->
94                  new Triple(describedID, t.getPredicate(), t.getObject()));
95          return new DefaultRdfStream(describedID, triples);
96      }
97  }
