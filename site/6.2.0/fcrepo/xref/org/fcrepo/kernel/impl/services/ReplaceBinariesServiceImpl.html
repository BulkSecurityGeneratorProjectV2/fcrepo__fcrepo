32  /**
33   * Implementation of a service for replacing/updating binary resources
34   *
35   * @author bbpennel
36   */
37  @Component
38  public class ReplaceBinariesServiceImpl extends AbstractService implements ReplaceBinariesService {
39  
40      private static final Logger LOGGER = getLogger(ReplaceBinariesServiceImpl.class);
41  
42      @Inject
43      private PersistentStorageSessionManager psManager;
44  
45      @Inject
46      private NonRdfSourceOperationFactory factory;
47  
48      @Override
49      public void perform(final Transaction tx,
50                          final String userPrincipal,
51                          final FedoraId fedoraId,
52                          final String filename,
53                          final String contentType,
54                          final Collection<URI> digests,
55                          final InputStream contentBody,
56                          final long contentSize,
57                          final ExternalContent externalContent) {
58          try {
59              final PersistentStorageSession pSession = this.psManager.getSession(tx);
60  
61              String mimeType = contentType;
62              long size = contentSize;
63              final NonRdfSourceOperationBuilder builder;
64              if (externalContent == null || externalContent.isCopy()) {
65                  var contentInputStream = contentBody;
66                  if (externalContent != null) {
67                      LOGGER.debug("External content COPY '{}', '{}'", fedoraId, externalContent.getURL());
68                      contentInputStream = externalContent.fetchExternalContent();
69                  }
70  
71                  builder = factory.updateInternalBinaryBuilder(tx, fedoraId, contentInputStream);
72              } else {
73                  builder = factory.updateExternalBinaryBuilder(tx, fedoraId,
74                          externalContent.getHandling(),
75                          externalContent.getURI());
76  
77                  if (contentSize == -1L) {
78                      size = externalContent.getContentSize();
79                  }
80                  if (!digests.isEmpty()) {
81                      final var multiDigestWrapper = new MultiDigestInputStreamWrapper(
82                              externalContent.fetchExternalContent(),
83                              digests,
84                              Collections.emptyList());
85                      multiDigestWrapper.checkFixity();
86                  }
87              }
88  
89              if (externalContent != null && externalContent.getContentType() != null) {
90                  mimeType = externalContent.getContentType();
91              }
92  
93              builder.mimeType(mimeType)
94                     .contentSize(size)
95                     .filename(filename)
96                     .contentDigests(digests)
97                     .userPrincipal(userPrincipal);
98              final var replaceOp = builder.build();
99  
100             lockArchivalGroupResource(tx, pSession, fedoraId);
101             tx.lockResource(fedoraId);
102             // Descriptions are always under the binary, so just lock it.
103             tx.lockResource(fedoraId.asDescription());
104 
105             pSession.persist(replaceOp);
106             this.searchIndex.addUpdateIndex(tx, pSession.getHeaders(fedoraId, null));
107             recordEvent(tx, fedoraId, replaceOp);
108         } catch (final PersistentStorageException ex) {
109             throw new RepositoryRuntimeException(format("failed to replace binary %s",
110                     fedoraId), ex);
111         }
112     }
