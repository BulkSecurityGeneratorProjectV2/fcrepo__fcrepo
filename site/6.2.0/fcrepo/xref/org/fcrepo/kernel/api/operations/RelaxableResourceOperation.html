6   package org.fcrepo.kernel.api.operations;
7   
8   import java.time.Instant;
9   
10  /**
11   * Operation involving a resource with relaxable server managed properties
12   *
13   * @author bbpennel
14   */
15  public interface RelaxableResourceOperation extends ResourceOperation {
16      /**
17       * Get last modified by
18       *
19       * @return user that last modified the resource
20       */
21      String getLastModifiedBy();
22  
23      /**
24       * Get created by
25       *
26       * @return user that created the resource
27       */
28      String getCreatedBy();
29  
30      /**
31       * Get the timestamp the resource was last modified
32       *
33       * @return timestamp
34       */
35      Instant getLastModifiedDate();
36  
37      /**
38       * Get the timestamp the resource was created
39       *
40       * @return timestamp
41       */
42      Instant getCreatedDate();
43  }
