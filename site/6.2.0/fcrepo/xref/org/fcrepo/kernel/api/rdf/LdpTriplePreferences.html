6   package org.fcrepo.kernel.api.rdf;
7   
8   /**
9    * Kernel level API to hold the LdpPreferTag and internal logic decisions.
10   * @author whikloj
11   * @since 6.0.0
12   */
13  public interface LdpTriplePreferences {
14  
15      /**
16       * What the prefer tag choice is.
17       */
18      enum PreferChoice {
19          INCLUDE,
20          EXCLUDE,
21          SILENT
22      }
23  
24      /**
25       * @return Whether to display user rdf based on this preference tag and internal defaults.
26       */
27      boolean displayUserRdf();
28  
29      /**
30       * @return Whether to display membership triples based on this preference tag and internal defaults.
31       */
32      boolean displayMembership();
33  
34      /**
35       * @return Whether to display containment triples based on this preference tag and internal defaults.
36       */
37      boolean displayContainment();
38  
39      /**
40       * @return Whether to display inbound reference triples based on this preference tag and internal defaults.
41       */
42      boolean displayReferences();
43  
44      /**
45       * @return Whether to display contained resources' triples based on this preference tag and internal defaults.
46       */
47      boolean displayEmbed();
48  
49      /**
50       * @return Whether to display server managed triples based on this preference tag and internal defaults.
51       */
52      boolean displayServerManaged();
53  
54  }
