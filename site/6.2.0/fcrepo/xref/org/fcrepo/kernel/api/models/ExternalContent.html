6   package org.fcrepo.kernel.api.models;
7   
8   import java.io.InputStream;
9   import java.net.URI;
10  
11  /**
12   * Interface for the ExternalContent information class.
13   * @author bseeger
14   */
15  public interface ExternalContent {
16  
17      String PROXY = "proxy";
18      String REDIRECT = "redirect";
19      String COPY = "copy";
20  
21      /**
22       * Returns the content type located in the link header.
23       * @return content type if in Link header, else null
24       */
25      public String getContentType();
26  
27      /**
28       * Returns the size of the content located at the link header
29       * @return content size
30       */
31      public long getContentSize();
32  
33      /**
34       * Retrieve handling information
35       * @return a String containing the type of handling requested ["proxy", "copy" or "redirect"]
36       */
37      public String getHandling();
38  
39      /**
40       * Retrieve url in link header
41       * @return a String of the URL that was in the Link header
42       */
43      public String getURL();
44  
45      /**
46       * Retrieve URI in link header
47       * @return a URI to the external content
48       */
49      public URI getURI();
50  
51      /**
52       * Returns whether or not the handling parameter is "copy"
53       * @return boolean value representing whether or not the content handling is "copy"
54       */
55      public boolean isCopy();
56  
57      /**
58       * Returns whether or not the handling parameter is "redirect"
59       * @return boolean value representing whether or not the content handling is "redirect"
60       */
61      public boolean isRedirect();
62  
63      /**
64       * Returns whether or not the handling parameter is "proxy"
65       * @return boolean value representing whether or not the content handling is "proxy"
66       */
67      public boolean isProxy();
68  
69      /**
70       * Fetch the external content
71       * @return InputStream containing the external content
72       */
73      public InputStream fetchExternalContent();
74  }
